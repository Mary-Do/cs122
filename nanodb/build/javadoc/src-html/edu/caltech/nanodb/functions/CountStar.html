001package edu.caltech.nanodb.functions;
002
003
004import java.util.List;
005
006import edu.caltech.nanodb.expressions.Expression;
007import edu.caltech.nanodb.relations.ColumnType;
008import edu.caltech.nanodb.relations.SQLDataType;
009import edu.caltech.nanodb.relations.Schema;
010
011
012/**
013 * Created by donnie on 11/1/14.
014 */
015public class CountStar extends AggregateFunction {
016    int count = 0;
017
018
019    public CountStar() {
020        super(/* supportsDistinct */ false);
021    }
022
023
024    @Override
025    public void clearResult() {
026        count = 0;
027    }
028
029
030    @Override
031    public void addValue(Object value) {
032        count++;
033    }
034
035
036    @Override
037    public Object getResult() {
038        return Integer.valueOf(count);
039    }
040
041
042    @Override
043    public ColumnType getReturnType(List<Expression> args, Schema schema) {
044        if (args.size() != 1) {
045            throw new IllegalArgumentException(
046                "Count(*) aggregate function takes 1 argument; got " +
047                    args.size());
048        }
049
050        // When counting, the resulting aggregate column is always an integer
051        return new ColumnType(SQLDataType.INTEGER);
052    }
053}
