001package edu.caltech.nanodb.functions;
002
003
004import java.util.List;
005
006import edu.caltech.nanodb.expressions.Environment;
007import edu.caltech.nanodb.expressions.Expression;
008import edu.caltech.nanodb.expressions.ExpressionException;
009import edu.caltech.nanodb.expressions.TypeConverter;
010
011import edu.caltech.nanodb.relations.ColumnType;
012import edu.caltech.nanodb.relations.SQLDataType;
013import edu.caltech.nanodb.relations.Schema;
014
015
016/**
017 * Computes the arc-tangent of a single argument.  Returns <tt>NULL</tt> if
018 * argument is <tt>NULL</tt>.
019 * 
020 * @author emil
021 */
022public class ArcTan extends SimpleFunction {
023    @Override
024    public ColumnType getReturnType(List<Expression> args, Schema schema) {
025        return new ColumnType(SQLDataType.DOUBLE);
026    }
027
028
029    @Override
030    public Object evaluate(Environment env, List<Expression> args) {
031        if (args.size() != 1) {
032            throw new ExpressionException("Cannot call ATAN on " + args.size() 
033                    + " arguments");
034        }
035
036        Object argVal = args.get(0).evaluate(env);
037
038        if (argVal == null)
039            return null;
040
041        return Math.atan(TypeConverter.getDoubleValue(argVal));
042    }
043}
