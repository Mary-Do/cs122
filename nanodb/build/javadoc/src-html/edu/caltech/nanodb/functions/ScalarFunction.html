001package edu.caltech.nanodb.functions;
002
003
004import java.util.List;
005
006import edu.caltech.nanodb.expressions.Expression;
007import edu.caltech.nanodb.relations.ColumnType;
008import edu.caltech.nanodb.relations.Schema;
009
010
011/** This class represents functions that return a scalar value. */
012public abstract class ScalarFunction extends Function {
013    /**
014     * Returns the column type of the resulting column after applying the
015     * function. This generally depends on the column type of the inputs.
016     *
017     * @param args the arguments to the function call
018     * @param schema the schema of the table
019     * @return the column type of the resulting column
020     */
021    public abstract ColumnType getReturnType(List<Expression> args, Schema schema);
022}
