001package edu.caltech.nanodb.functions;
002
003
004import java.util.List;
005
006import edu.caltech.nanodb.expressions.Environment;
007import edu.caltech.nanodb.expressions.Expression;
008import edu.caltech.nanodb.expressions.ExpressionException;
009
010import edu.caltech.nanodb.relations.ColumnType;
011import edu.caltech.nanodb.relations.Schema;
012
013
014/**
015 * Returns the first non-<tt>NULL</tt> argument.  The function's return-type is
016 * reported to be whatever the first argument's type is.
017 * 
018 * @author emil
019 */
020public class Coalesce extends SimpleFunction {
021    @Override
022    public ColumnType getReturnType(List<Expression> args, Schema schema) {
023        if (args.size() < 1) {
024            throw new ExpressionException("Cannot call COALESCE on " +
025                args.size() + " arguments");
026        }
027
028        // Return the type of the first argument.
029        return args.get(0).getColumnInfo(schema).getType();
030    }
031
032    @Override
033    public Object evaluate(Environment env, List<Expression> args) {
034        if (args.size() < 1) {
035            throw new ExpressionException("Cannot call COALESCE on " +
036                args.size() + " arguments");
037        }
038        
039        for (Expression arg : args) {
040            Object val = arg.evaluate(env);
041            
042            if (val != null)
043                return val;
044        }
045        
046        return null;
047    }
048}
