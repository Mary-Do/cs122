001package edu.caltech.nanodb.functions;
002
003
004import java.util.List;
005
006import edu.caltech.nanodb.expressions.Environment;
007import edu.caltech.nanodb.expressions.Expression;
008import edu.caltech.nanodb.expressions.ExpressionException;
009
010
011/**
012 * Created by donnie on 12/5/13.
013 */
014public abstract class SimpleFunction extends ScalarFunction {
015    /**
016     * Evaluates the function.
017     *
018     * Should be called at runtime for every function call.
019     *
020     * @param env Environment, in which the arguments are evaluated
021     * @param args Arguments for this function
022     *
023     * @return The value of the function
024     *
025     * @throws ExpressionException when there is a problem with the evaluation
026     *                             like wrong number of arguments.
027     */
028    public abstract Object evaluate(Environment env, List<Expression> args);
029}
