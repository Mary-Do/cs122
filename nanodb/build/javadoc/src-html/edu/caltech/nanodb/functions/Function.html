001package edu.caltech.nanodb.functions;
002
003
004/**
005 * This is the root class of all kinds of functions in NanoDB.  The class
006 * hierarchy for functions is somewhat complicated; these are the important
007 * details:
008 * <ul>
009 *   <li>{@link ScalarFunction} is the parent class for any kind of function
010 *       that returns a scalar value
011 *       <ul>
012 *           <li>{@link SimpleFunction} is the parent class for functions that
013 *               take zero or more arguments, and compute and return a result
014 *           </li>
015 *           <li>{@link AggregateFunction} is the parent class for functions
016 *               that consume a (potentially large) collection of input values
017 *               and compute a single aggregated value from the collection
018 *           </li>
019 *       </ul>
020 *   </li>
021 *   <li>{@link TableFunction} is the parent class for table-returning
022 *       functions</li>
023 * </ul>
024 *
025 * Functions must support cloning because the implementation classes often
026 * carry their own internal state values, and clearly the same function being
027 * invoked in two different parts of the same query, or being invoked
028 * concurrently by two different queries, shouldn't have a single shared set
029 * of state values.  So, the simple thing to do is to just clone functions
030 * when they are retrieved from the {@link FunctionDirectory}.
031 */
032public abstract class Function implements Cloneable {
033
034    /**
035     * Creates a copy of expression.
036     */
037    @Override
038    public Object clone() throws CloneNotSupportedException {
039        return super.clone();
040    }
041}
