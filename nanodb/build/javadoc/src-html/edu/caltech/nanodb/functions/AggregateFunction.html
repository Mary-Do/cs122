001package edu.caltech.nanodb.functions;
002
003
004/**
005 * This class provides the general abstraction for aggregate functions.
006 */
007public abstract class AggregateFunction extends ScalarFunction {
008
009    /**
010     * If this flag is true, the aggregate function supports the
011     * <tt>DISTINCT</tt> keyword before its first argument.
012     */
013    private boolean supportsDistinct;
014
015
016    public AggregateFunction(boolean supportsDistinct) {
017        this.supportsDistinct = supportsDistinct;
018    }
019
020
021    public boolean supportsDistinct() {
022        return supportsDistinct;
023    }
024
025
026    /**
027     * Clears the aggregate function's current state so that the object can be
028     * reused to compute an aggregate on another set of input values.
029     */
030    public abstract void clearResult();
031
032
033    /**
034     * Adds a value to the aggregate function so that it can update its
035     * internal state.  Generally, aggregate functions ignore {@code null}
036     * inputs (which represent SQL <tt>NULL</tt> values) when computing their
037     * results.
038     *
039     * @param value the value to add to the aggregate function
040     */
041    public abstract void addValue(Object value);
042
043
044    /**
045     * Returns the aggregated result computed for this aggregate function.
046     * Generally, if aggregate functions receive no non-<tt>null</tt> inputs
047     * then they should produce a <tt>null</tt> result.  (<tt>COUNT</tt> is an
048     * exception to this rule, producing 0 in that case.)
049     *
050     * @return the result of the aggregate computation.
051     */
052    public abstract Object getResult();
053}
