054    /**
055     * Returns true if the transaction processing system is enabled, or false
056     * otherwise.
057     *
058     * @return true if the transaction processing system is enabled, or false
059     *         otherwise.
060     */
061    public static boolean isEnabled() {
062        // Transaction processing defaults to "off" until we actually
063        // turn it on.
064        return "on".equalsIgnoreCase(System.getProperty(PROP_TXNS, "off"));
065    }

068    private static class TransactionPropertyHandler implements PropertyHandler {
069
070        @Override
071        public Object getPropertyValue(String propertyName)
072                throws UnrecognizedPropertyException {
073
074            if (PROP_TXNS.equals(propertyName)) {
075                return isEnabled();
076            }
077            else {
078                throw new UnrecognizedPropertyException("No property named " +
079                        propertyName);
080            }
081        }
082
083        @Override
084        public void setPropertyValue(String propertyName, Object value)
085                throws UnrecognizedPropertyException, ReadOnlyPropertyException,
086                TypeCastException {
087
088            if (PROP_TXNS.equals(propertyName)) {
089                throw new ReadOnlyPropertyException(propertyName +
090                        " is read-only");
091            }
092            else {
093                throw new UnrecognizedPropertyException("No property named " +
094                        propertyName);
095            }
096        }
097    }

123    public TransactionManager(NanoDBServer server) {
124
125        this.server = server;
126        this.storageManager = server.getStorageManager();
127
128        BufferManager bufferManager = storageManager.getBufferManager();
129
130        // Register the transaction manager on the buffer manager so that we
131        // can enforce the write-ahead logging rule for evicted pages.
132        bufferManager.addObserver(this);
133
134        this.nextTxnID = new AtomicInteger();
135
136        walManager = new WALManager(storageManager, bufferManager);
137
138        // Register properties that the Transaction Manager exposes.
139        server.getPropertyRegistry().registerProperties(
140            new TransactionPropertyHandler(), PROP_TXNS);
141
142    }

145    /**
146     * This helper function initializes a brand new transaction-state file for
147     * the transaction manager to use for providing transaction atomicity and
148     * durability.
149     *
150     * @return a {@code DBFile} object for the newly created and initialized
151     *         transaction-state file.
152     *
153     * @throws IOException if the transaction-state file can't be created for
154     *         some reason.
155     */
156    private TransactionStatePage createTxnStateFile() throws IOException {
157        // Create a brand new transaction-state file for the Transaction Manager
158        // to use.
159
160        DBFile dbfTxnState;
161        dbfTxnState = storageManager.createDBFile(TXNSTATE_FILENAME,
162            DBFileType.TXNSTATE_FILE);
163
164        DBPage dbpTxnState = storageManager.loadDBPage(dbfTxnState, 0);
165        TransactionStatePage txnState = new TransactionStatePage(dbpTxnState);
166
167        // Set the "next transaction ID" value to an initial default.
168        txnState.setNextTransactionID(1);
169        nextTxnID.set(1);
170
171        // Set the "first LSN" and "next LSN values to initial defaults.
172        LogSequenceNumber lsn =
173            new LogSequenceNumber(0, WALManager.OFFSET_FIRST_RECORD);
174
175        txnState.setFirstLSN(lsn);
176        txnState.setNextLSN(lsn);
177        txnStateNextLSN = lsn;
178
179        storageManager.getBufferManager().writeDBFile(dbfTxnState, /* sync */ true);
180
181        return txnState;
182    }
183
184
185    private TransactionStatePage loadTxnStateFile() throws IOException {
186        DBFile dbfTxnState = storageManager.openDBFile(TXNSTATE_FILENAME);
187        DBPage dbpTxnState = storageManager.loadDBPage(dbfTxnState, 0);
188        TransactionStatePage txnState = new TransactionStatePage(dbpTxnState);
189
190        // Set the "next transaction ID" value properly.
191        nextTxnID.set(txnState.getNextTransactionID());
192
193        // Retrieve the "first LSN" and "next LSN values so we know the range of
194        // the write-ahead log that we need to apply for recovery.
195        txnStateNextLSN = txnState.getNextLSN();
196
197        return txnState;
198    }
199
200
201    private void storeTxnStateToFile() throws IOException {
202        DBFile dbfTxnState = storageManager.openDBFile(TXNSTATE_FILENAME);
203        DBPage dbpTxnState = storageManager.loadDBPage(dbfTxnState, 0);
204        TransactionStatePage txnState = new TransactionStatePage(dbpTxnState);
205
206        txnState.setNextTransactionID(nextTxnID.get());
207        txnState.setFirstLSN(walManager.getFirstLSN());
208        txnState.setNextLSN(txnStateNextLSN);
209
210        storageManager.getBufferManager().writeDBFile(dbfTxnState, /* sync */ true);
211    }

214    public void initialize() throws IOException {
215        if (!isEnabled())
216            throw new IllegalStateException("Transactions are disabled!");
217
218        // Read the transaction-state file so we can initialize the
219        // Transaction Manager.
220
221        TransactionStatePage txnState;
222        try {
223            txnState = loadTxnStateFile();
224        }
225        catch (FileNotFoundException e) {
226            // BUGBUG:  If we find any other files in the data directory, we
227            //          really should fail initialization, because the old files
228            //          may have been created without transaction processing...
229
230            logger.info("Couldn't find transaction-state file " +
231                TXNSTATE_FILENAME + ", creating.");
232
233            txnState = createTxnStateFile();
234        }
235
236        // Perform recovery, and get the new "first LSN" value
237
238        LogSequenceNumber firstLSN = txnState.getFirstLSN();
239        LogSequenceNumber nextLSN = txnState.getNextLSN();
240        logger.debug(String.format("Txn State has FirstLSN = %s, NextLSN = %s",
241            firstLSN, nextLSN));
242
243        RecoveryInfo recoveryInfo = walManager.doRecovery(firstLSN, nextLSN);
244
245        // Set the "next transaction ID" value based on what recovery found
246        int recNextTxnID = recoveryInfo.maxTransactionID + 1;
247        if (recNextTxnID != -1 && recNextTxnID > nextTxnID.get()) {
248            logger.info("Advancing NextTransactionID from " +
249                nextTxnID.get() + " to " + recNextTxnID);
250            nextTxnID.set(recNextTxnID);
251        }
252
253        // Update and sync the transaction state if any changes were made.
254        storeTxnStateToFile();
255
256        // Register the component that manages indexes when tables are modified.
257        server.getEventDispatcher().addCommandEventListener(
258            new TransactionStateUpdater(this));
259    }

284    public void startTransaction(boolean userStarted) throws TransactionException {
285        SessionState state = SessionState.get();
286        TransactionState txnState = state.getTxnState();
287
288        if (txnState.isTxnInProgress())
289            throw new IllegalStateException("A transaction is already in progress!");
290
291        int txnID = getAndIncrementNextTxnID();
292        txnState.setTransactionID(txnID);
293        txnState.setUserStartedTxn(userStarted);
294
295        logger.debug("Starting transaction with ID " + txnID +
296            (userStarted ? " (user-started)" : ""));
297
298        // Don't record a "start transaction" WAL record until the transaction
299        // actually writes to something in the database.
300    }

303    public void recordPageUpdate(DBPage dbPage) throws IOException {
304        if (!dbPage.isDirty()) {
305            logger.debug("Page reports it is not dirty; not logging update.");
306            return;
307        }
308
309        logger.debug("Recording page-update for page " + dbPage.getPageNo() +
310            " of file " + dbPage.getDBFile());
311
312        TransactionState txnState = SessionState.get().getTxnState();
313        if (!txnState.hasLoggedTxnStart()) {
314            walManager.writeTxnRecord(WALRecordType.START_TXN);
315            txnState.setLoggedTxnStart(true);
316        }
317
318        walManager.writeUpdatePageRecord(dbPage);
319        dbPage.syncOldPageData();
320    }

323    public void commitTransaction() throws TransactionException {
324        SessionState state = SessionState.get();
325        TransactionState txnState = state.getTxnState();
326
327        if (!txnState.isTxnInProgress()) {
328            // The user issued a COMMIT without starting a transaction!
329
330            state.getOutputStream().println(
331                "No transaction is currently in progress.");
332
333            return;
334        }
335
336        int txnID = txnState.getTransactionID();
337
338        if (txnState.hasLoggedTxnStart()) {
339            // Must record the transaction as committed to the write-ahead log.
340            // Then, we must force the WAL to include this commit record.
341            try {
342                walManager.writeTxnRecord(WALRecordType.COMMIT_TXN);
343                forceWAL(walManager.getNextLSN());
344            }
345            catch (IOException e) {
346                throw new TransactionException("Couldn't commit transaction " +
347                    txnID + "!", e);
348            }
349        }
350        else {
351            logger.debug("Transaction " + txnID + " has made no changes; not " +
352                "recording transaction-commit to WAL.");
353        }
354
355        // Now that the transaction is successfully committed, clear the current
356        // transaction state.
357        logger.debug("Transaction completed, resetting transaction state.");
358        txnState.clear();
359    }

362    public void rollbackTransaction() throws TransactionException {
363        SessionState state = SessionState.get();
364        TransactionState txnState = state.getTxnState();
365
366        if (!txnState.isTxnInProgress()) {
367            // The user issued a ROLLBACK without starting a transaction!
368
369            state.getOutputStream().println(
370                "No transaction is currently in progress.");
371
372            return;
373        }
374
375        int txnID = txnState.getTransactionID();
376
377        if (txnState.hasLoggedTxnStart()) {
378            // Must rollback the transaction using the write-ahead log.
379            try {
380                walManager.rollbackTransaction();
381            }
382            catch (IOException e) {
383                throw new TransactionException(
384                    "Couldn't rollback transaction " + txnID + "!", e);
385            }
386        }
387        else {
388            logger.debug("Transaction " + txnID + " has made no changes; not " +
389                "recording transaction-rollback to WAL.");
390        }
391
392        // Now that the transaction is successfully rolled back, clear the
393        // current transaction state.
394        logger.debug("Transaction completed, resetting transaction state.");
395        txnState.clear();
396    }
