113    /**
114     * Returns the first tuple in this table file, or <tt>null</tt> if
115     * there are no tuples in the file.
116     */
117    @Override
118    public Tuple getFirstTuple() throws IOException {
119        HeapFilePageTuple first = null;
120        try {
121            // Scan through the data pages until we hit the end of the table
122            // file.  It may be that the first run of data pages is empty,
123            // so just keep looking until we hit the end of the file.
124
125            // Header page is page 0, so first data page is page 1.
126page_scan:  // So we can break out of the outer loop from inside the inner one
127            for (int iPage = 1; /* nothing */ ; iPage++) {
128                // Look for data on this page.
129                DBPage dbPage = storageManager.loadDBPage(dbFile, iPage);
130                int numSlots = DataPage.getNumSlots(dbPage);
131                for (int iSlot = 0; iSlot < numSlots; iSlot++) {
132                    // Get the offset of the tuple in the page.  If it's 0 then
133                    // the slot is empty, and we skip to the next slot.
134                    int offset = DataPage.getSlotValue(dbPage, iSlot);
135                    if (offset == DataPage.EMPTY_SLOT)
136                        continue;
137
138                    // This is the first tuple in the file.  Build up the
139                    // HeapFilePageTuple object and return it.
140                    first = new HeapFilePageTuple(schema, dbPage, iSlot, offset);
141                    break page_scan;
142                }
143            }
144        }
145        catch (EOFException e) {
146            // We ran out of pages.  No tuples in the file!
147            logger.debug("No tuples in table-file " + dbFile +
148                         ".  Returning null.");
149        }
150
151        return first;
152    }

155    /**
156     * Returns the tuple corresponding to the specified file pointer.  This
157     * method is used by many other operations in the database, such as
158     * indexes.
159     *
160     * @throws InvalidFilePointerException if the specified file-pointer
161     *         doesn't actually point to a real tuple.
162     */
163    @Override
164    public Tuple getTuple(FilePointer fptr)
165        throws InvalidFilePointerException, IOException {
166
167        DBPage dbPage;
168        try {
169            // This could throw EOFException if the page doesn't actually exist.
170            dbPage = storageManager.loadDBPage(dbFile, fptr.getPageNo());
171        }
172        catch (EOFException eofe) {
173            throw new InvalidFilePointerException("Specified page " +
174                fptr.getPageNo() + " doesn't exist in file " +
175                dbFile.getDataFile().getName(), eofe);
176        }
177
178        // The file-pointer points to the slot for the tuple, not the tuple itself.
179        // So, we need to look up that slot's value to get to the tuple data.
180
181        int slot;
182        try {
183            slot = DataPage.getSlotIndexFromOffset(dbPage, fptr.getOffset());
184        }
185        catch (IllegalArgumentException iae) {
186            throw new InvalidFilePointerException(iae);
187        }
188
189        // Pull the tuple's offset from the specified slot, and make sure
190        // there is actually a tuple there!
191
192        int offset = DataPage.getSlotValue(dbPage, slot);
193        if (offset == DataPage.EMPTY_SLOT) {
194            throw new InvalidFilePointerException("Slot " + slot +
195                " on page " + fptr.getPageNo() + " is empty.");
196        }
197
198        return new HeapFilePageTuple(schema, dbPage, slot, offset);
199    }

202    /**
203     * Returns the tuple that follows the specified tuple,
204     * or <tt>null</tt> if there are no more tuples in the file.
205     **/
206    @Override
207    public Tuple getNextTuple(Tuple tup) throws IOException {
208
209        /* Procedure:
210         *   1)  Get slot index of current tuple.
211         *   2)  If there are more slots in the current page, find the next
212         *       non-empty slot.
213         *   3)  If we get to the end of this page, go to the next page
214         *       and try again.
215         *   4)  If we get to the end of the file, we return null.
216         */
217
218        if (!(tup instanceof HeapFilePageTuple)) {
219            throw new IllegalArgumentException(
220                "Tuple must be of type HeapFilePageTuple; got " + tup.getClass());
221        }
222        HeapFilePageTuple ptup = (HeapFilePageTuple) tup;
223
224        DBPage dbPage = ptup.getDBPage();
225        DBFile dbFile = dbPage.getDBFile();
226
227        HeapFilePageTuple nextTup = null;
228
229        int nextSlot = ptup.getSlot() + 1;
230
231page_scan:  // So we can break out of the outer loop from inside the inner loop.
232        while (true) {
233            int numSlots = DataPage.getNumSlots(dbPage);
234
235            while (nextSlot < numSlots) {
236                int nextOffset = DataPage.getSlotValue(dbPage, nextSlot);
237                if (nextOffset != DataPage.EMPTY_SLOT) {
238                    // Pins the page a second time.
239                    nextTup = new HeapFilePageTuple(schema, dbPage, nextSlot,
240                                                    nextOffset);
241                    break page_scan;
242                }
243
244                nextSlot++;
245            }
246
247            // If we got here then we reached the end of this page with no
248            // tuples.  Go on to the next data-page, and start with the first
249            // tuple in that page.
250
251            try {
252                dbPage = storageManager.loadDBPage(dbFile, dbPage.getPageNo() + 1);
253                nextSlot = 0;
254            }
255            catch (EOFException e) {
256                // Hit the end of the file with no more tuples.  We are done
257                // scanning.
258                break;
259            }
260        }
261
262        return nextTup;
263    }

266    /**
267     * Adds the specified tuple into the table file.  A new
268     * <tt>HeapFilePageTuple</tt> object corresponding to the tuple is returned.
269     *
270     * @review (donnie) This could be made a little more space-efficient.
271     *         Right now when computing the required space, we assume that we
272     *         will <em>always</em> need a new slot entry, whereas the page may
273     *         contain empty slots.  (Note that we don't always create a new
274     *         slot when adding a tuple; we will reuse an empty slot.  This
275     *         inefficiency is simply in estimating the size required for the
276     *         new tuple.)
277     */
278    @Override
279    public Tuple addTuple(Tuple tup) throws IOException {
280
281        /*
282         * Check to see whether any constraints are violated by
283         * adding this tuple
284         *
285         * Find out how large the new tuple will be, so we can find a page to
286         * store it.
287         *
288         * Find a page with space for the new tuple.
289         *
290         * Generate the data necessary for storing the tuple into the file.
291         */
292
293        int tupSize = PageTuple.getTupleStorageSize(schema, tup);
294        logger.debug("Adding new tuple of size " + tupSize + " bytes.");
295
296        // Sanity check:  Make sure that the tuple would actually fit in a page
297        // in the first place!
298        // The "+ 2" is for the case where we need a new slot entry as well.
299        if (tupSize + 2 > dbFile.getPageSize()) {
300            throw new IOException("Tuple size " + tupSize +
301                " is larger than page size " + dbFile.getPageSize() + ".");
302        }
303
304        // Search for a page to put the tuple in.  If we hit the end of the
305        // data file, create a new page.
306        int pageNo = 1;
307        DBPage dbPage = null;
308        while (true) {
309            // Try to load the page without creating a new one.
310            try {
311                dbPage = storageManager.loadDBPage(dbFile, pageNo);
312            }
313            catch (EOFException eofe) {
314                // Couldn't load the current page, because it doesn't exist.
315                // Break out of the loop.
316                logger.debug("Reached end of data file without finding " +
317                             "space for new tuple.");
318                break;
319            }
320
321            int freeSpace = DataPage.getFreeSpaceInPage(dbPage);
322
323            logger.trace(String.format("Page %d has %d bytes of free space.",
324                         pageNo, freeSpace));
325
326            // If this page has enough free space to add a new tuple, break
327            // out of the loop.  (The "+ 2" is for the new slot entry we will
328            // also need.)
329            if (freeSpace >= tupSize + 2) {
330                logger.debug("Found space for new tuple in page " + pageNo + ".");
331                break;
332            }
333
334            // If we reached this point then the page doesn't have enough
335            // space, so go on to the next data page.
336            dbPage = null;  // So the next section will work properly.
337            pageNo++;
338        }
339
340        if (dbPage == null) {
341            // Try to create a new page at the end of the file.  In this
342            // circumstance, pageNo is *just past* the last page in the data
343            // file.
344            logger.debug("Creating new page " + pageNo + " to store new tuple.");
345            dbPage = storageManager.loadDBPage(dbFile, pageNo, true);
346            DataPage.initNewPage(dbPage);
347        }
348
349        int slot = DataPage.allocNewTuple(dbPage, tupSize);
350        int tupOffset = DataPage.getSlotValue(dbPage, slot);
351
352        logger.debug(String.format(
353            "New tuple will reside on page %d, slot %d.", pageNo, slot));
354
355        HeapFilePageTuple pageTup =
356            HeapFilePageTuple.storeNewTuple(schema, dbPage, slot, tupOffset, tup);
357
358        DataPage.sanityCheck(dbPage);
359
360        return pageTup;
361    }

365    /**
366     * @review (donnie) This method will fail if a tuple is modified in a way
367     *         that requires more space than is currently available in the data
368     *         page.  One solution would be to move the tuple to a different
369     *         page and then perform the update, but that would cause all kinds
370     *         of additional issues.  So, if the page runs out of data, oh well.
371     */
372    @Override
373    public void updateTuple(Tuple tup, Map<String, Object> newValues)
374        throws IOException {
375
376        if (!(tup instanceof HeapFilePageTuple)) {
377            throw new IllegalArgumentException(
378                "Tuple must be of type HeapFilePageTuple; got " + tup.getClass());
379        }
380        HeapFilePageTuple ptup = (HeapFilePageTuple) tup;
381
382        for (Map.Entry<String, Object> entry : newValues.entrySet()) {
383            String colName = entry.getKey();
384            Object value = entry.getValue();
385
386            int colIndex = schema.getColumnIndex(colName);
387            ptup.setColumnValue(colIndex, value);
388        }
389
390        DBPage dbPage = ptup.getDBPage();
391        DataPage.sanityCheck(dbPage);
392    }

396    @Override
397    public void deleteTuple(Tuple tup) throws IOException {
398
399        if (!(tup instanceof HeapFilePageTuple)) {
400            throw new IllegalArgumentException(
401                "Tuple must be of type HeapFilePageTuple; got " + tup.getClass());
402        }
403        HeapFilePageTuple ptup = (HeapFilePageTuple) tup;
404
405        DBPage dbPage = ptup.getDBPage();
406        DataPage.deleteTuple(dbPage, ptup.getSlot());
407        DataPage.sanityCheck(dbPage);
408
409        // Note that we don't invalidate the page-tuple when it is deleted,
410        // so that the tuple can still be unpinned, etc.
411    }
