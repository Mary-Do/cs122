015/**
016 * This class represents a single page in a database file.  The page's
017 * (zero-based) index in the file, and whether the page has been changed in
018 * memory, are tracked by the object.
019 * <p>
020 * Database pages do not provide any locking mechanisms to guard against
021 * concurrent access.  Locking must be managed at a level above what this class
022 * provides.
023 * <p>
024 * The class provides methods to read and write a wide range of data types.
025 * Multibyte values are stored in big-endian format, with the most significant
026 * byte (MSB) stored at the lowest index, and the least significant byte (LSB)
027 * stored at the highest index.  (This is also the network byte order specified
028 * by the Internet Protocol.)
029 *
030 * @see PageReader
031 * @see PageWriter
032 *
033 * @design (Donnie) It is very important that the page is marked dirty
034 *         <em>before</em> any changes are made, because this is the point
035 *         when the old version of the page data is copied before changes
036 *         are made.  Additionally, the page's data must not be manipulated
037 *         separately from the methods provided by this class, or else the
038 *         old version of the page won't be recorded properly.
039 */
040public class DBPage implements Pinnable, AutoCloseable {
041
042    /** A logging object for reporting anything interesting that happens. */
043    private static Logger logger = Logger.getLogger(DBPage.class);
044
045
046    /**
047     * The buffer manager is used by this class to request buffer space when
048     * needed, so that a maximum cap can be placed on memory usage.
049     */
050    private BufferManager bufferManager;
051
052
053    /** A reference to the database file that this page is from. */
054    private DBFile dbFile;
055
056    /**
057     * The page number in the table file.  A value of 0 is the first page
058     * in the file.
059     */
060    private int pageNo;
061
062
063    /**
064     * The pin-count of this page.  When nonzero, the page is not allowed to be
065     * flushed from the buffer manager since the page is being used by at least
066     * one session.
067     */
068    private int pinCount;
069
070
071    /** This flag is true if this page has been modified in memory. */
072    private boolean dirty;

074
075    /**
076     * For dirty pages, this field is set to the Log Sequence Number of the
077     * write-ahead log record corresponding to the most recent write to the
078     * page.  When the page is being flushed back to disk, the write-ahead log
079     * must be written to at least this point, or else the write-ahead logging
080     * rule will be violated.
081     */
082    private LogSequenceNumber pageLSN;

084
085    /** The actual data for the table-page. */
086    private byte[] pageData;
087
088
089    /**
090     * When the page is marked dirty, this gets set to the original version of
091     * the page, so that we can properly record changes to the write-ahead log.
092     */
093    private byte[] oldPageData;

096    /**
097     * Constructs a new, empty table-page for the specified table file.
098     * Note that the page data is not loaded into the object; that must be
099     * done in a separate step.
100     *
101     * @param dbFile The database file that this page is contained within.
102     *
103     * @param pageNo The page number within the database file.
104     *
105     * @throws NullPointerException if <tt>dbFile</tt> is <tt>null</tt>
106     *
107     * @throws IllegalArgumentException if <tt>pageNo</tt> is negative
108     */
109    public DBPage(BufferManager bufferManager, DBFile dbFile, int pageNo) {
110        if (bufferManager == null)
111            throw new IllegalArgumentException("bufferManager cannot be null");
112
113        if (dbFile == null)
114            throw new IllegalArgumentException("dbFile cannot be null");
115
116        if (pageNo < 0) {
117            throw new IllegalArgumentException("pageNo must be >= 0 (got " +
118                pageNo + ")");
119        }
120
121        this.bufferManager = bufferManager;
122        this.dbFile = dbFile;
123        this.pageNo = pageNo;
124        pinCount = 0;
125        dirty = false;
126        pageLSN = null;
127
128        // This operation could fail with an IOException, because more
129        // space must be allocated to keep track of the original page data,
130        // that might cause the buffer manager to swap out some other pages
131        // to disk.  Since all the existing code wouldn't compile if an
132        // IOException is thrown, we just wrap it with a RuntimeException.
133        // TODO:  Come up with a better approach?!  Maybe a custom exception??
134        try {
135            pageData = bufferManager.allocBuffer(dbFile.getPageSize());
136        }
137        catch (IOException e) {
138            throw new RuntimeException(e);
139        }
140
141        oldPageData = null;
142    }

188    @Override
189    public void pin() {
190        pinCount++;
191
192        // To facilitate debugging of pinned-page leaks!
193        bufferManager.recordPagePinned(this);
194    }
195
196
197    @Override
198    public void unpin() {
199        if (pinCount <= 0) {
200            throw new IllegalStateException(
201                "pinCount is not positive (value is " + pinCount + ")");
202        }
203
204        // To facilitate debugging of pinned-page leaks!
205        bufferManager.recordPageUnpinned(this);
206
207        pinCount--;
208
209        // Don't do anything else to the data page, because at this point,
210        // it may still contain dirty data that needs to be flushed to disk
211        // when the page is evicted.
212    }

217        return pinCount;
218    }

223        return (pinCount > 0);

251    /**
252     * For a dirty page, this method copies the "new page data" into the "old
253     * page data" so that they are the same.  This is necessary when changes are
254     * recorded to the write-ahead log; since the changes are reflected in the
255     * WAL, it's not necessary to represent the deltas anymore.
256     *
257     * @throws IllegalStateException if the page is not currently marked dirty
258     */
259    public void syncOldPageData() {
260        if (oldPageData == null)
261            throw new IllegalStateException("No old page data to sync");
262
263        System.arraycopy(pageData, 0, oldPageData, 0, pageData.length);
264    }

267    /**
268     * Returns true if the page's data has been changed in memory; false
269     * otherwise.
270     *
271     * @return true if the page's data has been changed in memory
272     */
273    public boolean isDirty() {
274        return dirty;
275    }
276
277    /**
278     * Sets the dirty flag to true or false, indicating whether the page's data
279     * has or has not been changed in memory.
280     *
281     * @param dirty the dirty flag; true if the page's data is dirty, or false
282     *        otherwise
283     */
284    public void setDirty(boolean dirty) {
285        if (!this.dirty && dirty) {
286            // Page is being changed from clean to dirty.  Duplicate the current
287            // data so that we have it when updating the write-ahead log.
288
289            // This operation could fail with an IOException, because more
290            // space must be allocated to keep track of the original page data,
291            // that might cause the buffer manager to swap out some other pages
292            // to disk.  Since all the existing code wouldn't compile if an
293            // IOException is thrown, we just wrap it with a RuntimeException.
294            // TODO:  Come up with a better approach?!  Maybe a custom exception??
295            try {
296                oldPageData = bufferManager.allocBuffer(pageData.length);
297                System.arraycopy(pageData, 0, oldPageData, 0, pageData.length);
298            }
299            catch (IOException e) {
300                throw new RuntimeException(e);
301            }
302        }
303        else if (this.dirty && !dirty) {
304            // Page is being changed from dirty to clean.  Clear out the old
305            // page data since we don't need it anymore.
306            bufferManager.releaseBuffer(oldPageData);
307            oldPageData = null;
308
309            // Clear out the page-LSN value as well.
310            pageLSN = null;
311        }
312
313        this.dirty = dirty;
314    }

318        return pageLSN;

327    /**
328     * This method makes the {@code DBPage} invalid by clearing all of its
329     * internal references.  It is used by the Buffer Manager when a page is
330     * removed from the cache so that no other database code will continue to
331     * try to use the page.
332     */
333    public void invalidate() {
334        bufferManager.recordPageInvalidated(this);
335
336        dbFile = null;
337        pageNo = -1;
338
339        if (pageData != null) {
340            bufferManager.releaseBuffer(pageData);
341            pageData = null;
342        }
343
344        if (oldPageData != null) {
345            bufferManager.releaseBuffer(oldPageData);
346            oldPageData = null;
347        }
348    }

533    /**
534     * Reads and returns a signed short from the specified position.  The
535     * value is returned as a <tt>short</tt>.
536     *
537     * @param position the location in the page to start reading the value from
538     *
539     * @return the signed short value
540     */
541    public short readShort(int position) {
542        // Don't chop off high-order bits.  When byte is cast to int, the sign
543        // will be extended, so if original byte is negative, the resulting
544        // int will be too.
545        int value = ((pageData[position++]       ) <<  8)
546                  | ((pageData[position  ] & 0xFF)      );
547
548        return (short) value;
549    }

552     * Writes a (signed or unsigned) short to the specified position.  The short
553     * value is specified as an integer for the sake of convenience
554     * (specifically to avoid having to cast an argument to a short value), but
555     * the input is also truncated down to the low 16 bits.

558     *
559     * @param value the byte value

562        setDirty(true);
563
564        pageData[position++] = (byte) (0xFF & (value >> 8));

718    /**
719     * This method reads and returns a variable-length string whose maximum
720     * length is 255 bytes.  The string is expected to be in US-ASCII
721     * encoding, so multibyte characters are not supported.
722     * <p>
723     * The string's data format is expected to be a single unsigned byte
724     * <em>b</em> specifying the string's length, followed by <em>b</em> more
725     * bytes consisting of the string value itself.
726     *
727     * @param position the location in the page to start reading the value from
728     *
729     * @return a string object containing the stored value, up to a maximum of
730     *         255 characters in length
731     */
732    public String readVarString255(int position) {
733        int len = readUnsignedByte(position++);
734
735        String str = null;
736
737        try {
738            str = new String(pageData, position, len, "US-ASCII");
739        }
740        catch (UnsupportedEncodingException e) {
741            // According to the Java docs, the US-ASCII character-encoding is
742            // required to be supported by all JVMs.  So, this is not supposed
743            // to happen.
744            logger.error("The unthinkable has happened:  " + e);
745        }
746
747        return str;
748    }

750    /**
751     * This method stores a variable-length string whose maximum length is
752     * 255 bytes.  The string is expected to be in US-ASCII encoding, so
753     * multibyte characters are not supported.
754     * <p>
755     * The string is stored as a single unsigned byte <em>b</em> specifying the
756     * string's length, followed by <em>b</em> more bytes consisting of the
757     * string value itself.
758     *
759     * @param position the location in the page to start writing the value to
760     *

866     * This method reads and returns a string whose length is fixed at a
867     * constant size.  The string is expected to be in US-ASCII encoding, so
868     * multibyte characters are not supported.
869     * <p>
870     * Strings shorter than the specified length are padded with 0 bytes at the end of the string, and
871     * this padding is removed when the string is read.
872     *
873     *
874     * The string's characters are stored starting with the specified position.
875     * If the string is shorter than the fixed length then the data is expected
876     * to be terminated with a <tt>\\u0000</tt> (i.e. <tt>NUL</tt>) value.  (If
877     * the string is exactly the given length then no string terminator is
878     * expected.)  <b>The implication of this storage format is that embedded
879     * <tt>NUL</tt> characters are not allowed with this storage format.</b>
880     *
881     * @param position the location in the page to start reading the value from
882     *
883     * @param len the length of the fixed-size string
884     *
885     * @return a string object containing the stored value, up to a maximum of
886     *         <tt>len</tt> characters in length
887     */
888    public String readFixedSizeString(int position, int len) {
889        String str = null;
890
891        // Fixed-size strings are padded with 0-bytes, so trim these off the
892        // end of the string value.
893        while (len > 0 && pageData[position + len - 1] == 0)
894            len--;
895
896        try {
897            str = new String(pageData, position, len, "US-ASCII");
898        }
899        catch (UnsupportedEncodingException e) {
900            // According to the Java docs, the US-ASCII character-encoding is
901            // required to be supported by all JVMs.  So, this is not supposed
902            // to happen.
903            logger.error("The unthinkable has happened:  " + e);
904        }
905
906        return str;
907    }

917     * exactly the given length then no string terminator is stored.  <b>The
918     * implication of this storage format is that embedded <tt>NUL</tt>
919     * characters are not allowed with this storage format.</b>
920     *
921     * @param position the location in the page to start writing the value to
922     *
923     * @param value the string object containing the data to store
924     *
925     * @param len the number of bytes used to store the string field
926     *
927     * @throws NullPointerException if <tt>value</tt> is <tt>null</tt>
928     *
929     * @throws IllegalArgumentException if the input string is longer than
930     *         <tt>len</tt> characters
931     */
932    public void writeFixedSizeString(int position, String value, int len) {
933        byte[] bytes;
934
935        try {
936            bytes = value.getBytes("US-ASCII");
937        }
938        catch (UnsupportedEncodingException e) {
939            // According to the Java docs, the US-ASCII character-encoding is
940            // required to be supported by all JVMs.  So, this is not supposed
941            // happen.
942            logger.error("The unthinkable has happened!", e);
943            throw new RuntimeException("The unthinkable has happened!", e);

960     * This method provides a higher-level wrapper around the other methods in
961     * the <tt>DBPage</tt> class, allowing object-values to be read, as long
962     * as the data-type is provided along with the object.
963     *
964     * @param position the location in the page to start reading the value from
965     *
966     * @param colType the type of the value being read
967     *
968     * @return the object read from the data
969     *
970     * @throws NullPointerException if <tt>colType</tt> or <tt>value</tt> is
971     *         <tt>null</tt>
972     *
973     * @throws IllegalArgumentException if the input string is longer than
974     *         <tt>len</tt> characters
975     */
976    public Object readObject(int position, ColumnType colType) {
977        Object value = null;
978
979        switch (colType.getBaseType()) {
980
981        case INTEGER:
982            value = Integer.valueOf(readInt(position));
983            break;
984
985        case SMALLINT:
986            value = Short.valueOf(readShort(position));
987            break;
988
989        case BIGINT:
990            value = Long.valueOf(readLong(position));
991            break;
992
993        case TINYINT:
994            value = Byte.valueOf(readByte(position));
995            break;
996
997        case FLOAT:
998            value = Float.valueOf(readFloat(position));
999            break;
1000
1001        case DOUBLE:
1002            value = Double.valueOf(readDouble(position));
1003            break;
1004
1005        case CHAR:
1006            value = readFixedSizeString(position, colType.getLength());
1007            break;
1008
1009        case VARCHAR:
1010            value = readVarString65535(position);
1011            break;
1012
1013        case FILE_POINTER:
1014            value = new FilePointer(readUnsignedShort(position),
1015                                    readUnsignedShort(position + 2));
1016            break;
1017
1018        default:
1019            throw new UnsupportedOperationException(
1020                "Cannot currently read type " + colType.getBaseType());
1021        }
1022
1023        return value;
1024    }

1028     * This method provides a higher-level wrapper around the other methods in
1029     * the <tt>DBPage</tt> class, allowing object-values to be stored, as long
1030     * as the object isn't <tt>null</tt> and the data-type is provided along

1032     *
1033     * @param position the location in the page to start writing the value to
1034     *
1035     * @param colType the type of the value being stored
1036     *
1037     * @param value the object containing the data to store
1038     *
1039     * @return the total number of bytes written in the operation; i.e. this is
1040     *         the amount that the position is advanced.
1041     *
1042     * @throws NullPointerException if <tt>colType</tt> or <tt>value</tt> is
1043     *         <tt>null</tt>
1044     *
1045     * @throws IllegalArgumentException if the input string is longer than
1046     *         <tt>len</tt> characters
1047     */
1048    public int writeObject(int position, ColumnType colType, Object value) {
1049
1050        if (colType == null)
1051            throw new NullPointerException("colType cannot be null");
1052
1053        if (value == null)
1054            throw new NullPointerException("value cannot be null");
1055
1056        int dataSize;
1057
1058        // This code relies on Java autoboxing.  Go, syntactic sugar.
1059        switch (colType.getBaseType()) {
1060
1061        case INTEGER:
1062            {
1063                int iVal = TypeConverter.getIntegerValue(value);
1064                writeInt(position, iVal);
1065                dataSize = 4;
1066                break;
1067            }
1068
1069        case SMALLINT:
1070            {
1071                short sVal = TypeConverter.getShortValue(value);
1072                writeShort(position, sVal);
1073                dataSize = 2;
1074                break;
1075            }
1076
1077        case BIGINT:
1078            {
1079                long lVal = TypeConverter.getLongValue(value);
1080                writeLong(position, lVal);
1081                dataSize = 8;
1082                break;
1083            }
1084
1085        case TINYINT:
1086            {
1087                byte bVal = TypeConverter.getByteValue(value);
1088                writeByte(position, bVal);
1089                dataSize = 1;
1090                break;
1091            }
1092
1093        case FLOAT:
1094            {
1095                float fVal = TypeConverter.getFloatValue(value);
1096                writeFloat(position, fVal);
1097                dataSize = 4;
1098                break;
1099            }
1100
1101        case DOUBLE:
1102            {
1103                double dVal = TypeConverter.getDoubleValue(value);
1104                writeDouble(position, dVal);
1105                dataSize = 8;
1106                break;
1107            }
1108
1109        case CHAR:
1110            {
1111                String strVal = TypeConverter.getStringValue(value);
1112                writeFixedSizeString(position, strVal, colType.getLength());
1113                dataSize = colType.getLength();
1114                break;
1115            }
1116
1117        case VARCHAR:
1118            {
1119                String strVal = TypeConverter.getStringValue(value);
1120                writeVarString65535(position, strVal);
1121                dataSize = 2 + strVal.length();
1122                break;
1123            }
1124
1125        case FILE_POINTER:
1126            {
1127                FilePointer fptr = (FilePointer) value;
1128                writeShort(position, fptr.getPageNo());
1129                writeShort(position + 2, fptr.getOffset());
1130                dataSize = 4;
1131                break;
1132            }
1133
1134        default:
1135            throw new UnsupportedOperationException(
1136                "Cannot currently store type " + colType.getBaseType());
1137        }
1138
1139        return dataSize;
1140    }

1175    /**
1176     * This helper method returns a formatted string describing all changes
1177     * made to the page's contents; that is, the differences between the
1178     * {@link #pageData} and the {@link #oldPageData} byte-arrays.  The output
1179     * is formatted to inclue rows of 32 bytes, and only includes rows where
1180     * the data between old and new pages are actually different.
1181     *
1182     * @return a formatted string describing all changes made to the page's
1183     *         contents
1184     *
1185     * @throws IllegalStateException if the method is called on a non-dirty
1186     *         page
1187     */
1188    public String getChangesAsString() {
1189        if (!dirty)
1190            throw new IllegalStateException("Page is not dirty");
1191
1192        StringBuilder buf = new StringBuilder();
1193
1194        int i = 0;
1195        int pageSize = dbFile.getPageSize();
1196        while (i < pageSize) {
1197            boolean same = true;
1198            for (int j = 0; j < 32; j++) {
1199                if (oldPageData[i + j] != pageData[i + j]) {
1200                    same = false;
1201                    break;
1202                }
1203            }
1204
1205            if (!same) {
1206                buf.append(String.format("0x%04X OLD: ", i));
1207                for (int j = 0; j < 32; j++)
1208                    buf.append(String.format(" %02X", oldPageData[i + j]));
1209                buf.append('\n');
1210
1211                buf.append(String.format("0x%04X NEW: ", i));
1212                for (int j = 0; j < 32; j++) {
1213                    if (pageData[i + j] != oldPageData[i + j])
1214                        buf.append(String.format(" %02X", pageData[i + j]));
1215                    else
1216                        buf.append(" ..");
1217                }
1218                buf.append('\n');
1219            }
1220
1221            i += 32;
1222        }
1223
1224        return buf.toString();
1225    }
