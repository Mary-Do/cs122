055    /**
056     * The system property that can be used to specify the default page-size
057     * to use when creating new database files.
058     */
059    public static final String PROP_PAGESIZE = "nanodb.pagesize";
060

062    /**
063     * The default base-directory path used by the storage manager.  This
064     * value is set to "<tt>./datafiles</tt>" (or "<tt>.\datafiles</tt>"
065     * if on Windows).
066     */
067    public static final String DEFAULT_BASEDIR_PATH =
068        "." + File.separator + "datafiles";

071    /**
072     * Returns the current page size to use for new database files.  If the
073     * <tt>nanodb.pagesize</tt> system property is a valid page size then this
074     * value is used.  Otherwise, the {@link DBFile#DEFAULT_PAGESIZE} value is
075     * used.
076     *
077     * @return the current page size to use for new database files
078     */
079    public static int getCurrentPageSize() {
080        // Use the default page size if no property value is specified.
081        int pageSize = DBFile.DEFAULT_PAGESIZE;
082
083        String pageSizeStr = System.getProperty(PROP_PAGESIZE);
084        if (pageSizeStr != null) {
085            try {
086                pageSize = Integer.parseInt(pageSizeStr);
087            }
088            catch (NumberFormatException nfe) {
089                logger.warn("Current value of " + PROP_PAGESIZE +
090                    " property is not an integer:  \"" + pageSizeStr + "\"");
091            }
092
093            if (!DBFile.isValidPageSize(pageSize)) {
094                logger.warn("Current value of " + PROP_PAGESIZE +
095                    " property is not a valid page size:  " + pageSize);
096
097                pageSize = DBFile.DEFAULT_PAGESIZE;
098            }
099        }
100
101        return pageSize;
102    }

105    private class StoragePropertyHandler implements PropertyHandler {
106
107        @Override
108        public Object getPropertyValue(String propertyName)
109            throws UnrecognizedPropertyException {
110
111            if (PROP_PAGESIZE.equals(propertyName)) {
112                return getCurrentPageSize();
113            }
114            else if (PROP_BASEDIR.equals(propertyName)) {
115                return getBaseDir().toString();
116            }
117            else {
118                throw new UnrecognizedPropertyException("No property named " +
119                    propertyName);
120            }
121        }
122
123        @Override
124        public void setPropertyValue(String propertyName, Object value)
125            throws UnrecognizedPropertyException, ReadOnlyPropertyException,
126                   TypeCastException {
127
128            if (PROP_PAGESIZE.equals(propertyName)) {
129                throw new ReadOnlyPropertyException(propertyName +
130                    " is read-only");
131            }
132            else if (PROP_BASEDIR.equals(propertyName)) {
133                throw new ReadOnlyPropertyException(propertyName +
134                        " is read-only");
135            }
136            else {
137                throw new UnrecognizedPropertyException("No property named " +
138                    propertyName);
139            }
140        }
141    }

200    /**
201     * This method initializes the storage manager.  It should only be called
202     * once.
203     *
204     * @throws IllegalStateException if <tt>init()</tt> has already been called
205     * @throws IOException if the storage manager cannot access the data
206     *         directory for some reason
207     */
208    public void initialize(NanoDBServer server, File baseDir) throws IOException {
209        if (initialized) {
210            throw new IllegalStateException(
211                "Storage manager is already initialized.");
212        }
213
214        if (server == null)
215            throw new IllegalArgumentException("server cannot be null");
216
217        this.server = server;
218
219        // Make sure the base directory exists and is valid and all that.
220
221        if (baseDir == null) {
222            String defaultBaseDir = System.getProperty(PROP_BASEDIR,
223                DEFAULT_BASEDIR_PATH);
224
225            baseDir = new File(defaultBaseDir);
226        }
227
228        if (!baseDir.exists()) {
229            logger.info("Base directory " + baseDir + " doesn't exist; creating.");
230            if (!baseDir.mkdirs()) {
231                throw new IOException("Couldn't create base directory " + baseDir);
232            }
233        }
234
235        if (!baseDir.isDirectory()) {
236            throw new IOException("Base-directory path " + baseDir +
237                " doesn't refer to a directory.");
238        }
239
240        logger.info("Using base directory " + baseDir);
241        this.baseDir = baseDir;
242
243        // Register properties that the Storage Manager exposes.
244        server.getPropertyRegistry().registerProperties(
245            new StoragePropertyHandler(), PROP_PAGESIZE, PROP_BASEDIR);
246
247        fileManager = new FileManagerImpl(baseDir);
248        bufferManager = new BufferManager(server, fileManager);
249
250        tupleFileManagers.put(DBFileType.HEAP_TUPLE_FILE,
251            new HeapTupleFileManager(this));
252
253        // TODO:  Register B-tree file-type here.
254
255        if (TransactionManager.isEnabled()) {
256            logger.info("Initializing transaction manager.");
257            transactionManager = new TransactionManager(server);
258
259            // This method opens the transaction-state file, performs any
260            // necessary recovery operations, and so forth.
261            transactionManager.initialize();
262        }
263        else {
264            logger.info("Transaction manager is disabled.");
265        }
266
267        tableManager = new IndexedTableManager(this);
268        indexManager = new BasicIndexManager(this);
269
270        if (ENABLE_INDEXES) {
271            EventDispatcher eventDispatcher = server.getEventDispatcher();
272
273            // TODO:  Register the event-handler that enforces database constraints!
274            // eventDispatcher.addRowEventListener(new DatabaseConstraintEnforcer(this));
275
276            // TODO:  Register the event-handler that updates indexes when tables change.
277            // eventDispatcher.addRowEventListener(new IndexUpdater(this));
278        }
279
280        initialized = true;
281    }

284    /**
285     * This method shuts down the storage manager.  It should only be called
286     * once.
287     *
288     * @throws IllegalStateException if <tt>init()</tt> has not been called
289     * @throws IOException if the storage manager cannot save all data for some
290     *         reason
291     */
292    public void shutdown() throws IOException {
293        // Detect if already shut down...
294        if (!initialized) {
295            throw new IllegalStateException(
296                "Storage manager is not initialized.");
297        }
298
299        if (transactionManager != null)
300            transactionManager.forceWAL();
301
302        List<DBFile> dbFiles = bufferManager.removeAll();
303        for (DBFile dbFile : dbFiles)
304            fileManager.closeDBFile(dbFile);
305
306        // Register properties that the Storage Manager exposes.
307        server.getPropertyRegistry().unregisterProperties(
308            PROP_PAGESIZE, PROP_BASEDIR);
309
310        initialized = false;
311    }

339    public DBFile createDBFile(String filename, DBFileType type)
340        throws IOException {
341
342        if (bufferManager.getFile(filename) != null) {
343            throw new IllegalStateException("A file " + filename +
344                " is already cached in the Buffer Manager!  Does it already exist?");
345        }
346
347        DBFile dbFile = fileManager.createDBFile(filename, type,
348            getCurrentPageSize());
349
350        bufferManager.addFile(dbFile);
351
352        return dbFile;
353    }
354
355
356    public DBFile openDBFile(String filename) throws IOException {
357        DBFile dbFile = bufferManager.getFile(filename);
358        if (dbFile == null) {
359            dbFile = fileManager.openDBFile(filename);
360            bufferManager.addFile(dbFile);
361        }
362
363        return dbFile;
364    }

367    public TupleFile openTupleFile(String filename) throws IOException {
368        DBFile dbFile = fileManager.openDBFile(filename);
369        DBFileType type = dbFile.getType();
370        TupleFileManager tfManager = getTupleFileManager(type);
371
372        logger.debug(String.format("Opened DBFile for tuple file at path %s.",
373            dbFile.getDataFile()));
374        logger.debug(String.format("Type is %s, page size is %d bytes.",
375            type, dbFile.getPageSize()));
376
377        return tfManager.openTupleFile(dbFile);
378    }

421    /**
422     * This method returns a database page to use, retrieving it from the buffer
423     * manager if it is already loaded, or reading it from the specified data
424     * file if it is not already loaded.  If the page must be loaded from the
425     * file, it will be added to the buffer manager.  This operation may cause
426     * other database pages to be evicted from the buffer manager, and written
427     * back to disk if the evicted pages are dirty.
428     * <p>
429     * The <tt>create</tt> flag controls whether an error is propagated, if the
430     * requested page is past the current end of the data file.  (Note that if a
431     * new page is created, the file's size will not reflect the new page until
432     * it is actually written to the file.)
433     *
434     * @param dbFile the database file to load the page from
435     * @param pageNo the number of the page to load
436     * @param create a flag specifying whether the page should be created if it
437     *        doesn't already exist
438     *
439     * @return the database page, either from cache or from the data file
440     *
441     * @throws IllegalArgumentException if the page number is negative
442     *
443     * @throws java.io.EOFException if the requested page is not in the data file,
444     *         and the <tt>create</tt> flag is set to <tt>false</tt>.
445     */
446    public DBPage loadDBPage(DBFile dbFile, int pageNo, boolean create)
447        throws IOException {
448
449        // Try to retrieve from the buffer manager.
450        DBPage dbPage = bufferManager.getPage(dbFile, pageNo);
451        if (dbPage == null) {
452            // Buffer manager didn't have it.  Read the page directly from
453            // the file, then add it to the buffer manager.
454            dbPage = new DBPage(bufferManager, dbFile, pageNo);
455            try {
456                fileManager.loadPage(dbFile, pageNo, dbPage.getPageData(), create);
457                bufferManager.addPage(dbPage);
458            }
459            catch (IOException e) {
460                // Make sure to release the DBPage's buffer, or else we will
461                // have a resource leak...
462                dbPage.invalidate();
463                throw e;
464            }
465        }
466
467        return dbPage;
468    }

472     * This method returns a database page to use, retrieving it from the buffer
473     * manager if it is already loaded, or reading it from the specified data
474     * file if it is not already loaded.  If the page must be loaded from the
475     * file, it will be added to the buffer manager.  This operation may cause
476     * other database pages to be evicted from the buffer manager, and written
477     * back to disk if the evicted pages are dirty.
478     * <p>
479     * (This method is simply a wrapper of
480     * {@link #loadDBPage(DBFile, int, boolean)}, passing <tt>false</tt> for
481     * <tt>create</tt>.)
482     *
483     * @param dbFile the database file to load the page from
484     * @param pageNo the number of the page to load
485     *
486     * @return the database page, either from cache or from the data file
487     *
488     * @throws IllegalArgumentException if the page number is negative
489     *
490     * @throws java.io.EOFException if the requested page is not in the data file,
491     *         and the <tt>create</tt> flag is set to <tt>false</tt>.
492     */
493    public DBPage loadDBPage(DBFile dbFile, int pageNo) throws IOException {
494        return loadDBPage(dbFile, pageNo, false);
495    }

498    /**
499     * This method causes any changes to the specified page to be logged by
500     * the transaction manager's write-ahead log, so that the changes can be
501     * redone or undone as may be appropriate.  Once the page's changes have
502     * been logged, the {@link DBPage#syncOldPageData} method is called on
503     * the page, since the page's changes have been recorded in the WAL.
504     *
505     * @param dbPage the page to record changes for
506     */
507    public void logDBPageWrite(DBPage dbPage) throws IOException {
508        // If the page is dirty, record its changes to the write-ahead log.
509        if (transactionManager != null)
510            transactionManager.recordPageUpdate(dbPage);
511    }
