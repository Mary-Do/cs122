001package edu.caltech.nanodb.storage;
002
003import java.io.IOException;
004import java.util.List;
005
006import edu.caltech.nanodb.expressions.Expression;
007import edu.caltech.nanodb.relations.Tuple;
008
009
010/**
011 * This interface extends the {@link TupleFile} interface, adding operations
012 * that can be provided on files of tuples that are hashed on a specific key.
013 * The key used to hash tuples is returned by the {@link #getKeySpec} method.
014 */
015public interface HashedTupleFile extends TupleFile {
016    /**
017     * Returns the column(s) that comprise the hash key in this tuple file.
018     *
019     * @return the column(s) that comprise the hash key in this tuple file.
020     */
021    List<Expression> getKeySpec();
022
023
024    /**
025     * Returns the first tuple in the file that has the same hash-key values,
026     * or {@code null} if there are no tuples with this hash-key value in
027     * the tuple file.
028     *
029     * @param hashKey the tuple to search for
030     *
031     * @return The first tuple in the file with the same hash-key values, or
032     *         {@code null} if the file contains no files with the specified
033     *         search key value.  This tuple will actually be backed by the
034     *         tuple file, so typically it will be a subclass of
035     *         {@link PageTuple}.
036     *
037     * @throws IOException if an IO error occurs during the operation
038     */
039    Tuple findFirstTupleEquals(Tuple hashKey) throws IOException;
040
041
042    /**
043     * Returns the next entry in the index that has the same hash-key value,
044     * or {@code null} if there are no more entries with this hash-key value
045     * in the tuple file.
046     *
047     * @param prevTuple The tuple from which to resume the search for the next
048     *        tuple with the same hash-key values.  This should be a tuple
049     *        returned by a previous call to {@link #findFirstTupleEquals} or
050     *        {@link #findNextTupleEquals}; using any other tuple would be an
051     *        error.
052     *
053     * @return The next tuple in the file with the same hash-key values, or
054     *         {@code null} if there are no more entries with this hash-key
055     *         value in the file.
056     *
057     * @throws IOException if an IO error occurs during the operation
058     */
059    Tuple findNextTupleEquals(Tuple prevTuple) throws IOException;
060}
