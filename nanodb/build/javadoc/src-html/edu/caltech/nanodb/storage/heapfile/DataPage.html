009/**
010 * This class provides the constants and operations necessary for manipulating
011 * a data page within a heap file.  Heap files use a slotted-page structure
012 * where tuple-data is filled from the end of the page forward, so for two slots
013 * at indexes <em>i</em> and <em>j</em>, where <em>i</em> &lt; <em>j</em>, the
014 * start of <em>i</em>'s tuple data will be <u>after</u> <em>j</em>'s tuple
015 * data.
016 *
017 * @design (Donnie) Why is this class a static class, instead of a wrapper class
018 *         around the {@link DBPage}?  No particular reason, really.  The class
019 *         is used relatively briefly when a table is being accessed, and there
020 *         is no real need for it to manage its own object-state, so it was just
021 *         as convenient to provide all functionality as static methods.  This
022 *         avoids the (small) overhead of instantiating an object as well.  But
023 *         really, these are not particularly strong reasons.
024 */
025public class DataPage {
026    /** A logging object for reporting anything interesting that happens. */
027    private static Logger logger = Logger.getLogger(DataPage.class);
028
029
030    /**
031     * The offset in the data page where the number of slots in the slot table
032     * is stored.
033     */
034    public static final int OFFSET_NUM_SLOTS = 0;
035
036
037    /**
038     * This offset-value is stored into a slot when it is empty.  It is set to
039     * zero because this is where the page's slot-count is stored and therefore
040     * this is obviously an invalid offset for a tuple to be located at.
041     */
042    public static final int EMPTY_SLOT = 0;
043
044
045    /**
046     * Initialize a newly allocated data page.  Currently this involves setting
047     * the number of slots to 0.  There is no other internal structure in data
048     * pages at this point.
049     *
050     * @param dbPage the data page to initialize
051     */
052    public static void initNewPage(DBPage dbPage) {
053        setNumSlots(dbPage, 0);
054    }

057    public static int getSlotOffset(int slot) {
058        return (1 + slot) * 2;
059    }
060
061
062    /**
063     * Returns the number of slots in this data page.  This can be considered
064     * to be the current "capacity" of the page, since any number of the slots
065     * could be set to {@link #EMPTY_SLOT} to indicate that they are empty.
066     *
067     * @param dbPage the data page to retrieve the number of slots for
068     * @return the current number of slots in the page
069     */
070    public static int getNumSlots(DBPage dbPage) {
071        return dbPage.readUnsignedShort(OFFSET_NUM_SLOTS);
072    }
073
074
075    /**
076     * Sets the number of slots in this data page.  Note that if the number of
077     * slots is increased, the new slots <em>must</em> be initialized to
078     * {@link #EMPTY_SLOT}, or else functions like {@link #getFreeSpaceInPage}
079     * (which relies on the offset stored in the last slot) will produce wrong
080     * results.
081     *
082     * @param dbPage the data page to set the number of slots for
083     * @param numSlots the value to store
084     */
085    public static void setNumSlots(DBPage dbPage, int numSlots) {
086        dbPage.writeShort(OFFSET_NUM_SLOTS, numSlots);
087    }

090    /**
091     * This static helper function returns the index where the slot list ends in
092     * the data page.
093     *
094     * @param dbPage the data page to examine
095     * @return the index in the page data where the slot-table ends.  This also
096     *         happens to be the size of the slot-table in bytes.
097     */
098    public static int getSlotsEndIndex(DBPage dbPage) {
099        // Slots are at indexes [0, numSlots), so just pass the total number of
100        // slots into the helper function.
101        return getSlotOffset(getNumSlots(dbPage));
102    }

105    /**
106     * This static helper function returns the value stored in the specified
107     * slot.  This will either be the offset of the start of a tuple in the data
108     * page, or it will be {@link #EMPTY_SLOT} if the slot is empty.
109     *
110     * @param dbPage the data page to retrieve the slot-value from
111     * @param slot the slot to retrieve the value for; valid values are from 0
112     *        to {@link #getNumSlots} - 1.
113     *
114     * @return the current value stored for the slot in the data page
115     *
116     * @throws IllegalArgumentException if the specified slot number is outside
117     *         the range [0, {@link #getNumSlots}).
118     */
119    public static int getSlotValue(DBPage dbPage, int slot) {
120        int numSlots = getNumSlots(dbPage);
121
122        if (slot < 0 || slot >= numSlots) {
123            throw new IllegalArgumentException("Valid slots are in range [0," +
124                numSlots + ").  Got " + slot);
125        }
126
127        return dbPage.readUnsignedShort(getSlotOffset(slot));
128    }

131    /**
132     * This static helper function sets the value for the specified slot.  This
133     * should either be the offset of the start of a tuple in the data
134     * page, or it should be {@link #EMPTY_SLOT} if the slot is empty.
135     *
136     * @param dbPage the data page to set the slot-value for
137     * @param slot the slot to set the value of; valid values are from 0 to
138     *        {@link #getNumSlots} - 1.
139     * @param value the value to store for the slot
140     *
141     * @throws IllegalArgumentException if the specified slot number is outside
142     *         the range [0, {@link #getNumSlots}).
143     */
144    public static void setSlotValue(DBPage dbPage, int slot, int value) {
145        int numSlots = getNumSlots(dbPage);
146
147        if (slot < 0 || slot >= numSlots) {
148            throw new IllegalArgumentException("Valid slots are in range [0," +
149                numSlots + ").  Got " + slot);
150        }
151
152        dbPage.writeShort(getSlotOffset(slot), value);
153    }

156    public static int getSlotIndexFromOffset(DBPage dbPage, int offset)
157        throws IllegalArgumentException {
158
159        if (offset % 2 != 0) {
160            throw new IllegalArgumentException(
161                "Slots occur at even indexes (each slot is a short).");
162        }
163
164        int slot = (offset - 2) / 2;
165        int numSlots = getNumSlots(dbPage);
166
167        if (slot < 0 || slot >= numSlots) {
168            throw new IllegalArgumentException("Valid slots are in range [0," +
169                numSlots + ").  Got " + slot);
170        }
171
172        return slot;
173    }

176    /**
177     * This static helper function returns the index of where tuple data
178     * currently starts in the specified data page.  This method uses the last
179     * slot in the slot-table to determine where the tuple data begins, since
180     * the slots must be organized in decreasing order of offset-value.  (In
181     * other words, we add tuple data from the back of the block towards the
182     * front.)
183     *
184     * @see #setNumSlots
185     *
186     * @param dbPage the data page to examine
187     *
188     * @return the index where the tuple data starts in this data page
189     */
190    public static int getTupleDataStart(DBPage dbPage) {
191        int numSlots = getNumSlots(dbPage);
192
193        // If there are no tuples in this page, "data start" is the top of the
194        // page data.
195        int dataStart = getTupleDataEnd(dbPage);
196
197        int slot = numSlots - 1;
198        while (slot >= 0) {
199            int slotValue = getSlotValue(dbPage, slot);
200            if (slotValue != EMPTY_SLOT) {
201                dataStart = slotValue;
202                break;
203            }
204
205            --slot;
206        }
207
208        return dataStart;
209    }

212    /**
213     * This static helper function returns the index of where tuple data
214     * currently ends in the specified data page.  This value depends more on
215     * the overall structure of the data page, and at present is simply the
216     * page-size.
217     *
218     * @param dbPage the data page to examine
219     *
220     * @return the index where the tuple data ends in this data page
221     */
222    public static int getTupleDataEnd(DBPage dbPage) {
223        return dbPage.getPageSize();
224    }

227    /**
228     * Returns the length of the tuple stored at the specified slot.  It is
229     * invalid to use this method on an empty slot.
230     *
231     * @param dbPage the data page being examined
232     * @param slot the slot of the tuple to retrieve the length of
233     * @return the length of the tuple's data stored in this slot
234     *
235     * @throws IllegalArgumentException if the specified slot is invalid, or if
236     *         the specified slot has {@link #EMPTY_SLOT} for its value
237     */
238    public static int getTupleLength(DBPage dbPage, int slot) {
239        int numSlots = getNumSlots(dbPage);
240
241        if (slot < 0 || slot >= numSlots) {
242            throw new IllegalArgumentException(
243                "Valid slots are in range [0," + slot + ").  Got " + slot);
244        }
245
246        int tupleStart = getSlotValue(dbPage, slot);
247
248        if (tupleStart == EMPTY_SLOT)
249            throw new IllegalArgumentException("Slot " + slot + " is empty.");
250
251        int tupleLength = -1;
252
253        int prevSlot = slot - 1;
254        while (prevSlot >= 0) {
255            int prevTupleStart = getSlotValue(dbPage, prevSlot);
256            if (prevTupleStart != EMPTY_SLOT) {
257
258                // Earlier slots have higher offsets.  (Yes it's weird.)
259                tupleLength = prevTupleStart - tupleStart;
260
261                break;
262            }
263
264            prevSlot--;
265        }
266
267        if (prevSlot < 0) {
268            // The specified slot held the last tuple in the page.
269            tupleLength = getTupleDataEnd(dbPage) - tupleStart;
270        }
271
272        return tupleLength;
273    }

276    /**
277     * This static helper function returns the amount of free space in
278     * a tuple data page.  It simply uses other methods in this class to
279     * perform the simple computation.
280     *
281     * @param dbPage the data page to examine
282     * @return the amount of free space in the data page, in bytes
283     */
284    public static int getFreeSpaceInPage(DBPage dbPage) {
285        return getTupleDataStart(dbPage) - getSlotsEndIndex(dbPage);
286    }

289    /**
290     * This static helper method verifies that the specified data page has
291     * proper structure and organization by performing various sanity checks.
292     * Currently, the only sanity check it performs is to verify that
293     * slot-offsets do indeed decrease monotonically as the slot-index
294     * increases.
295     * <p>
296     * Any issues will cause warnings to be logged.
297     *
298     * @param dbPage the data page to check
299     */
300    public static void sanityCheck(DBPage dbPage) {
301        int numSlots = getNumSlots(dbPage);
302        if (numSlots == 0)
303            return;
304
305        // Find the first occupied slot, and get its offset into prevOffset.
306        int iSlot = -1;
307        int prevSlot = -1;
308        int prevOffset = EMPTY_SLOT;
309        while (iSlot + 1 < numSlots && prevOffset == EMPTY_SLOT) {
310            iSlot++;
311            prevSlot = iSlot;
312            prevOffset = getSlotValue(dbPage, iSlot);
313        }
314
315        while (iSlot + 1 < numSlots) {
316
317            // Find the next occupied slot, and get its offset into offset.
318            int offset = EMPTY_SLOT;
319            while (iSlot + 1 < numSlots && offset == EMPTY_SLOT) {
320                iSlot++;
321                offset = getSlotValue(dbPage, iSlot);
322            }
323
324            if (iSlot < numSlots) {
325                // Tuple offsets should be strictly decreasing.
326                if (prevOffset <= offset) {
327                    logger.warn(String.format(
328                        "Slot %d and %d offsets are not strictly decreasing " +
329                        "(%d should be greater than %d)", prevSlot, iSlot,
330                        prevOffset, offset));
331                }
332
333                prevSlot = iSlot;
334                prevOffset = offset;
335            }
336        }
337    }

341    /**
342     * <p>
343     * This static helper function creates a space in the data page of the
344     * specified size, sliding tuple data below the offset down to
345     * create a gap.  Because tuples below the specified offset will actually
346     * move, some of the slots in the page may also need to be modified.
347     * </p>
348     * <p>The new space is initialized to all zero values.</p>
349     *
350     * @param dbPage The table data-page to insert space into.
351     *
352     * @param off The offset in the page where the space will be added.
353     *
354     * @param len The number of bytes to insert.
355     */
356    public static void insertTupleDataRange(DBPage dbPage, int off, int len) {
357
358        int tupDataStart = getTupleDataStart(dbPage);
359
360        if (off < tupDataStart) {
361            throw new IllegalArgumentException("Specified offset " + off +
362                " is not actually in the tuple data portion of this page " +
363                "(data starts at offset " + tupDataStart + ").");
364        }
365
366        if (len < 0)
367            throw new IllegalArgumentException("Length must not be negative.");
368
369        if (len > getFreeSpaceInPage(dbPage)) {
370            throw new IllegalArgumentException("Specified length " + len +
371                " is larger than amount of free space in this page (" +
372                getFreeSpaceInPage(dbPage) + " bytes).");
373        }
374
375        // If off == tupDataStart then there's no need to move anything.
376        if (off > tupDataStart) {
377            // Move the data in the range [tupDataStart, off) to
378            // [tupDataStart - len, off - len).  Thus there will be a gap in the
379            // range [off - len, off) after the operation is completed.
380
381            dbPage.moveDataRange(tupDataStart, tupDataStart - len,
382                off - tupDataStart);
383        }
384
385        // Zero out the gap that was just created.
386        int startOff = off - len;
387        dbPage.setDataRange(startOff, len, (byte) 0);
388
389        // Update affected slots; this includes all slots below the specified
390        // offset.  The update is easy; slot values just move down by len bytes.
391
392        int numSlots = getNumSlots(dbPage);
393        for (int iSlot = 0; iSlot < numSlots; iSlot++) {
394
395            int slotValue = getSlotValue(dbPage, iSlot);
396            if (slotValue != EMPTY_SLOT && slotValue < off) {
397                // Update this slot's offset.
398                slotValue -= len;
399                setSlotValue(dbPage, iSlot, slotValue);
400            }
401        }
402    }

405    /**
406     * This static helper function removes a sequence of bytes from the current
407     * tuple data in the page, sliding tuple data below the offset forward to
408     * fill in the gap.  Because tuples below the specified offset will actually
409     * move, some of the slots in the page may also need to be modified.
410     *
411     * @param dbPage The table data-page to insert space into.
412     *
413     * @param off The offset in the page where the space will be removed.
414     *
415     * @param len The number of bytes to remove.
416     */
417    public static void deleteTupleDataRange(DBPage dbPage, int off, int len) {
418        int tupDataStart = getTupleDataStart(dbPage);
419
420        logger.debug(String.format(
421            "Deleting tuple data-range offset %d, length %d", off, len));
422
423        if (off < tupDataStart) {
424            throw new IllegalArgumentException("Specified offset " + off +
425                " is not actually in the tuple data portion of this page " +
426                "(data starts at offset " + tupDataStart + ").");
427        }
428
429        if (len < 0)
430            throw new IllegalArgumentException("Length must not be negative.");
431
432        if (getTupleDataEnd(dbPage) - off < len) {
433            throw new IllegalArgumentException("Specified length " + len +
434                " is larger than size of tuple data in this page (" +
435                (getTupleDataEnd(dbPage) - off) + " bytes).");
436        }
437
438        // Move the data in the range [tupDataStart, off) to
439        // [tupDataStart + len, off + len).
440
441        logger.debug(String.format(
442            "    Moving %d bytes of data from [%d, %d) to [%d, %d)",
443            off - tupDataStart, tupDataStart, off,
444            tupDataStart + len, off + len));
445
446        dbPage.moveDataRange(tupDataStart, tupDataStart + len, off - tupDataStart);
447
448        // Update affected slots; this includes all (non-empty) slots whose
449        // offset is below the specified offset.  The update is easy; slot
450        // values just move up by len bytes.
451
452        int numSlots = getNumSlots(dbPage);
453        for (int iSlot = 0; iSlot < numSlots; iSlot++) {
454
455            int slotValue = getSlotValue(dbPage, iSlot);
456            if (slotValue != EMPTY_SLOT && slotValue <= off) {
457                // Update this slot's offset.
458                slotValue += len;
459                setSlotValue(dbPage, iSlot, slotValue);
460            }
461        }
462    }

465    /**
466     * Update the data page so that it has space for a new tuple of the
467     * specified size.  The new tuple is assigned a slot (whose index is
468     * returned by this method), and the space for the tuple is initialized
469     * to all zero values.
470     *
471     * @param dbPage The data page to store the new tuple in.
472     *
473     * @param len The length of the new tuple's data.
474     *
475     * @return The slot-index for the new tuple.  The offset to the start
476     *         of the requested space is available via that slot.  (Use
477     *         {@link #getSlotValue} to retrieve that offset.)
478     */
479    public static int allocNewTuple(DBPage dbPage, int len) {
480
481        if (len < 0) {
482            throw new IllegalArgumentException(
483                "Length must be nonnegative; got " + len);
484        }
485
486        // The amount of free space we need in the database page, if we are
487        // going to add the new tuple.
488        int spaceNeeded = len;
489
490        logger.debug("Allocating space for new " + len + "-byte tuple.");
491
492        // Search through the current list of slots in the page.  If a slot
493        // is marked as "empty" then we can use that slot.  Otherwise, we
494        // will need to add a new slot to the end of the list.
495
496        int slot;
497        int numSlots = getNumSlots(dbPage);
498
499        logger.debug("Current number of slots on page:  " + numSlots);
500
501        // This variable tracks where the new tuple should END.  It starts
502        // as the page-size, and gets moved down past each valid tuple in
503        // the page, until we find an available slot in the page.
504        int newTupleEnd = getTupleDataEnd(dbPage);
505
506        for (slot = 0; slot < numSlots; slot++) {
507            // currSlotValue is either the start of that slot's tuple-data,
508            // or it is set to EMPTY_SLOT.
509            int currSlotValue = getSlotValue(dbPage, slot);
510
511            if (currSlotValue == EMPTY_SLOT)
512                break;
513            else
514                newTupleEnd = currSlotValue;
515        }
516
517        // First make sure we actually have enough space for the new tuple.
518
519        if (slot == numSlots) {
520            // We'll need to add a new slot to the list.  Make sure there's
521            // room.
522            spaceNeeded += 2;
523        }
524
525        if (spaceNeeded > getFreeSpaceInPage(dbPage)) {
526            // Switch this to a checked exception?  The table manager has
527            // already verified that the page should have enough space, so if
528            // this fails, it would indicate a bug.  So, runtime exception is
529            // fine for now.
530            throw new IllegalArgumentException(
531                "Space needed for new tuple (" + spaceNeeded +
532                " bytes) is larger than the free space in this page (" +
533                getFreeSpaceInPage(dbPage) + " bytes).");
534        }
535
536        // Now we know we have space for the tuple.  Update the slot list,
537        // and the update page's layout to make room for the new tuple.
538
539        if (slot == numSlots) {
540            logger.debug("No empty slot available.  Adding a new slot.");
541
542            // Add the new slot to the page, and update the total number of
543            // slots.
544            numSlots++;
545            setNumSlots(dbPage, numSlots);
546            setSlotValue(dbPage, slot, EMPTY_SLOT);
547        }
548
549        logger.debug(String.format(
550            "Tuple will get slot %d.  Final number of slots:  %d",
551            slot, numSlots));
552
553        int newTupleStart = newTupleEnd - len;
554
555        logger.debug(String.format(
556            "New tuple of %d bytes will reside at location [%d, %d).",
557            len, newTupleStart, newTupleEnd));
558
559        // Make room for the new tuple's data to be stored into.  Since
560        // tuples are stored from the END of the page going backwards, we
561        // specify the new tuple's END index, and the tuple's length.
562        // (Note:  This call also updates all affected slots whose offsets
563        // would be changed.)
564        insertTupleDataRange(dbPage, newTupleEnd, len);
565
566        // Set the slot's value to be the starting offset of the tuple.
567        // We have to do this *after* we insert the new space for the new
568        // tuple, or else insertTupleDataRange() will clobber the
569        // slot-value of this tuple.
570        setSlotValue(dbPage, slot, newTupleStart);
571
572        // Finally, return the slot-index of the new tuple.
573        return slot;
574    }
