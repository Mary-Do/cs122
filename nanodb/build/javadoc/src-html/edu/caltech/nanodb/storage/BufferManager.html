027/**
028 * The buffer manager reduces the number of disk IO operations by managing an
029 * in-memory cache of data pages.  It also imposes a limit on the maximum
030 * amount of space that can be used for data pages in the database.
031 *
032 * @todo Eventually add integrity checks, e.g. to make sure every cached
033 *       page's file appears in the collection of cached files.
034 */
035public class BufferManager {
036
037    /**
038     * The system property that can be used to specify the size of the page
039     * cache in the buffer manager.
040     */
041    public static final String PROP_PAGECACHE_SIZE = "nanodb.pagecache.size";
042
043    /** The default page-cache size is defined to be 20MB. */
044    public static final long DEFAULT_PAGECACHE_SIZE = 20 * 1024 * 1024;
045
046
047    /**
048     * The system property that can be used to specify the page replacement
049     * policy in the buffer manager.

051    public static final String PROP_PAGECACHE_POLICY = "nanodb.pagecache.policy";
052
053    /** The default page-cache policy is LRU. */
054    public static final String DEFAULT_PAGECACHE_POLICY = "lru";

057    private static class DBPageID {
058        private File file;
059
060        private int pageNo;
061
062        public DBPageID(File file, int pageNo) {
063            this.file = file;
064            this.pageNo = pageNo;
065        }
066
067        public DBPageID(DBPage dbPage) {
068            this(dbPage.getDBFile().getDataFile(), dbPage.getPageNo());
069        }
070
071        @Override
072        public boolean equals(Object obj) {
073            if (obj instanceof DBPageID) {
074                DBPageID other = (DBPageID) obj;
075                return file.equals(other.file) && pageNo == other.pageNo;
076            }
077            return false;
078        }
079
080        @Override
081        public int hashCode() {
082            int hash = 17;
083            hash = hash * 37 + file.hashCode();
084            hash = hash * 37 + pageNo;
085            return hash;
086        }
087    }

111            }
112            return false;

115        @Override
116        public int hashCode() {
117            int hash = 7;
118            hash = 31 * hash + dbFile.hashCode();
119            hash = 31 * hash + pageNo;
120            return hash;
121        }

144    private class BufferManagerPropertyHandler implements PropertyHandler {
145
146        @Override
147        public Object getPropertyValue(String propertyName)
148                throws UnrecognizedPropertyException {
149
150            if (PROP_PAGECACHE_SIZE.equals(propertyName)) {
151                return maxCacheSize;
152            }
153            else if (PROP_PAGECACHE_POLICY.equals(propertyName)) {
154                return replacementPolicy;
155            }
156            else {
157                throw new UnrecognizedPropertyException("No property named " +
158                        propertyName);
159            }
160        }
161
162        @Override
163        public void setPropertyValue(String propertyName, Object value)
164                throws UnrecognizedPropertyException, ReadOnlyPropertyException,
165                TypeCastException {
166
167            if (PROP_PAGECACHE_SIZE.equals(propertyName)) {
168                throw new ReadOnlyPropertyException(propertyName +
169                        " is read-only");
170            }
171            else if (PROP_PAGECACHE_POLICY.equals(propertyName)) {
172                throw new ReadOnlyPropertyException(propertyName +
173                        " is read-only");
174            }
175            else {
176                throw new UnrecognizedPropertyException("No property named " +
177                        propertyName);
178            }
179        }
180    }

241    public BufferManager(NanoDBServer server, FileManager fileManager) {
242        this.fileManager = fileManager;

244        observers = new ArrayList<>();
245
246        configureMaxCacheSize();
247
248        cachedFiles = new LinkedHashMap<>();
249
250        replacementPolicy = configureReplacementPolicy();
251        cachedPages = new LinkedHashMap<>(16, 0.75f, "lru".equals(replacementPolicy));
252
253        totalBytesCached = 0;
254        allocatedBuffers = new HashSet<>();
255
256        sessionPinCounts = new HashMap<>();

258        if (server != null) {
259            // Register properties that the Buffer Manager exposes.
260            server.getPropertyRegistry().registerProperties(
261                new BufferManagerPropertyHandler(),
262                PROP_PAGECACHE_POLICY, PROP_PAGECACHE_SIZE);
263        }
264    }

267    private void configureMaxCacheSize() {
268        // Set the default up-front; it's just easier that way.
269        maxCacheSize = DEFAULT_PAGECACHE_SIZE;
270
271        String str = System.getProperty(PROP_PAGECACHE_SIZE);
272        if (str != null) {
273            str = str.trim().toLowerCase();
274
275            long scale = 1;
276            if (str.length() > 1) {
277                char modifierChar = str.charAt(str.length() - 1);
278                boolean removeModifier = true;
279                if (modifierChar == 'k')
280                    scale = 1024;
281                else if (modifierChar == 'm')
282                    scale = 1024 * 1024;
283                else if (modifierChar == 'g')
284                    scale = 1024 * 1024 * 1024;
285                else
286                    removeModifier = false;
287
288                if (removeModifier)
289                    str = str.substring(0, str.length() - 1);
290            }
291
292            try {
293                maxCacheSize = Long.parseLong(str);
294                maxCacheSize *= scale;
295            }
296            catch (NumberFormatException e) {
297                logger.error(String.format(
298                    "Could not parse page-cache size value \"%s\"; " +
299                    "using default value of %d bytes",
300                    System.getProperty(PROP_PAGECACHE_SIZE),
301                    DEFAULT_PAGECACHE_SIZE));
302
303                maxCacheSize = DEFAULT_PAGECACHE_SIZE;
304            }
305        }
306    }
307
308
309    private String configureReplacementPolicy() {
310        String str = System.getProperty(PROP_PAGECACHE_POLICY,
311            DEFAULT_PAGECACHE_POLICY);
312
313        str = str.trim().toLowerCase();
314
315        if (!("lru".equals(str) || "fifo".equals(str))) {
316            logger.error(String.format(
317                "Unrecognized value \"%s\" for page-cache replacement " +
318                "policy; using default value of LRU.",
319                System.getProperty(PROP_PAGECACHE_POLICY)));
320        }
321
322        return str;
323    }

339    /**
340     * This method attempts to allocate a buffer of the specified size,
341     * possibly evicting some existing buffers in order to make space.
342     *
343     * @param size the size of the buffer to allocate
344     *
345     * @return an array of bytes, of the specified size
346     *
347     * @throws IOException if a dirty page must be evicted from the buffer
348     *         manager, and an IO error occurred while writing the page to
349     *         persistent storage.
350     */
351    public byte[] allocBuffer(int size) throws IOException {
352        if (size <= 0)
353            throw new IllegalArgumentException("size must be > 0, got " + size);
354
355        ensureSpaceAvailable(size);
356
357        if (totalBytesCached + size > maxCacheSize) {
358            throw new IllegalStateException(
359                "Not enough room to allocate a buffer of " + size + " bytes!");
360        }
361
362        // Perform the allocation so that we know the JVM also has space...
363        // Then update the total bytes in use by the buffer manager.
364        byte[] buffer = new byte[size];
365        totalBytesCached += size;
366
367        // Record the identity of the buffer that we allocated, so that
368        // releaseBuffer() can verify that it came from the buffer manager.
369        // TODO:  System.identityHashCode() is not guaranteed to return a
370        //        distinct value for every object, so we can have collisions
371        //        on this value.  Come up with a different approach.
372        // allocatedBuffers.add(System.identityHashCode(buffer));
373
374        return buffer;
375    }
376
377
378    public void releaseBuffer(byte[] buffer) {
379        // Verify that this was a buffer we allocated?
380        // TODO:  System.identityHashCode() is not guaranteed to return a
381        //        distinct value for every object, so we can have collisions
382        //        on this value.  Come up with a different approach.
383        /*
384        if (!allocatedBuffers.remove(System.identityHashCode(buffer))) {
385            throw new IllegalArgumentException("Received a buffer that " +
386                "wasn't allocated by the Buffer Manager");
387        }
388        */
389
390        // Record that the buffer's space is now available.
391        totalBytesCached -= buffer.length;
392    }

437    /**
438     * Records that the page was pinned by the current session.  This method
439     * does not actually pin the page; it is presumed that the page is already
440     * pinned.
441     *
442     * @param dbPage the page that was pinned by the session
443     */
444    public void recordPagePinned(DBPage dbPage) {
445        int sessionID = SessionState.get().getSessionID();
446
447        // Retrieve the set of pages pinned by the current session.
448        HashMap<DBPageID, SessionPinCount> pinnedBySession =
449            sessionPinCounts.get(sessionID);
450        if (pinnedBySession == null) {
451            pinnedBySession = new HashMap<>();
452            sessionPinCounts.put(sessionID, pinnedBySession);
453        }
454
455        // Find the session-specific pin-count for the data page.
456        SessionPinCount spc = pinnedBySession.get(new DBPageID(dbPage));
457        if (spc == null) {
458            spc = new SessionPinCount(dbPage);
459            pinnedBySession.put(new DBPageID(dbPage), spc);
460        }
461
462        // Finally, increment the session's pin-count on this page.
463        spc.pinCount++;
464    }

467    /**
468     * Records that the page was unpinned by the current session.  This method
469     * does not actually unpin the page; it is presumed that the page will be
470     * unpinned after this call.
471     *
472     * @param dbPage the page that was unpinned
473     */
474    public void recordPageUnpinned(DBPage dbPage) {
475        int sessionID = SessionState.get().getSessionID();
476
477        // Retrieve the set of pages pinned by the current session.
478        HashMap<DBPageID, SessionPinCount> pinnedBySession =
479            sessionPinCounts.get(sessionID);
480        if (pinnedBySession == null) {
481            logger.error(String.format("DBPage %d is being unpinned by " +
482                "session %d, but we have no record of the session!",
483                dbPage.getPageNo(), sessionID));
484            return;
485        }
486
487        // Find the session-specific pin-count for the data page.
488        DBPageID pageID = new DBPageID(dbPage);
489        SessionPinCount spc = pinnedBySession.get(pageID);
490        if (spc == null) {
491            logger.error(String.format("DBPage %d is being unpinned by " +
492                "session %d, but we have no record of it having been pinned!",
493                dbPage.getPageNo(), sessionID));
494            return;
495        }
496
497        // Record that the page was unpinned.
498        spc.pinCount--;
499
500        // If the pin-count went to zero, remove the SessionPinCount object.
501        if (spc.pinCount == 0) {
502            pinnedBySession.remove(pageID);
503
504            // If the set of pages pinned by the current session is now empty,
505            // remove the set of pages.
506            if (pinnedBySession.isEmpty())
507                sessionPinCounts.remove(sessionID);
508        }
509    }

512    /**
513     * This method unpins all pages pinned by the current session.  This is
514     * generally done at the end of each transaction so that pages aren't
515     * pinned forever, and can actually be evicted from the buffer manager.
516     */
517    public void unpinAllSessionPages() {
518        // Unpin all pages pinned by this session.
519        int sessionID = SessionState.get().getSessionID();
520
521        // Retrieve the set of pages pinned by the current session.
522        HashMap<DBPageID, SessionPinCount> pinnedBySession =
523            sessionPinCounts.get(sessionID);
524
525        if (pinnedBySession == null) {
526            // Nothing to release!  Nice -- the session is very clean.
527            return;
528        }
529
530        // Duplicate the collection's values so that we don't get concurrent
531        // modification exceptions.
532        ArrayList<SessionPinCount> spcs = new ArrayList<>();
533        spcs.addAll(pinnedBySession.values());
534        for (SessionPinCount spc : spcs) {
535            // It would be an overstatement to say this is an error, since we
536            // can recover from it.
537            logger.warn(String.format("Session %d pinned DBPage %d %d times" +
538                " without a corresponding unpin call", sessionID,
539                spc.dbPage.getPageNo(), spc.pinCount));
540
541            while (spc.pinCount > 0)
542                spc.dbPage.unpin();
543        }
544
545        // Since unpinning the pages calls back into the buffer manager,
546        // we should automatically have all our "sessionPinCounts" state
547        // cleaned up along the way.
548    }

551    public void recordPageInvalidated(DBPage dbPage) {
552        if (dbPage == null)
553            throw new IllegalArgumentException("dbPage cannot be null");
554
555        int pageNo = dbPage.getPageNo();
556        DBPageID pageID = new DBPageID(dbPage);
557        if (dbPage.getPinCount() > 0) {
558            logger.warn(String.format("DBPage %d is being invalidated, but " +
559                "it has a pin-count of %d", pageNo, dbPage.getPinCount()));
560        }
561
562        for (int sessionID : sessionPinCounts.keySet()) {
563            HashMap<DBPageID, SessionPinCount> pinnedBySession =
564                sessionPinCounts.get(sessionID);
565
566            SessionPinCount spc = pinnedBySession.remove(pageID);
567            if (spc != null) {
568                logger.warn(String.format("DBPage %d is being invalidated, " +
569                    "but session %d has pinned it %d times", pageNo, sessionID,
570                    spc.pinCount));
571            }
572        }
573    }

577     * Retrieves the specified {@code DBPage} from the Buffer Manager if it's
578     * currently buffered, or {@code null} if the page is not currently
579     * buffered.  If a page is returned, it is pinned before it is returned.
580     *
581     * @param dbFile the file containing the page to retrieve
582     * @param pageNo the page number in the {@code DBFile} to retrieve
583     * @return the requested {@code DBPage}, or {@code null} if not found
584     */
585    public DBPage getPage(DBFile dbFile, int pageNo) {
586        DBPage dbPage = cachedPages.get(new CachedPageInfo(dbFile, pageNo));
587
588        logger.debug(String.format(
589            "Requested page [%s,%d] is%s in page-cache.",
590            dbFile, pageNo, (dbPage != null ? "" : " NOT")));
591
592        if (dbPage != null) {
593            // Make sure this page is pinned by the session so that we don't
594            // flush it until the session is done with it.
595            dbPage.pin();
596        }
597
598        return dbPage;
599    }

602    /**
603     * <p>
604     * Adds a new, previously unbuffered {@code DBPage} to the Buffer Manager.
605     * The page is pinned during the operation.
606     * </p>
607     * <p>
608     * It is an error if the page is already in the Buffer Manager when this
609     * method is called.
610     * </p>
611     *
612     * @param dbPage the page to add to the Buffer Manager
613     *
614     * @throws IllegalStateException if the page already appears in the Buffer
615     *         Manager.
616     */
617    public void addPage(DBPage dbPage) throws IOException {
618        if (dbPage == null)
619            throw new IllegalArgumentException("dbPage cannot be null");
620
621        DBFile dbFile = dbPage.getDBFile();
622        int pageNo = dbPage.getPageNo();
623
624        CachedPageInfo cpi = new CachedPageInfo(dbFile, pageNo);
625        if (cachedPages.containsKey(cpi)) {
626            throw new IllegalStateException(String.format(
627                "Page cache already contains page [%s,%d]", dbFile, pageNo));
628        }
629
630        logger.debug(String.format("Adding page [%s,%d] to page-cache.",
631            dbFile, pageNo));
632
633        int pageSize = dbPage.getPageSize();
634        ensureSpaceAvailable(pageSize);
635
636        cachedPages.put(cpi, dbPage);
637
638        // Make sure this page is pinned by the session so that we don't flush
639        // it until the session is done with it.
640        dbPage.pin();
641    }

644    /**
645     * This helper function ensures that the buffer manager has the specified
646     * amount of space available.  This is done by removing pages out of the
647     * buffer manager's cache
648     *
649     * @param bytesRequired the amount of space that should be made available
650     *        in the cache, in bytes
651     *
652     * @throws IOException if an IO error occurs when flushing dirty pages out
653     *         to disk
654     */
655    private void ensureSpaceAvailable(int bytesRequired) throws IOException {
656        // If we already have enough space, return without doing anything.
657        if (bytesRequired + totalBytesCached <= maxCacheSize)
658            return;
659
660        // We don't currently have enough space in the cache.  Try to solve
661        // this problem by evicting pages.  We collect together the pages to
662        // evict, so that we can update the write-ahead log before flushing
663        // the pages.
664
665        ArrayList<DBPage> dirtyPages = new ArrayList<>();
666
667        if (!cachedPages.isEmpty()) {
668            // The cache will be too large after adding this page.
669
670            Iterator<Map.Entry<CachedPageInfo, DBPage>> entries =
671                cachedPages.entrySet().iterator();
672
673            while (entries.hasNext() &&
674                bytesRequired + totalBytesCached > maxCacheSize) {
675                Map.Entry<CachedPageInfo, DBPage> entry = entries.next();
676
677                DBPage oldPage = entry.getValue();
678
679                if (oldPage.isPinned())  // Can't flush pages that are in use.
680                    continue;
681
682                logger.debug(String.format(
683                    "    Evicting page [%s,%d] from page-cache to make room.",
684                    oldPage.getDBFile(), oldPage.getPageNo()));
685
686                entries.remove();
687                totalBytesCached -= oldPage.getPageSize();
688
689                // If the page is dirty, we need to write its data to disk before
690                // invalidating it.  Otherwise, just invalidate it.
691                if (oldPage.isDirty()) {
692                    logger.debug("    Evicted page is dirty; must save to disk.");
693                    dirtyPages.add(oldPage);
694                }
695                else {
696                    oldPage.invalidate();
697                }
698            }
699        }
700
701        // If we have any dirty data pages, they need to be flushed to disk.
702        writeDirtyPages(dirtyPages, /* invalidate */ true);
703
704        if (bytesRequired + totalBytesCached > maxCacheSize)
705            logger.warn("Buffer manager is currently using too much space.");
706    }

709    /**
710     * This helper method writes out a list of dirty pages from the buffer
711     * manager, ensuring that if transactions are enabled, the
712     * write-ahead-logging rule is satisfied.
713     *
714     * @param dirtyPages the list of dirty pages to write
715     * @param invalidate if true then the dirty pages are invalidated so they
716     *        must be reloaded from disk
717     *
718     * @throws IOException if an IO error occurs while flushing dirty pages
719     */
720    private void writeDirtyPages(List<DBPage> dirtyPages, boolean invalidate)
721        throws IOException {
722
723        if (!dirtyPages.isEmpty()) {
724            // Pass the observers a read-only version of the pages so they
725            // can't change things.
726            List<DBPage> readOnlyPages =
727                Collections.unmodifiableList(dirtyPages);
728
729            for (BufferManagerObserver obs : observers)
730                obs.beforeWriteDirtyPages(readOnlyPages);
731
732            // Finally, we can write out each dirty page.
733            for (DBPage dbPage : dirtyPages) {
734                fileManager.savePage(dbPage.getDBFile(), dbPage.getPageNo(),
735                                     dbPage.getPageData());
736
737                dbPage.setDirty(false);
738
739                if (invalidate)
740                    dbPage.invalidate();
741            }
742        }
743    }

747     * This method writes all dirty pages in the specified file, optionally
748     * syncing the file after performing the write.  The pages are not removed
749     * from the buffer manager after writing them; their dirty state is simply
750     * cleared.
751     *
752     * @param dbFile the file whose dirty pages should be written to disk
753     *
754     * @param minPageNo dirty pages with a page-number less than this value
755     *        will not be written to disk
756     *
757     * @param maxPageNo dirty pages with a page-number greater than this value
758     *        will not be written to disk
759     *
760     * @param sync If true then the database file will be sync'd to disk;
761     *        if false then no sync will occur.  The sync will always occur,
762     *        in case dirty pages had previously been flushed to disk without
763     *        syncing.
764     *
765     * @throws IOException if an IO error occurs while updating the write-ahead
766     *         log, or while writing the file's contents.
767     */
768    public void writeDBFile(DBFile dbFile, int minPageNo, int maxPageNo,
769                            boolean sync) throws IOException {
770
771        logger.info(String.format("Writing all dirty pages for file %s to disk%s.",
772            dbFile, (sync ? " (with sync)" : "")));
773
774        Iterator<Map.Entry<CachedPageInfo, DBPage>> entries =
775            cachedPages.entrySet().iterator();
776
777        ArrayList<DBPage> dirtyPages = new ArrayList<>();
778
779        while (entries.hasNext()) {
780            Map.Entry<CachedPageInfo, DBPage> entry = entries.next();
781
782            CachedPageInfo info = entry.getKey();
783            if (dbFile.equals(info.dbFile)) {
784                DBPage oldPage = entry.getValue();
785                if (!oldPage.isDirty())
786                    continue;
787
788                int pageNo = oldPage.getPageNo();
789                if (pageNo < minPageNo || pageNo > maxPageNo)
790                    continue;
791
792                logger.debug(String.format("    Saving page [%s,%d] to disk.",
793                    oldPage.getDBFile(), oldPage.getPageNo()));
794
795                dirtyPages.add(oldPage);
796            }
797        }
798
799        writeDirtyPages(dirtyPages, /* invalidate */ false);
800
801        if (sync) {
802            logger.debug("Syncing file " + dbFile);
803            fileManager.syncDBFile(dbFile);
804        }
805    }

829    /**
830     * This method writes all dirty pages in the buffer manager to disk.  The
831     * pages are not removed from the buffer manager after writing them; their
832     * dirty state is simply cleared.
833     *
834     * @param sync if true, this method will sync all files in which dirty pages
835     *        were found, with the exception of WAL files and the
836     *        transaction-state file.  If false, no file syncing will be
837     *        performed.
838     *
839     * @throws IOException if an IO error occurs while updating the write-ahead
840     *         log, or while writing the file's contents.
841     */
842    public void writeAll(boolean sync) throws IOException {
843        logger.info("Writing ALL dirty pages in the Buffer Manager to disk.");
844
845        Iterator<Map.Entry<CachedPageInfo, DBPage>> entries =
846            cachedPages.entrySet().iterator();
847
848        ArrayList<DBPage> dirtyPages = new ArrayList<>();
849        HashSet<DBFile> dirtyFiles = new HashSet<>();
850
851        while (entries.hasNext()) {
852            Map.Entry<CachedPageInfo, DBPage> entry = entries.next();
853
854            DBPage oldPage = entry.getValue();
855            if (!oldPage.isDirty())
856                continue;
857
858            DBFile dbFile = oldPage.getDBFile();
859            DBFileType type = dbFile.getType();
860            if (type != DBFileType.WRITE_AHEAD_LOG_FILE &&
861                type != DBFileType.TXNSTATE_FILE) {
862                dirtyFiles.add(oldPage.getDBFile());
863            }
864
865            logger.debug(String.format("    Saving page [%s,%d] to disk.",
866                dbFile, oldPage.getPageNo()));
867
868            dirtyPages.add(oldPage);
869        }
870
871        writeDirtyPages(dirtyPages, /* invalidate */ false);
872
873        if (sync) {
874            logger.debug("Synchronizing all files containing dirty pages to disk.");
875            for (DBFile dbFile : dirtyFiles)
876                fileManager.syncDBFile(dbFile);
877        }
878    }

880    /**
881     * This method removes all cached pages in the specified file from the
882     * buffer manager, writing out any dirty pages in the process.  This method
883     * is not generally recommended to be used, as it basically defeats the
884     * purpose of the buffer manager in the first place; rather, the
885     * {@link #writeDBFile} method should be used instead.  There is a specific
886     * situation in which it is used, when a file is being removed from the
887     * Buffer Manager by the Storage Manager.
888     *
889     * @param dbFile the file whose pages should be flushed from the cache
890     *
891     * @throws IOException if an IO error occurs while updating the write-ahead
892     *         log, or the file's contents
893     */
894    public void flushDBFile(DBFile dbFile) throws IOException {
895        logger.info("Flushing all pages for file " + dbFile +
896            " from the Buffer Manager.");
897
898        Iterator<Map.Entry<CachedPageInfo, DBPage>> entries =
899            cachedPages.entrySet().iterator();
900
901        ArrayList<DBPage> dirtyPages = new ArrayList<>();
902
903        while (entries.hasNext()) {
904            Map.Entry<CachedPageInfo, DBPage> entry = entries.next();
905
906            CachedPageInfo info = entry.getKey();
907            if (dbFile.equals(info.dbFile)) {
908                DBPage oldPage = entry.getValue();
909
910                logger.debug(String.format(
911                    "    Evicting page [%s,%d] from page-cache.",
912                    oldPage.getDBFile(), oldPage.getPageNo()));
913
914                // Remove the page from the cache.
915                entries.remove();
916                totalBytesCached -= oldPage.getPageSize();
917
918                // If the page is dirty, we need to write its data to disk before
919                // invalidating it.  Otherwise, just invalidate it.
920                if (oldPage.isDirty()) {
921                    logger.debug("    Evicted page is dirty; must save to disk.");
922                    dirtyPages.add(oldPage);
923                }
924                else {
925                    oldPage.invalidate();
926                }
927            }
928        }
929
930        writeDirtyPages(dirtyPages, /* invalidate */ true);
931    }

984     * This method removes a file from the cache, first flushing all pages from
985     * the file out of the cache.  This operation is used by the Storage Manager
986     * to close a data file.
987     *
988     * @param dbFile the file to remove from the cache.
989     *

992    public void removeDBFile(DBFile dbFile) throws IOException {

999    /**
1000     * This method removes ALL files from the cache, first flushing all pages
1001     * from the cache so that any dirty pages will be saved to disk (possibly
1002     * updating the write-ahead log in the process).  This operation is used by
1003     * the Storage Manager during shutdown.
1004     *
1005     * @return a list of the files that were in the cache, so that they can be
1006     *         used by the caller if necessary (e.g. to sync and close each one)
1007     *
1008     * @throws IOException if an IO error occurs while writing out dirty pages
1009     */
1010    public List<DBFile> removeAll() throws IOException {
1011        logger.info("Removing ALL DBFiles from buffer manager");
1012
1013        // Flush all pages, ensuring that dirty pages will be written too.
1014        flushAll();
1015
1016        // Get the list of DBFiles we had in the cache, then clear the cache.
1017        ArrayList<DBFile> dbFiles = new ArrayList<>(cachedFiles.values());
1018        cachedFiles.clear();
1019
1020        return dbFiles;
1021    }
