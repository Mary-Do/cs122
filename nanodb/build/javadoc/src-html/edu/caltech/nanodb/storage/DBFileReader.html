001package edu.caltech.nanodb.storage;
002
003
004import org.apache.log4j.Logger;
005
006import java.io.IOException;
007import java.io.UnsupportedEncodingException;
008
009
010/**
011 * <p>
012 * This class provides the basic abilty to read a {@link DBFile} as a single
013 * sequential file, ignoring the fact that it is actually broken into pages.
014 * As the file-pointer moves through the file, the Storage Manager is used to
015 * load individual pages into the buffer manager.
016 * </p>
017 * <p>
018 * It is certainly possible that a value being read might span two adjacent
019 * data pages.  In these cases, the read will be a little slower, as the reader
020 * must read partial data from the first page, and then load and read the
021 * remainder of the data from the next page.
022 * </p>
023 * <p>
024 * There is a subclass of this reader, the {@link DBFileWriter}, which allows
025 * data to be read and written sequentially to a {@link DBFile}.  This is mainly
026 * provided to support write-ahead logging, which is generally a very sequential
027 * operation.
028 * </p>
029 */
030public class DBFileReader {
031    /** A logging object for reporting anything interesting that happens. */
032    private static Logger logger = Logger.getLogger(DBFileReader.class);
033
034
035    /**
036     * A reference to the storage manager, which we will use to read pages
037     * needed for the various operations.
038     */
039    protected StorageManager storageManager;
040
041
042    /** The database file being read by this reader. */
043    protected DBFile dbFile;
044
045
046    /**
047     * A flag controlling whether the file being read should be extended as
048     * it is read.  The reader is expected to not extend the file, but the
049     * {@link DBFileWriter}, a subclass of this class, sets this flag to true.
050     */
051    protected boolean extendFile = false;
052
053
054    /** The page-size of the database file being read from. */
055    protected int pageSize;
056
057
058    /** The last page used for reading the database file. */
059    protected DBPage dbPage;
060    
061    
062    /** The current position in the file where reads will occur from. */
063    protected int position;
064
065
066    /**
067     * This temporary buffer is used to read primitive values that overlap the
068     * boundaries between two pages.
069     */
070    protected byte[] tmpBuf = new byte[8];
071
072
073    public DBFileReader(DBFile dbFile, StorageManager storageManager) {
074        if (dbFile == null)
075            throw new IllegalArgumentException("dbFile cannot be null");
076
077        this.storageManager = storageManager;
078
079        this.dbFile = dbFile;
080        position = 0;
081
082        // The presumption is that page sizes are a power of two.
083        pageSize = dbFile.getPageSize();
084    }
085
086
087    public DBFile getDBFile() {
088        return dbFile;
089    }
090
091
092    /**
093     * Returns the current location in the page where the next operation will
094     * start from.
095     *
096     * @return the current location in the page
097     */
098    public int getPosition() {
099        return position;
100    }
101
102
103    /**
104     * Sets the location in the page where the next operation will start from.
105     *
106     * @param position the new location in the page
107     */
108    public void setPosition(int position) {
109        if (position < 0) {
110            throw new IllegalArgumentException("position must be >= 0, got " +
111                position);
112        }
113
114        this.position = position;
115    }
116
117
118    /**
119     * Move the current position by <tt>n</tt> bytes.  A negative value of
120     * <tt>n</tt> will move the position backward.
121     *
122     * @param n the delta to apply to the current position
123     */
124    public void movePosition(int n) {
125        if (position + n < 0)
126            throw new IllegalArgumentException("can't move position before file start");
127
128        position += n;
129    }
130
131
132    /**
133     * Computes and returns the page-number of the page that the
134     * {@code position} value currently falls within.
135     *
136     * @return the page-number of the page that the {@code position} value
137     *         currently falls within
138     */
139    protected int getPositionPageNo() {
140        return position / pageSize;
141    }
142
143
144    /**
145     * Computes and returns the offset within the current page that the
146     * {@code position} value currently falls at.
147     *
148     * @return the offset within the current page that the {@code position}
149     *         value currently falls at
150     */
151    protected int getPositionPageOffset() {
152        return position % pageSize;
153    }
154
155
156    /**
157     * This helper function checks to see whether the currently cached
158     * {@link DBPage} object matches the page-number of the current
159     * {@link #position} value.  If they don't match, or if there currently
160     * isn't any {@code DBPage} cached, then this method will load the required
161     * {@code DBPage} from the Storage Manager.
162     *
163     * @throws IOException if an IO error occurs while reading the required
164     *         database page
165     */
166    protected void checkDBPage() throws IOException {
167        int pageNo = getPositionPageNo();
168
169        // If the current DBPage is the one we need, use it!
170        if (dbPage != null && dbPage.getPageNo() == pageNo)
171            return;
172
173        // Need to load the required DBPage.
174        dbPage = storageManager.loadDBPage(dbFile, pageNo, extendFile);
175    }
176
177
178    /**
179     * This method causes the writer to set the current {@link DBPage} to
180     * {@code null}, so that the reader doesn't hold on to a page that might
181     * need to be flushed out of the buffer manager.
182     *
183     * @todo Once we have page-pinning, this will also need to unpin the page.
184     */
185    public void releasePage() {
186        dbPage = null;
187    }
188
189
190    /**
191     * Read a sequence of bytes into the provided byte-array, starting with the
192     * specified offset, and reading the specified number of bytes.
193     *
194     * @param b the byte-array to read bytes into
195     *
196     * @param off the offset to read the bytes into the array
197     *
198     * @param len the number of bytes to read into the array
199     */
200    public void read(byte[] b, int off, int len) throws IOException {
201        checkDBPage();
202
203        int pagePosition = getPositionPageOffset();
204        
205        if (pagePosition + len <= pageSize) {
206            dbPage.read(pagePosition, b, off, len);
207            position += len;
208        }
209        else {
210            // Read part of the data from this page, then load the next page and
211            // read the remainder of the data.
212            int page1Len = pageSize - pagePosition;
213            assert page1Len < len;
214            dbPage.read(pagePosition, b, off, page1Len);
215
216            // Load the second page and read the data.  The next page will be
217            // loaded automatically since the position will move forward to the
218            // first byte of the next page.
219            position += page1Len;
220            checkDBPage();
221            dbPage.read(0, b, off + page1Len, len - page1Len);
222
223            position += (len - page1Len);
224        }
225    }
226
227
228    /**
229     * Read a sequence of bytes into the provided byte-array.  The entire array
230     * is filled from start to end.
231     *
232     * @param b the byte-array to read bytes into
233     */
234    public void read(byte[] b) throws IOException {
235        read(b, 0, b.length);
236    }
237
238
239    /**
240     * Reads and returns a Boolean value from the current position.  A zero
241     * value is interpreted as <tt>false</tt>, and a nonzero value is
242     * interpreted as <tt>true</tt>.
243     */
244    public boolean readBoolean() throws IOException {
245        checkDBPage();
246        boolean b = dbPage.readBoolean(getPositionPageOffset());
247        position++;
248        return b;
249    }
250
251
252    /** Reads and returns a signed byte from the current position. */
253    public byte readByte() throws IOException {
254        checkDBPage();
255        byte b = dbPage.readByte(getPositionPageOffset());
256        position++;
257        return b;
258    }
259
260    /**
261     * Reads and returns an unsigned byte from the current position.  The value
262     * is returned as an <tt>int</tt> whose value will be between 0 and 255,
263     * inclusive.
264     */
265    public int readUnsignedByte() throws IOException {
266        checkDBPage();
267        int b = dbPage.readUnsignedByte(getPositionPageOffset());
268        position++;
269        return b;
270    }
271
272
273    /**
274     * Reads and returns an unsigned short from the current position.  The value
275     * is returned as an <tt>int</tt> whose value will be between 0 and 65535,
276     * inclusive.
277     */
278    public int readUnsignedShort() throws IOException {
279        int pagePosition = getPositionPageOffset();
280
281        int value;
282        if (pagePosition + 2 <= pageSize) {
283            checkDBPage();
284            value = dbPage.readUnsignedShort(pagePosition);
285            position += 2;
286        }
287        else {
288            // Need to read the bytes spanning this page and the next.
289            // Note that read() moves the file position forward.
290            read(tmpBuf, 0, 2);
291            value = ((tmpBuf[0] & 0xFF) << 8) | (tmpBuf[1] & 0xFF);
292        }
293
294        return value;
295    }
296
297
298    /** Reads and returns a signed short from the current position. */
299    public short readShort() throws IOException {
300        int pagePosition = getPositionPageOffset();
301
302        short value;
303        if (pagePosition + 2 <= pageSize) {
304            checkDBPage();
305            value = dbPage.readShort(pagePosition);
306            position += 2;
307        }
308        else {
309            // Need to read the bytes spanning this page and the next.
310            // Note that read() moves the file position forward.
311            read(tmpBuf, 0, 2);
312
313            // Don't chop off high-order bits.  When byte is cast to int,
314            // the sign will be extended, so if original byte is negative,
315            // the resulting int will be too.
316            value = (short) ((tmpBuf[0] <<  8) | (tmpBuf[1] & 0xFF));
317        }
318
319        return value;
320    }

323    /** Reads and returns a two-byte char value from the current position. */
324    public char readChar() throws IOException {
325        int pagePosition = getPositionPageOffset();
326
327        char value;
328        if (pagePosition + 2 <= pageSize) {
329            checkDBPage();
330            value = dbPage.readChar(pagePosition);
331            position += 2;
332        }
333        else {
334            // Need to read the bytes spanning this page and the next.
335            // Note that read() moves the file position forward.
336            read(tmpBuf, 0, 2);
337
338            // Don't chop off high-order bits.  When byte is cast to int,
339            // the sign will be extended, so if original byte is negative,
340            // the resulting int will be too.
341            value = (char) ((tmpBuf[0] <<  8) | (tmpBuf[1] & 0xFF));
342        }
343
344        return value;
345    }

348    /**
349     * Reads and returns a four-byte unsigned integer value from the current
350     * position.
351     */
352    public long readUnsignedInt() throws IOException {
353        int pagePosition = getPositionPageOffset();
354
355        long value;
356        if (pagePosition + 4 <= pageSize) {
357            checkDBPage();
358            value = dbPage.readUnsignedInt(pagePosition);
359            position += 4;
360        }
361        else {
362            // Need to read the bytes spanning this page and the next.
363            // Note that read() moves the file position forward.
364            read(tmpBuf, 0, 4);
365            value = ((tmpBuf[0] & 0xFF) << 24) | ((tmpBuf[1] & 0xFF) << 16) |
366                    ((tmpBuf[2] & 0xFF) <<  8) | ((tmpBuf[3] & 0xFF)      );
367        }
368
369        return value;
370    }

373    /** Reads and returns a four-byte integer value from the current position. */
374    public int readInt() throws IOException {
375        int pagePosition = getPositionPageOffset();
376
377        int value;
378        if (pagePosition + 4 <= pageSize) {
379            checkDBPage();
380            value = dbPage.readInt(pagePosition);
381            position += 4;
382        }
383        else {
384            // Need to read the bytes spanning this page and the next.
385            // Note that read() moves the file position forward.
386            read(tmpBuf, 0, 4);
387            value = ((tmpBuf[0] & 0xFF) << 24) | ((tmpBuf[1] & 0xFF) << 16) |
388                    ((tmpBuf[2] & 0xFF) <<  8) | ((tmpBuf[3] & 0xFF)      );
389        }
390
391        return value;
392    }
393
394    /**
395     * Reads and returns an eight-byte long integer value from the current
396     * position.
397     */
398    public long readLong() throws IOException {
399        int pagePosition = getPositionPageOffset();
400
401        long value;
402        if (pagePosition + 8 <= pageSize) {
403            checkDBPage();
404            value = dbPage.readLong(pagePosition);
405            position += 8;
406        }
407        else {
408            // Need to read the bytes spanning this page and the next.
409            // Note that read() moves the file position forward.
410            read(tmpBuf, 0, 8);
411            value = ((long) (tmpBuf[0] & 0xFF) << 56) |
412                    ((long) (tmpBuf[1] & 0xFF) << 48) |
413                    ((long) (tmpBuf[2] & 0xFF) << 40) |
414                    ((long) (tmpBuf[3] & 0xFF) << 32) |
415                    ((long) (tmpBuf[4] & 0xFF) << 24) |
416                    ((long) (tmpBuf[5] & 0xFF) << 16) |
417                    ((long) (tmpBuf[6] & 0xFF) <<  8) |
418                    ((long) (tmpBuf[7] & 0xFF)      );
419        }
420
421        return value;
422    }
423
424    public float readFloat() throws IOException {
425        return Float.intBitsToFloat(readInt());
426    }
427
428    public double readDouble() throws IOException {
429        return Double.longBitsToDouble(readLong());
430    }

433    /**
434     * This method reads and returns a variable-length string whose maximum
435     * length is 255 bytes.  The string is expected to be in US-ASCII encoding,
436     * so multibyte characters are not supported.
437     * <p>
438     * The string's data format is expected to be a single unsigned byte
439     * <em>b</em> specifying the string's length, followed by <em>b</em> more
440     * bytes consisting of the string value itself.
441     */
442    public String readVarString255() throws IOException {
443        int len = readUnsignedByte();
444        byte[] strBytes = new byte[len];
445        read(strBytes);
446
447        String str = null;
448        try {
449            str = new String(strBytes, 0, len, "US-ASCII");
450        }
451        catch (UnsupportedEncodingException e) {
452            // According to the Java docs, the US-ASCII character-encoding is
453            // required to be supported by all JVMs.  So, this is not supposed
454            // to happen.
455            logger.error("The unthinkable has happened:  " + e);
456        }
457
458        return str;
459    }
