025/**
026 * <p>
027 * This class contains a general purpose implementation for reading and
028 * writing a table schema into a data page.  Specific file formats can
029 * customize this representation by subclassing this class, and overriding
030 * the various methods to modify the details they read and write.
031 * </p>
032 * <p>
033 * The schema is written out in the following format:
034 * </p>
035 * <dl>
036 *   <dt>Column Details:</dt>
037 *   <dd>
038 *     <ul>
039 *       <li>Number of Columns (unsigned byte)</li>
040 *       <li>For each column:
041 *         <ul>
042 *           <li>Column's Type (signed byte, converted to {@link SQLDataType} enum)</li>
043 *           <li>If type requires a length:  Column's Max Length (unsigned short)</li>
044 *           <li>Column's Name (a string up to 255 characters, stored as
045 *               {@link PageReader#readVarString255}</li>
046 *         </ul>
047 *       </li>
048 *     </ul>
049 *     Column names and types are checked to be valid at load time.
050 *   </dd>
051 *
052 *   <dt>NOT-NULL Constraints:</dt>
053 *
054 *   <dd>
055 *     <ul>
056 *       <li>Number of NOT-NULL Columns (unsigned byte)</li>
057 *       <li>For each column:
058 *         <ul>
059 *           <li>Index of NOT-NULL Column (unsigned byte)</li>
060 *         </ul>
061 *       </li>
062 *     </ul>
063 *     Every index may only appear once.  Additionally, indexes must reference
064 *     valid columns within the table schema.
065 *   </dd>
066 *
067 *   <dt>Primary/Foreign/Candidate Key Constraints:</dt>
068 *
069 *   <dd>
070 *     <ul>
071 *       <li>Total Number of Primary + Candidate + Foreign Keys (unsigned byte)</li>
072 *       <li>Primary Key (if present):
073 *         <ul>
074 *           <li>Constraint Type, and Name Flag (unsigned byte)</li>
075 *           <li>Constraint Name (if present) (a string up to 255 characters,
076 *               stored as {@link PageReader#readVarString255})</li>
077 *           <li>Number of Columns in the Key (unsigned byte)</li>
078 *           <li>For each column:
079 *             <ul>
080 *               <li>Index of Column in Key (unsigned byte)</li>
081 *             </ul>
082 *           </li>
083 *           <li>Name of Index used to Enforce Key (a string up to 255 characters,
084 *               stored as {@link PageReader#readVarString255})</li>
085 *           <li>Number of Foreign-Key References to this Key (unsigned byte)</li>
086 *           <li>For each FK reference:
087 *             <ul>
088 *               <li>Table Name (a string up to 255 characters, stored as
089 *                   {@link PageReader#readVarString255})</li>
090 *               <li>Index Name (a string up to 255 characters, stored as
091 *                   {@link PageReader#readVarString255})</li>
092 *             </ul>
093 *           </li>
094 *         </ul>
095 *       </li>
096 *       <li>For each candidate key:
097 *         <ul>
098 *           <li>Constraint Type, and Name Flag (unsigned byte)</li>
099 *           <li>Constraint Name (if present) (a string up to 255 characters,
100 *               stored as {@link PageReader#readVarString255})</li>
101 *           <li>Number of Columns in the Key (unsigned byte)</li>
102 *           <li>For each column:
103 *             <ul>
104 *               <li>Index of Column in Key (unsigned byte)</li>
105 *             </ul>
106 *           </li>
107 *           <li>Name of Index used to Enforce Key (a string up to 255 characters,
108 *               stored as {@link PageReader#readVarString255})</li>
109 *           <li>Number of Foreign-Key References to this Key (unsigned byte)</li>
110 *           <li>For each FK reference:
111 *             <ul>
112 *               <li>Table Name (a string up to 255 characters, stored as
113 *                   {@link PageReader#readVarString255})</li>
114 *               <li>Index Name (a string up to 255 characters, stored as
115 *                   {@link PageReader#readVarString255})</li>
116 *             </ul>
117 *           </li>
118 *         </ul>
119 *       </li>
120 *       <li>For each foreign key:
121 *         <ul>
122 *           <li>Constraint Type, and Name Flag (unsigned byte)</li>
123 *           <li>Constraint Name (if present) (a string up to 255 characters,
124 *               stored as {@link PageReader#readVarString255})</li>
125 *           <li>Name of Referenced Table (a string up to 255 characters,
126 *               stored as {@link PageReader#readVarString255})</li>
127 *           <li>On-Delete Cascade Option (unsigned byte)</li>
128 *           <li>On-Update Cascade Option (unsigned byte)</li>
129 *           <li>Number of Columns in the Key (unsigned byte)</li>
130 *           <li>For each column:
131 *             <ul>
132 *               <li>Index of Column in Key (unsigned byte)</li>
133 *               <li>Corresponding Index of Column in Referenced Table
134 *                   (unsigned byte)</li>
135 *             </ul>
136 *           </li>
137 *         </ul>
138 *       </li>
139 *     </ul>
140 *   </dd>
141 *
142 *   <dt>Indexes:</dt>
143 *
144 *   <dd>
145 *     <ul>
146 *       <li>Number of Indexes (unsigned byte)</li>
147 *       <li>For each index:
148 *         <ul>
149 *           <li>Number of Columns in Index (unsigned byte)</li>
150 *           <li>For each column:
151 *             <ul>
152 *               <li>Index of Column in Index (unsigned byte)</li>
153 *             </ul>
154 *           </li>
155 *           <li>Name of Index (a string up to 255 characters,
156 *               stored as {@link PageReader#readVarString255})</li>
157 *         </ul>
158 *       </li>
159 *     </ul>
160 *   </dd>
161 *
162 * </dl>
163 */
164public class SchemaWriter {
165    /** A logging object for reporting anything interesting that happens. */
166    private static Logger logger = Logger.getLogger(SchemaWriter.class);
167
168
169    /**
170     *
171     * @param schema the schema to write to the page
172     * @param pgWriter the {@code PageWriter} that can be used to write to the
173     *        data page
174     */
175    public void writeTableSchema(TableSchema schema, PageWriter pgWriter) {
176        // Write out the schema details now.
177        logger.info("Writing table schema:  " + schema);
178
179        int startPosition = pgWriter.getPosition();
180        logger.debug("Starting at position " + startPosition);
181
182        // Column details:
183        writeColumnInfos(schema, pgWriter);
184
185        // Write all NOT-NULL constraints:
186        writeNotNullConstraints(schema, pgWriter);
187
188        // Write all primary/candidate/foreign key constraints:
189        writeKeyConstraints(schema, pgWriter);
190
191        // Write all details of indexes on this table:
192        writeIndexes(schema, pgWriter);
193
194        // Report how much space was used by schema info.
195        if (logger.isDebugEnabled()) {
196            int size = pgWriter.getPosition() - startPosition;
197            logger.debug("Table schema occupies " + size + " bytes.");
198        }
199    }

202    /**
203     * This helper writes out the column details of the schema.  For each
204     * column, the function writes: the column's type (1 byte), plus any
205     * @param schema
206     * @param pgWriter
207     */
208    protected void writeColumnInfos(TableSchema schema, PageWriter pgWriter) {
209        int startPosition = pgWriter.getPosition();
210
211        // Record all table names that appear in the schema.  (Normally this
212        // will be exactly 1, but we may want to try fun stuff with schemas
213        // in the future...)
214
215        Set<String> tableNames = schema.getTableNames();
216        HashMap<String, Integer> tableNameMapping = new HashMap<String, Integer>();
217        pgWriter.writeByte((byte) tableNames.size());
218        logger.debug("Recording " + tableNames.size() + " table names");
219
220        int tableIndex = 0;
221        for (String tableName : tableNames) {
222            pgWriter.writeVarString255(tableName);
223            tableNameMapping.put(tableName, tableIndex);
224            tableIndex++;
225        }
226
227        // Record information about the columns themselves.
228        logger.debug("Recording " + schema.numColumns() + " columns");
229        pgWriter.writeByte(schema.numColumns());
230        for (ColumnInfo colInfo : schema.getColumnInfos()) {
231            ColumnType colType = colInfo.getType();
232
233            // Each column description consists of a type specification
234            // (including length/precision values, if appropriate), and a
235            // string specifying the column's name.
236
237            // Write the SQL data type and any associated details.
238            pgWriter.writeByte(colType.getBaseType().getTypeID());
239
240            // If this data type requires additional details, write that as well.
241            if (colType.hasLength()) {
242                // CHAR and VARCHAR fields have a 2 byte length value after the type.
243                pgWriter.writeShort(colType.getLength());
244            }
245
246            // Write the index of the table name for the column.
247            pgWriter.writeByte(tableNameMapping.get(colInfo.getTableName()));
248
249            // Write the column name.
250            pgWriter.writeVarString255(colInfo.getName());
251        }
252
253        // Report how much space was used by column information.
254        if (logger.isDebugEnabled()) {
255            int size = pgWriter.getPosition() - startPosition;
256            logger.debug("Column information occupies " + size + " bytes.");
257        }
258    }

261    /**
262     * @param schema
263     * @param pgWriter
264     */
265    protected void writeNotNullConstraints(TableSchema schema,
266                                           PageWriter pgWriter) {
267        int startPosition = pgWriter.getPosition();
268
269        // NOT-NULL columns are specified as a list of column-indexes in the
270        // schema object.  Retrieve a list of these column indexes, and sort
271        // it so that the data in the file will be easier to understand.
272        ArrayList<Integer> notNullCols =
273            new ArrayList<Integer>(schema.getNotNull());
274        Collections.sort(notNullCols);
275
276        int numNotNulls = notNullCols.size();
277        logger.debug("Writing " + numNotNulls + " not-null column constraints.");
278
279        pgWriter.writeByte(numNotNulls);
280        for (int notNullCol : notNullCols) {
281            pgWriter.writeByte(notNullCol);
282        }
283
284        if (logger.isDebugEnabled()) {
285            int size = pgWriter.getPosition() - startPosition;
286            logger.debug("Not-null constraints occupy " + size + " bytes.");
287        }
288    }
289
290
291    protected void writeKeyConstraints(TableSchema schema,
292                                       PageWriter pgWriter) {
293        int startPosition = pgWriter.getPosition();
294
295        // Count how many constraints we have to write.
296        int numConstraints = schema.numCandidateKeys() + schema.numForeignKeys();
297        KeyColumnRefs pk = schema.getPrimaryKey();
298        if (pk != null)
299            numConstraints++;
300
301        logger.debug("Writing " + numConstraints + " constraints");
302        pgWriter.writeByte(numConstraints);
303
304        if (pk != null)
305            writeKey(pgWriter, TableConstraintType.PRIMARY_KEY, pk);
306
307        for (KeyColumnRefs ck : schema.getCandidateKeys())
308            writeKey(pgWriter, TableConstraintType.UNIQUE, ck);
309
310        for (ForeignKeyColumnRefs fk : schema.getForeignKeys())
311            writeForeignKey(pgWriter, fk);
312
313        if (logger.isDebugEnabled()) {
314            int size = pgWriter.getPosition() - startPosition;
315            logger.debug("Constraints occupy " + size + " bytes.");
316        }
317    }

320    /**
321     * This helper function writes a primary key or candidate key to the
322     * schema representation.
323     *
324     * @param pgWriter the writer being used to write the table's schema to its
325     *        header page
326     *
327     * @param type the constraint type, either
328     *        {@link TableConstraintType#PRIMARY_KEY} or
329     *        {@link TableConstraintType#FOREIGN_KEY}.
330     *
331     * @param key a specification of what columns appear in the key
332     *
333     * @throws IllegalArgumentException if the <tt>type</tt> argument is
334     *         <tt>null</tt>, or is not one of the accepted values
335     */
336    protected void writeKey(PageWriter pgWriter, TableConstraintType type,
337                            KeyColumnRefs key) {
338
339        if (type == TableConstraintType.PRIMARY_KEY) {
340            logger.debug(String.format(" * Primary key %s, enforced with " +
341                                       "index %s", key, key.getIndexName()));
342        }
343        else if (type == TableConstraintType.UNIQUE) {
344            logger.debug(String.format(" * Candidate key %s, enforced with " +
345                                       "index %s", key, key.getIndexName()));
346        }
347        else {
348            throw new IllegalArgumentException(
349                "Invalid TableConstraintType value " + type);
350        }
351
352        // Constraint type, and name of constraint if it's specified.
353
354        int typeVal = type.getTypeID();
355        String cName = key.getConstraintName();
356        if (cName != null)
357            typeVal |= 0x80;
358
359        pgWriter.writeByte(typeVal);
360        if (cName != null)
361            pgWriter.writeVarString255(cName);
362
363        // Number of columns in the primary/candidate key
364        pgWriter.writeByte(key.size());
365
366        // The indexes of columns in the primary/candidate key
367        for (int i = 0; i < key.size(); i++)
368            pgWriter.writeByte(key.getCol(i));
369
370        // Name of the index used to enforce this constraint
371        pgWriter.writeVarString255(key.getIndexName());
372
373        // Record all tables and indexes that reference this key.
374        List<KeyColumnRefs.FKReference> fkRefs = key.getReferencingIndexes();
375        pgWriter.writeByte(fkRefs.size());
376        for (KeyColumnRefs.FKReference fkRef : fkRefs) {
377            pgWriter.writeVarString255(fkRef.tableName);
378            pgWriter.writeVarString255(fkRef.indexName);
379        }
380    }

382
383    protected void writeForeignKey(PageWriter pgWriter,
384                                   ForeignKeyColumnRefs key) {
385        logger.debug(" * Foreign key " + key);
386
387        // Constraint type, and name of constraint if it's specified.
388
389        int type = TableConstraintType.FOREIGN_KEY.getTypeID();
390        if (key.getConstraintName() != null)
391            type |= 0x80;
392
393        pgWriter.writeByte(type);
394        if (key.getConstraintName() != null)
395            pgWriter.writeVarString255(key.getConstraintName());
396
397        // Name of the table that the foreign key references
398        pgWriter.writeVarString255(key.getRefTable());
399
400        // Cascade options
401        pgWriter.writeByte(key.getOnDeleteOption().getTypeID());
402        pgWriter.writeByte(key.getOnUpdateOption().getTypeID());
403
404        // Number of columns in the foreign key
405        pgWriter.writeByte(key.size());
406
407        // The indexes of columns in the foreign key, and the corresponding
408        // indexes of columns in the referenced table
409        for (int i = 0; i < key.size(); i++) {
410            pgWriter.writeByte(key.getCol(i));
411            pgWriter.writeByte(key.getRefCol(i));
412        }
413    }

416    protected void writeIndexes(TableSchema schema, PageWriter pgWriter) {
417        int startPosition = pgWriter.getPosition();
418
419        // Count how many indexes we have to write.
420        int numIndexes = schema.getIndexes().size();
421
422        logger.debug("Writing " + numIndexes + " indexes");
423        pgWriter.writeByte(numIndexes);
424
425        Map<String, ColumnRefs> idxMap = schema.getIndexes();
426        for (String indexName : idxMap.keySet())
427            writeIndex(pgWriter, idxMap.get(indexName));
428
429        if (logger.isDebugEnabled()) {
430            int size = pgWriter.getPosition() - startPosition;
431            logger.debug("Indexes occupy " + size + " bytes in the schema");
432        }
433    }

437     * This helper function writes an index to the table's schema stored in the
438     * header page.
439     *
440     * @param hpWriter the writer being used to write the table's schema to its
441     *        header page
442     *
443     * @param idx a specification of what columns appear in the index
444     *
445     * @throws IllegalArgumentException if the <tt>type</tt> argument is
446     *         <tt>null</tt>, or is not one of the accepted values
447     */
448    protected void writeIndex(PageWriter hpWriter, ColumnRefs idx) {
449
450        logger.debug(String.format(" * Index %s, enforced with index %s",
451                                      idx, idx.getIndexName()));
452
453        hpWriter.writeByte(idx.size());
454        for (int i = 0; i < idx.size(); i++)
455            hpWriter.writeByte(idx.getCol(i));
456
457        // This should always be specified.
458        hpWriter.writeVarString255(idx.getIndexName());
459    }

462    /**
463     * This method opens the data file corresponding to the specified table
464     * name and reads in the table's schema.
465     */
466    public TableSchema readTableSchema(PageReader pgReader)
467        throws IOException {
468
469        if (pgReader == null)
470            throw new IllegalArgumentException("pgReader cannot be null");
471
472        TableSchema schema = new TableSchema();
473
474        logger.info("Reading table schema");
475
476        int startPosition = pgReader.getPosition();
477        logger.debug("Starting at position " + startPosition);
478
479        // Column details:
480        readColumnInfos(pgReader, schema);
481
482        // Read in all NOT-NULL constraints
483        readNotNullConstraints(pgReader, schema);
484
485        // Read all details of key constraints, foreign keys, and indexes:
486        readKeyConstraints(pgReader, schema);
487
488        readIndexes(pgReader, schema);
489
490        logger.info("Completed schema:  " + schema);
491        return schema;
492    }

495    protected void readColumnInfos(PageReader pgReader, TableSchema schema)
496        throws IOException {
497
498        // Read in the list of table names for the schema.
499
500        int numTables = pgReader.readUnsignedByte();
501        ArrayList<String> tableNames = new ArrayList<String>(numTables);
502
503        for (int i = 0; i < numTables; i++) {
504            String tableName = pgReader.readVarString255();
505            tableNames.add(tableName);
506        }
507
508        // Read in the column descriptions.
509
510        int numCols = pgReader.readUnsignedByte();
511        logger.debug("Table has " + numCols + " columns.");
512
513        if (numCols == 0)
514            throw new IOException("Table must have at least one column.");
515
516        for (int iCol = 0; iCol < numCols; iCol++) {
517            // Each column description consists of a type specification, a set
518            // of flags (1 byte), and a string specifying the column's name.
519
520            // Get the SQL data type, and begin to build the column's type
521            // with that.
522
523            byte sqlTypeID = pgReader.readByte();
524
525            SQLDataType baseType = SQLDataType.findType(sqlTypeID);
526            if (baseType == null) {
527                throw new IOException("Unrecognized SQL type " + sqlTypeID +
528                                      " for column " + iCol);
529            }
530
531            ColumnType colType = new ColumnType(baseType);
532
533            // If this data type requires additional details, read that as well.
534            if (colType.hasLength()) {
535                // CHAR and VARCHAR fields have a 2 byte length value after
536                // the type.
537                colType.setLength(pgReader.readUnsignedShort());
538            }
539
540            // Read the table-name index for the column.
541            int tableIndex = pgReader.readUnsignedByte();
542            String tableName = tableNames.get(tableIndex);
543
544            // Read and verify the column name.
545
546            String colName = pgReader.readVarString255();
547
548            if (colName.length() == 0) {
549                throw new IOException("Name of column " + iCol +
550                                      " is unspecified.");
551            }
552
553            for (int iCh = 0; iCh < colName.length(); iCh++) {
554                char ch = colName.charAt(iCh);
555
556                if (iCh == 0 && !(Character.isLetter(ch) || ch == '_' || ch == '#') ||
557                    iCh > 0 && !(Character.isLetterOrDigit(ch) || ch == '_')) {
558                    throw new IOException(String.format("Name of column " +
559                        "%d \"%s\" has an invalid character at index %d.",
560                        iCol, colName, iCh));
561                }
562            }
563
564            ColumnInfo colInfo = new ColumnInfo(colName, tableName, colType);
565
566            logger.debug(colInfo);
567
568            schema.addColumnInfo(colInfo);
569        }
570    }

573    protected void readNotNullConstraints(PageReader pgReader,
574                                          TableSchema schema) {
575        int numNotNulls = pgReader.readUnsignedByte();
576        logger.debug("Reading " + numNotNulls + " not-null constraints");
577
578        for (int i = 0; i < numNotNulls; i++) {
579            int index = pgReader.readUnsignedByte();
580            if (!schema.addNotNull(index)) {
581                throw new IllegalStateException("Index " + index +
582                    " appears multiple times in NOT-NULL specification!");
583            }
584        }
585    }
586
587
588    protected void readKeyConstraints(PageReader pgReader, TableSchema schema)
589        throws IOException {
590
591        int numConstraints = pgReader.readUnsignedByte();
592        logger.debug("Reading " + numConstraints + " constraints");
593
594        for (int i = 0; i < numConstraints; i++) {
595            int cTypeID = pgReader.readUnsignedByte();
596            TableConstraintType cType =
597                TableConstraintType.findType((byte) (cTypeID & 0x7F));
598            if (cType == null)
599                throw new IOException("Unrecognized constraint-type value " + cTypeID);
600
601            KeyColumnRefs key;
602            switch (cType) {
603                case PRIMARY_KEY:
604                    key = readKey(pgReader, cTypeID,
605                                  TableConstraintType.PRIMARY_KEY);
606                    schema.setPrimaryKey(key);
607                    break;
608
609                case UNIQUE:
610                    key = readKey(pgReader, cTypeID,
611                                  TableConstraintType.UNIQUE);
612                    schema.addCandidateKey(key);
613                    break;
614
615                case FOREIGN_KEY:
616                    ForeignKeyColumnRefs fkey =
617                        readForeignKey(pgReader, cTypeID);
618                    schema.addForeignKey(fkey);
619                    break;
620
621                default:
622                    throw new IOException(
623                        "Encountered unrecognized constraint type " + cType);
624            }
625        }
626    }

629    /**
630     * This helper function writes a primary key or candidate key to the table's
631     * schema stored in the header page.
632     *
633     * @param pgReader the writer being used to write the table's schema to its
634     *        header page
635     *
636     * @param typeID the unsigned-byte value read from the table's header page,
637     *        corresponding to this key's type.  Although this value is already
638     *        parsed before calling this method, it also contains flags that
639     *        this method handles, so it must be passed in as well.
640     *
641     * @param type the constraint type, either
642     *        {@link TableConstraintType#PRIMARY_KEY} or
643     *        {@link TableConstraintType#FOREIGN_KEY}.
644     *
645     * @return a specification of the key, including its name, what columns
646     *         appear in the key, and what index is used to enforce the key
647     *
648     * @throws IllegalArgumentException if the <tt>type</tt> argument is
649     *         <tt>null</tt>, or is not one of the accepted values
650     */
651    protected KeyColumnRefs readKey(PageReader pgReader, int typeID,
652                                       TableConstraintType type) {
653
654        if (type == TableConstraintType.PRIMARY_KEY) {
655            logger.debug(" * Reading primary key");
656        }
657        else if (type == TableConstraintType.UNIQUE) {
658            logger.debug(" * Reading candidate key");
659        }
660        else {
661            throw new IllegalArgumentException(
662                "Unhandled TableConstraintType value " + type);
663        }
664
665        // The constraint may or may not be named; if so, read in the name.
666        String constraintName = null;
667        if ((typeID & 0x80) != 0)
668            constraintName = pgReader.readVarString255();
669
670        // The indexes of the columns in the key.
671        int keySize = pgReader.readUnsignedByte();
672        int[] keyCols = new int[keySize];
673        for (int i = 0; i < keySize; i++)
674            keyCols[i] = pgReader.readUnsignedByte();
675
676        // This should always be specified.
677        String indexName = pgReader.readVarString255();
678
679        KeyColumnRefs key = new KeyColumnRefs(indexName, keyCols, type);
680        key.setConstraintName(constraintName);
681
682        // Read in all tables and indexes that reference this key.
683        int refSize = pgReader.readUnsignedByte();
684        for (int i = 0; i < refSize; i++) {
685            // First string value is table name, second one is index name.
686            key.addRef(pgReader.readVarString255(),
687                       pgReader.readVarString255());
688        }
689        return key;
690    }

693    protected ForeignKeyColumnRefs readForeignKey(PageReader pgReader, int typeID) {
694        logger.debug(" * Reading foreign key");
695
696        // Read the constraint's name, if it's specified.
697
698        String constraintName = null;
699        if ((typeID & 0x80) != 0)
700            constraintName = pgReader.readVarString255();
701
702        // Name of the table that the foreign key references
703        String refTableName = pgReader.readVarString255();
704
705        // Cascade options.
706        // TODO:  Verify that the options are valid.  We kinda assume it here.
707        ForeignKeyValueChangeOption onDeleteOption =
708            ForeignKeyValueChangeOption.findType(pgReader.readUnsignedByte());
709        ForeignKeyValueChangeOption onUpdateOption =
710            ForeignKeyValueChangeOption.findType(pgReader.readUnsignedByte());
711        int keySize = pgReader.readUnsignedByte();
712
713        // The indexes of columns in the foreign key, and the corresponding
714        // indexes of columns in the referenced table
715        int[] keyCols = new int[keySize];
716        int[] refCols = new int[keySize];
717        for (int i = 0; i < keySize; i++) {
718            keyCols[i] = pgReader.readUnsignedByte();
719            refCols[i] = pgReader.readUnsignedByte();
720        }
721
722        ForeignKeyColumnRefs fk = new ForeignKeyColumnRefs(
723            keyCols, refTableName, refCols, onDeleteOption, onUpdateOption);
724        fk.setConstraintName(constraintName);
725
726        return fk;
727    }
728
729
730    protected void readIndexes(PageReader pgReader, TableSchema schema) {
731        int numIndexes = pgReader.readUnsignedByte();
732        logger.debug("Reading " + numIndexes + " indexes");
733        for (int i = 0; i < numIndexes; i++)
734            schema.addIndex(readIndex(pgReader));
735    }

738    /**
739     * This helper function reads an index to the table's schema stored
740     * in the header page.
741     *
742     * @param pgReader the reader being used to read the table's schema to its
743     *        header page
744     *
745     * @return a specification of the index, including its name, what columns
746     *         appear in the index
747     *
748     * @throws IllegalArgumentException if the <tt>type</tt> argument is
749     *         <tt>null</tt>, or is not one of the accepted values
750     */
751    protected ColumnRefs readIndex(PageReader pgReader) {
752
753        logger.debug(" * Reading index");
754
755        int idxSize = pgReader.readUnsignedByte();
756        int[] idxCols = new int[idxSize];
757        for (int i = 0; i < idxSize; i++)
758            idxCols[i] = pgReader.readUnsignedByte();
759
760        // This should always be specified.
761        String indexName = pgReader.readVarString255();
762
763        return new ColumnRefs(indexName, idxCols);
764    }
