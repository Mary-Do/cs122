024/**
025 * <p>
026 * This class manages the write-ahead logs of the database.  There are methods
027 * to write the different kinds of log records, and also to perform recovery
028 * based on the contents of the write-ahead logs.  Note that many details of
029 * transaction coordination are handled by the Transaction Manager, such as
030 * syncing the write-ahead log, forcing the WAL, and so forth.
031 * </p>
032 * <p>
033 * The actual details of the write-ahead log format are in the package Javadocs:
034 * {@link edu.caltech.nanodb.storage.writeahead}.
035 * </p>
036 * <p>
037 * Some of the {@code writeXXXX()} methods require explicit transaction details,
038 * while others retrieve the transaction state from thread-local storage.  The
039 * main difference is that methods that require explicit transaction details are
040 * needed during recovery processing, when transaction state is dictated by the
041 * log file, not what is in thread-local storage.
042 * </p>
043 */
044public class WALManager {

164        }

180    /**
181     * Performs recovery processing starting at the specified log sequence
182     * number, and returns the LSN where the next recovery process should start
183     * from.
184     *
185     * @param storedFirstLSN the location of the write-ahead log record where
186     *        recovery should start from
187     *
188     * @param storedNextLSN the location in the write-ahead log that is
189     *        <em>just past</em> the last valid log record in the WAL
190     *
191     * @return the new location where recovery should start from the next time
192     *         recovery processing is performed
193     *         
194     * @throws IOException if an IO error occurs during recovery processing
195     */
196    public RecoveryInfo doRecovery(LogSequenceNumber storedFirstLSN,
197        LogSequenceNumber storedNextLSN) throws IOException {
198
199        firstLSN = storedFirstLSN;
200        nextLSN = storedNextLSN;
201        RecoveryInfo recoveryInfo = new RecoveryInfo(firstLSN, nextLSN);
202
203        if (firstLSN.equals(nextLSN)) {
204            // No recovery necessary!  Just return the passed-in info.
205            return recoveryInfo;
206        }
207
208        performRedo(recoveryInfo);
209        performUndo(recoveryInfo);
210
211        TransactionManager txnMgr = storageManager.getTransactionManager();
212
213        // Force the WAL out, up to the nextLSN value.  Then, write all dirty
214        // data pages, and sync all of the affected files.
215        txnMgr.forceWAL(nextLSN);
216        bufferManager.writeAll(true);
217
218        // At this point, all files in the database should be in sync with
219        // the entirety of the write-ahead log.  So, update the firstLSN value
220        // and update the transaction state file again.  (This won't write out
221        // any WAL records, but it will write and sync the txn-state file.)
222        firstLSN = nextLSN;
223        txnMgr.forceWAL(nextLSN);
224
225        recoveryInfo.firstLSN = firstLSN;
226        recoveryInfo.nextLSN = nextLSN;
227
228        return recoveryInfo;
229    }

232    /**
233     * This helper function performs redo processing using the write-ahead
234     * log.  As the log is traversed, the <tt>RecoveryInfo</tt> object is also
235     * updated with important redo/undo information.
236     *
237     * @param recoveryInfo the object used to track information about specific
238     *        transactions during recovery processing.  This object will be
239     *        passed to {@link #performUndo}.
240     */
241    private void performRedo(RecoveryInfo recoveryInfo) throws IOException {
242        LogSequenceNumber currLSN = recoveryInfo.firstLSN;
243        logger.debug("Starting redo processing at LSN " + currLSN);
244
245        LogSequenceNumber oldLSN = null;
246        DBFileReader walReader = null;
247        while (currLSN.compareTo(recoveryInfo.nextLSN) < 0) {
248            if (oldLSN == null || oldLSN.getLogFileNo() != currLSN.getLogFileNo())
249                walReader = getWALFileReader(currLSN);
250
251            // Read the parts of the log record that are always the same.
252            byte typeID = walReader.readByte();
253            WALRecordType type = WALRecordType.valueOf(typeID);
254
255            int transactionID = walReader.readInt();
256
257            logger.debug(String.format(
258                "Redo:  examining WAL record at %s.  Type = %s, TxnID = %d",
259                currLSN, type, transactionID));
260
261            // TODO:  IMPLEMENT THE REST
262            //
263            //        Use logging statements liberally to help verify and
264            //        debug your work.
265            //
266            //        If you encounter invalid WAL contents, throw a
267            //        WALFileException to indicate the problem immediately.
268            //
269            //        You can use Java enums in a switch statement, like this:
270            //
271            //            switch (type) {
272            //            case START_TXN:
273            //                ...
274            //
275            //            case COMMIT_TXN:
276            //                ...
277            //
278            //            default:
279            //                throw new WALFileException(
280            //                    "Encountered unrecognized WAL record type " +
281            //                    type + " at LSN " + currLSN +
282            //                    " during redo processing!");
283            //            }
284
285            oldLSN = currLSN;
286            currLSN = computeNextLSN(currLSN.getLogFileNo(), walReader.getPosition());
287        }
288
289        if (currLSN.compareTo(recoveryInfo.nextLSN) != 0) {
290            throw new WALFileException("Traversing WAL file didn't yield " +
291                " the same ending LSN as in the transaction-state file.  WAL " +
292                " result:  " + currLSN + "  TxnState:  " + recoveryInfo.nextLSN);
293        }
294
295        logger.debug("Redo processing is complete.  There are " +
296            recoveryInfo.incompleteTxns.size() + " incomplete transactions.");
297    }

300    /**
301     * This helper function performs undo processing using the write-ahead
302     * log.  As the log is traversed backwards, the <tt>RecoveryInfo</tt>
303     * object is also updated with important redo/undo information.
304     *
305     * @param recoveryInfo the object used to track information about specific
306     *        transactions during recovery processing.  This object is
307     *        populated by a previous call to {@link #performRedo}.
308     */
309    private void performUndo(RecoveryInfo recoveryInfo) throws IOException {
310        LogSequenceNumber currLSN = recoveryInfo.nextLSN;
311        logger.debug("Starting undo processing at " + currLSN);
312
313        LogSequenceNumber oldLSN = null;
314        DBFileReader walReader = null;
315        while (recoveryInfo.hasIncompleteTxns()) {
316            // Compute LSN of previous WAL record.  Start by getting the last
317            // byte of the previous WAL record.
318            int logFileNo = currLSN.getLogFileNo();
319            int fileOffset = currLSN.getFileOffset();
320
321            logger.debug("Finding record that comes before " + currLSN);
322
323            // Wrap to the previous WAL file if necessary, and if there is one.
324            if (fileOffset == OFFSET_FIRST_RECORD) {
325                // Need to read the "previous WAL file's last offset" value
326                // from the current WAL file.
327                walReader = getWALFileReader(currLSN);
328                walReader.setPosition(OFFSET_PREV_FILE_END);
329                int prevFileEndOffset = walReader.readInt();
330                if (prevFileEndOffset == 0) {
331                    logger.debug("Reached the very start of the write-ahead log!");
332                    break;
333                }
334
335                // Need to go back to the previous WAL file.
336                logFileNo--;
337                if (logFileNo < 0)  // Did we wrap around?
338                    logFileNo = MAX_WAL_FILE_NUMBER;
339
340                currLSN = new LogSequenceNumber(logFileNo, prevFileEndOffset);
341                fileOffset = currLSN.getFileOffset();
342            }
343            else if (fileOffset < OFFSET_FIRST_RECORD) {
344                // This would be highly unusual, but would indicate either a
345                // bug in the undo record-traversal, or a corrupt WAL file.
346                throw new WALFileException(String.format("Overshot the start " +
347                    "of WAL file %d's records; ended up at file-position %d",
348                    logFileNo, fileOffset));
349            }
350
351            if (currLSN.compareTo(recoveryInfo.firstLSN) <= 0)
352                break;
353
354            if (oldLSN == null || oldLSN.getLogFileNo() != logFileNo)
355                walReader = getWALFileReader(currLSN);
356            else
357                walReader.setPosition(currLSN.getFileOffset());
358
359            // Move backward one byte in the WAL file to read the previous
360            // record's type ID.
361            walReader.movePosition(-1);
362            byte typeID = walReader.readByte();
363            WALRecordType type = WALRecordType.valueOf(typeID);
364
365            // Compute the start of the previous record based on its type and
366            // other details.
367            int startOffset;
368            switch (type) {
369            case START_TXN:
370                // Type (1B) + TransactionID (4B) + Type (1B) = 6 bytes
371                startOffset = fileOffset - 6;
372                break;
373
374            case COMMIT_TXN:
375            case ABORT_TXN:
376                // Type (1B) + TransactionID (4B) + PrevLSN (2B+4B) + Type (1B)
377                // = 12 bytes
378                startOffset = fileOffset - 12;
379                break;
380
381            case UPDATE_PAGE:
382            case UPDATE_PAGE_REDO_ONLY:
383                // For these records, the WAL record's start offset is stored
384                // immediately before the last type-byte.  We go back 5 bytes
385                // because reading the type ID moves the position forward by
386                // 1 byte, and then we also have to get to the start of the
387                // 4-byte starting offset.
388                walReader.movePosition(-5);
389                startOffset = walReader.readInt();
390                break;
391
392            default:
393                throw new WALFileException(
394                    "Encountered unrecognized WAL record type " + type +
395                        " at LSN " + currLSN + " during redo processing!");
396            }
397
398            // Construct a new LSN pointing to the previous record.  If this
399            // happens to be before the range that we are using for recovery,
400            // we're done with undo-processing.
401            currLSN = new LogSequenceNumber(logFileNo, startOffset);
402            if (currLSN.compareTo(recoveryInfo.firstLSN) < 0)
403                break;
404
405            // Skip over the "record type" byte, which is at startOffset.
406            // This sets up to read the transaction ID, next.
407            walReader.setPosition(startOffset + 1);
408
409            // Read the transaction ID.
410            int transactionID = walReader.readInt();
411            if (recoveryInfo.isTxnComplete(transactionID)) {
412                // The current transaction is already completed, so skip the
413                // record.
414                oldLSN = currLSN;
415                continue;
416            }
417
418            // Undo specific operations.  Note that we don't have to set the
419            // reader's position to anything special at the end of each record,
420            // since the above code will always properly move to the appropriate
421            // position for the previous record, based on the value of currLSN.
422
423            logger.debug(String.format(
424                "Undo:  examining WAL record at %s.  Type = %s, TxnID = %d",
425                currLSN, type, transactionID));
426
427            // TODO:  IMPLEMENT THE REST
428            //
429            //        Use logging statements liberally to help verify and
430            //        debug your work.
431            //
432            //        If you encounter invalid WAL contents, throw a
433            //        WALFileException to indicate the problem immediately.
434            //
435            //        You can use Java enums in a switch statement, like this:
436            //
437            //            switch (type) {
438            //            case START_TXN:
439            //                ...
440            //
441            //            case COMMIT_TXN:
442            //                ...
443            //
444            //            default:
445            //                throw new WALFileException(
446            //                    "Encountered unrecognized WAL record type " +
447            //                    type + " at LSN " + currLSN +
448            //                    " during undo processing!");
449            //            }
450
451            oldLSN = currLSN;
452        }
453
454        logger.debug("Undo processing is complete.");
455    }

458    /**
459     * This static helper function takes the file number of a WAL file, and
460     * the offset in the WAL file where the next write-ahead log record would
461     * go if the WAL file can hold more data, and then creates a new
462     * {@code LogSequenceNumber} object, wrapping to the next file if
463     * necessary.
464     *
465     * @param fileNo the number of the current WAL file
466     * @param fileOffset the offset of where the next write-ahead log record
467     *        would go, in the absence of wrapping to the next file
468     * @return a {@code LogSequenceNumber} object that takes wrapping into
469     *         account
470     */
471    public static LogSequenceNumber computeNextLSN(int fileNo, int fileOffset) {
472        if (fileOffset >= MAX_WAL_FILE_SIZE) {
473            // This WAL file has reached the size limit.  Increment the file
474            // number, wrapping around if necessary, and reset the offset to 0.
475            fileNo += 1;
476            if (fileNo > MAX_WAL_FILE_NUMBER)
477                fileNo = 0;
478
479            // Need to make sure we skip past the file type and size at the
480            // start of the data file.
481            fileOffset = OFFSET_FIRST_RECORD;
482        }
483        return new LogSequenceNumber(fileNo, fileOffset);
484    }

487    /**
488     * This method opens the WAL file specified in the passed-in Log Sequence
489     * Number, wraps it with a {@link DBFileWriter} so that it can be read and
490     * written, and then seeks to the specified file offset.
491     *
492     * Because we are writing, the file may not yet exist.  In that case, this
493     * method will also create a new WAL file for the specified file number,
494     * initializing it with the proper values, and then seeking to the location
495     * where the first WAL record may be written.
496     *
497     * @param lsn The log sequence number specifying the WAL file and the offset
498     *            in the WAL file to go to.
499     *
500     * @return the WAL file, with the file position moved to the specified
501     *         offset.
502     *
503     * @throws IOException if the corresponding WAL file cannot be opened, or
504     *         if some other IO error occurs.
505     */
506    private DBFileWriter getWALFileWriter(LogSequenceNumber lsn)
507        throws IOException {
508
509        int fileNo = lsn.getLogFileNo();
510        int offset = lsn.getFileOffset();
511
512        DBFile walFile;
513        try {
514            walFile = openWALFile(fileNo);
515        }
516        catch (FileNotFoundException e) {
517            logger.debug("WAL file doesn't exist!  WAL is expanding into a new file.");
518            walFile = createWALFile(fileNo);
519            // TODO:  Write the previous WAL file's last file-offset into the new WAL file's start.
520        }
521
522        DBFileWriter writer = new DBFileWriter(walFile, storageManager);
523        writer.setPosition(offset);
524
525        return writer;
526    }

561    /**
562     * This function writes a transaction demarcation record
563     * ({@link WALRecordType#START_TXN}, {@link WALRecordType#COMMIT_TXN}, or
564     * {@link WALRecordType#ABORT_TXN}) to the write-ahead log.  The transaction
565     * state is passed explicitly so that this method can be used during
566     * recovery processing.  The alternate method
567     * {@link #writeTxnRecord(WALRecordType)} retrieves the transaction state
568     * from thread-local storage, and should be used during normal operation.
569     *
570     * @param type The type of the transaction demarcation to write, one of the
571     *        values {@link WALRecordType#START_TXN}, {@link WALRecordType#COMMIT_TXN},
572     *        or {@link WALRecordType#ABORT_TXN}.
573     *
574     * @param transactionID the transaction ID that the WAL record is for
575     *
576     * @param prevLSN the log sequence number of the transaction's immediately
577     *        previous WAL record, if the record type is either a commit or
578     *        abort record.
579     *
580     * @return the Log Sequence Number of the WAL record that was written
581     *
582     * @throws IOException if the write-ahead log can't be updated for some
583     *         reason.
584     *
585     * @throws IllegalArgumentException if <tt>type</tt> is <tt>null</tt>, or if
586     *         it isn't one of the values {@link WALRecordType#START_TXN},
587     *         {@link WALRecordType#COMMIT_TXN}, or {@link WALRecordType#ABORT_TXN}.
588     */
589    public LogSequenceNumber writeTxnRecord(WALRecordType type,
590        int transactionID, LogSequenceNumber prevLSN) throws IOException {
591
592        if (type != WALRecordType.START_TXN &&
593            type != WALRecordType.COMMIT_TXN &&
594            type != WALRecordType.ABORT_TXN) {
595            throw new IllegalArgumentException("Invalid record type " + type +
596                " passed to writeTxnRecord().");
597        }
598
599        if ((type == WALRecordType.COMMIT_TXN ||
600             type == WALRecordType.ABORT_TXN) && prevLSN == null) {
601            throw new IllegalArgumentException(
602                "prevLSN must be specified for records of type " + type);
603        }
604
605        LogSequenceNumber lsn = nextLSN;
606
607        logger.debug("Writing a " + type + " record for transaction " +
608            transactionID + " at LSN " + lsn);
609
610        // Record the WAL record.  First thing to do:  figure out where it goes.
611
612        DBFileWriter walWriter = getWALFileWriter(lsn);
613
614        walWriter.writeByte(type.getID());
615        walWriter.writeInt(transactionID);
616
617        if (type == WALRecordType.START_TXN) {
618            walWriter.writeByte(type.getID());
619
620            // TypeID (1B) + TransactionID (4B) + TypeID (1B)
621            lsn.setRecordSize(6);
622        }
623        else {
624            walWriter.writeShort(prevLSN.getLogFileNo());
625            walWriter.writeInt(prevLSN.getFileOffset());
626            walWriter.writeByte(type.getID());
627
628            // TypeID (1B) + TransactionID (4B) + PrevLSN (6B) + TypeID (1B)
629            lsn.setRecordSize(12);
630        }
631
632        nextLSN = computeNextLSN(nextLSN.getLogFileNo(), walWriter.getPosition());
633        logger.debug("Next-LSN value is now " + nextLSN);
634
635        return lsn;
636    }

639    /**
640     * This function writes a transaction demarcation record
641     * ({@link WALRecordType#START_TXN}, {@link WALRecordType#COMMIT_TXN}, or
642     * {@link WALRecordType#ABORT_TXN}) to the write-ahead log.  The transaction
643     * state is retrieved from thread-local storage so that it doesn't need to
644     * be passed.
645     *
646     * @param type The type of the transaction demarcation to write, one of the
647     *        values {@link WALRecordType#START_TXN}, {@link WALRecordType#COMMIT_TXN},
648     *        or {@link WALRecordType#ABORT_TXN}.
649     *
650     * @return the Log Sequence Number of the WAL record that was written
651     *
652     * @throws IOException if the write-ahead log can't be updated for some
653     *         reason.
654     *
655     * @throws IllegalArgumentException if <tt>type</tt> is <tt>null</tt>, or if
656     *         it isn't one of the values {@link WALRecordType#START_TXN},
657     *         {@link WALRecordType#COMMIT_TXN}, or {@link WALRecordType#ABORT_TXN}.
658     */
659    public LogSequenceNumber writeTxnRecord(WALRecordType type)
660        throws IOException {
661
662        // Retrieve and verify the transaction state.
663        TransactionState txnState = SessionState.get().getTxnState();
664        if (!txnState.isTxnInProgress()) {
665            throw new IllegalStateException(
666                "No transaction is currently in progress!");
667        }
668
669        LogSequenceNumber lsn = writeTxnRecord(type, txnState.getTransactionID(),
670            txnState.getLastLSN());
671
672        txnState.setLastLSN(lsn);
673
674        return lsn;
675    }

678    /**
679     * This method writes an update-page record to the write-ahead log,
680     * including both undo and redo details.
681     *
682     * @param dbPage The data page whose changes are to be recorded in the log.
683     *
684     * @return the Log Sequence Number of the WAL record that was written
685     *
686     * @throws IOException if the write-ahead log cannot be updated for some
687     *         reason.
688     * 
689     * @throws IllegalArgumentException if <tt>dbPage</tt> is <tt>null</tt>, or
690     *         if it shows no updates.
691     */
692    public LogSequenceNumber writeUpdatePageRecord(DBPage dbPage)
693        throws IOException {
694
695        if (dbPage == null)
696            throw new IllegalArgumentException("dbPage must be specified");
697
698        if (!dbPage.isDirty())
699            throw new IllegalArgumentException("dbPage has no updates to store");
700
701        // Retrieve and verify the transaction state.
702        TransactionState txnState = SessionState.get().getTxnState();
703        if (!txnState.isTxnInProgress()) {
704            throw new IllegalStateException(
705                "No transaction is currently in progress!");
706        }
707
708        LogSequenceNumber lsn = nextLSN;
709
710        logger.debug(String.format("Writing an %s record for transaction %d at LSN %s",
711            WALRecordType.UPDATE_PAGE, txnState.getTransactionID(), lsn));
712
713        // Record the WAL record.  First thing to do:  figure out where it goes.
714
715        DBFileWriter walWriter = getWALFileWriter(lsn);
716
717        walWriter.writeByte(WALRecordType.UPDATE_PAGE.getID());
718        walWriter.writeInt(txnState.getTransactionID());
719
720        // We need to store the previous log sequence number for this record.
721        LogSequenceNumber prevLSN = txnState.getLastLSN();
722        walWriter.writeShort(prevLSN.getLogFileNo());
723        walWriter.writeInt(prevLSN.getFileOffset());
724
725        // Store the filename and page number that is being updated.
726        walWriter.writeVarString255(dbPage.getDBFile().getDataFile().getName());
727        walWriter.writeShort(dbPage.getPageNo());
728
729        // This offset is where we will store the number of data segments we
730        // need to record.  We don't know the value until later, so remember
731        // the position and fill it in later.
732        int segCountOffset = walWriter.getPosition();
733        walWriter.writeShort(-1);
734        
735        byte[] oldData = dbPage.getOldPageData();
736        byte[] newData = dbPage.getPageData();
737        int pageSize = dbPage.getPageSize();
738
739        // DEBUG:  Show changes from old version of page to new version of page.
740        // logger.debug("DBPage changes:\n" + dbPage.getChangesAsString());
741
742        int numSegments = 0;
743        int index = 0;
744        while (index < pageSize) {
745            logger.debug("Skipping identical bytes starting at index " + index);
746            
747            // Skip data until we find stuff that's different.
748            index += ArrayUtil.sizeOfIdenticalRange(oldData, newData, index);
749            assert index <= pageSize;
750            if (index == pageSize)
751                break;
752
753            logger.debug("Recording changed bytes starting at index " + index);
754
755            // Find out how much data is actually changed.  We lump in small
756            // runs of unchanged data just to make things more efficient.
757            int size = 0;
758            while (index + size < pageSize) {
759                size += ArrayUtil.sizeOfDifferentRange(oldData, newData,
760                    index + size);
761                assert index + size <= pageSize;
762                if (index + size == pageSize)
763                    break;
764
765                // If there are 4 or less identical bytes after the different
766                // bytes, include them in this segment.
767                int sameSize = ArrayUtil.sizeOfIdenticalRange(oldData, newData,
768                    index + size);
769
770                if (sameSize > 4 || index + size + sameSize == pageSize)
771                    break;
772
773                size += sameSize;
774            }
775
776            logger.debug("Found " + size + " changed bytes starting at index " +
777                index);
778
779            // Write the starting index within the page, and the amount of
780            // data that will be recorded at that index.
781            walWriter.writeShort(index);
782            walWriter.writeShort(size);
783
784            // Write the old data (undo), and then the new data (redo).
785            walWriter.write(oldData, index, size);
786            walWriter.write(newData, index, size);
787
788            numSegments++;
789
790            index += size;
791        }
792        assert index == pageSize;
793
794        // Now that we know how many segments were recorded, store that value
795        // at the appropriate location.
796        int currOffset = walWriter.getPosition();
797        walWriter.setPosition(segCountOffset);
798        walWriter.writeShort(numSegments);
799        walWriter.setPosition(currOffset);
800
801        // Write the start of the update record at the end so that we can get
802        // back to the record's start when scanning the log backwards.
803
804        walWriter.writeInt(lsn.getFileOffset());
805        walWriter.writeByte(WALRecordType.UPDATE_PAGE.getID());
806
807        // Store the LSN of the change on the page.
808        lsn.setRecordSize(walWriter.getPosition() - lsn.getFileOffset());
809        dbPage.setPageLSN(lsn);
810        dbPage.syncOldPageData();
811
812        // Since we issued a new write-ahead log record for the current
813        // transaction, update the "last LSN" value for the transaction.
814        txnState.setLastLSN(lsn);
815
816        nextLSN = computeNextLSN(nextLSN.getLogFileNo(), walWriter.getPosition());
817
818        return lsn;
819    }

822    /**
823     * This helper function writes a sequence of redo-segments from an
824     * {@link WALRecordType#UPDATE_PAGE} or
825     * {@link WALRecordType#UPDATE_PAGE_REDO_ONLY} record.  Note that the
826     * {@code walReader} argument is expected to be positioned at the start of
827     * the segments containing the old and new versions of the page data (or
828     * just the new versions, for redo-only records).  Additionally, the
829     * reader position will be advanced by this method.
830     *
831     * @param type The record type, either {@link WALRecordType#UPDATE_PAGE}
832     *        or {@link WALRecordType#UPDATE_PAGE_REDO_ONLY}.
833     *
834     * @param walReader A reader positioned at the start of the redo/undo data
835     *        to apply to the data page.  This method will advance the reader's
836     *        position past this redo/undo data.
837     *
838     * @param dbPage the page that the redo should be applied to
839     * @param numSegments the number of segments containing redo[/undo] data;
840     *        this value is expected to already be unpacked from the log record
841     * @throws IOException
842     */
843    private void applyRedo(WALRecordType type, DBFileReader walReader,
844                           DBPage dbPage, int numSegments) throws IOException {
845
846        if (type != WALRecordType.UPDATE_PAGE &&
847            type != WALRecordType.UPDATE_PAGE_REDO_ONLY) {
848            throw new IllegalArgumentException("This method can only be " +
849                "used with UPDATE_PAGE and UPDATE_PAGE_REDO_ONLY records.");
850        }
851
852        for (int iSeg = 0; iSeg < numSegments; iSeg++) {
853            // Write the starting index within the page, and the amount of
854            // data that will be recorded at that index.
855            int index = walReader.readUnsignedShort();
856            int size = walReader.readUnsignedShort();
857
858            // If it's an UPDATE_PAGE record, skip over the undo data.
859            if (type == WALRecordType.UPDATE_PAGE)
860                walReader.movePosition(size);
861
862            // Write the redo data into the page.
863            byte[] redoData = new byte[size];
864            walReader.read(redoData);
865            dbPage.write(index, redoData);
866        }
867    }

870    /**
871     * This helper method uses a {@link WALRecordType#UPDATE_PAGE} record to
872     * undo changes to a data page, and at the same time the method generates
873     * the data that must go into a corresponding redo-only WAL record.
874     *
875     * @param walReader A reader positioned at the start of the redo/undo data
876     *        to apply to the data page.  This method will advance the reader's
877     *        position past this redo/undo data.
878     *
879     * @param dbPage the data page that undo operations should be applied to
880     *
881     * @param numSegments the number of segments in the redo/undo data
882     *
883     * @return a byte-array containing the same number of segments as the
884     *         original update record, with only the data necessary for the
885     *         redo-only record.
886     *
887     * @throws IOException if an IO error occurs while applying the undo
888     *         operation
889     */
890    private byte[] applyUndoAndGenRedoOnlyData(DBFileReader walReader,
891        DBPage dbPage, int numSegments) throws IOException {
892
893        ByteArrayOutputStream redoOnlyBAOS = new ByteArrayOutputStream();
894        DataOutputStream dos = new DataOutputStream(redoOnlyBAOS);
895
896        for (int i = 0; i < numSegments; i++) {
897            // Read the starting index and length of this segment.
898            int index = walReader.readUnsignedShort();
899            int size = walReader.readUnsignedShort();
900
901            // Apply the undo data to the data page.
902            byte[] undoData = new byte[size];
903            walReader.read(undoData);
904            dbPage.write(index, undoData);
905
906            // Skip past the redo data, because don't care about it.
907            walReader.movePosition(size);
908
909            // Record what we wrote into the redo-only record data.
910            dos.writeShort(index);
911            dos.writeShort(size);
912            dos.write(undoData);
913        }
914
915        // Return the data that will appear in the redo-only record body.
916        dos.flush();
917        return redoOnlyBAOS.toByteArray();
918    }
