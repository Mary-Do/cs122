004/**
005 * This enumeration represents the different kinds of indexes that NanoDB
006 * supports, since different kinds of indexes facilitate different kinds of
007 * lookups with different performance characteristics.
008 */
009public enum IndexType {
010    /** Represents indexes that order values to determine a tuple's location. */
011    ORDERED_INDEX,
012
013    /** Represents indexes that hash values to determine a tuple's location. */
014    HASHED_INDEX
015}
