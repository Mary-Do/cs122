034    /**
035     * This method takes the schema of a table, and a description of an index,
036     * and it builds the schema that the index should have.  This includes all
037     * of the columns referenced by the index in the order that the index
038     * references them, and it also includes a <tt>#TUPLE_PTR</tt> column so
039     * the index can reference columns in the table.
040     *
041     * @param tableSchema the schema of the table that the index is for
042     * @param indexDesc a specification of the index
043     *
044     * @return the schema of the index
045     */
046    public static TableSchema makeIndexSchema(TableSchema tableSchema,
047                                              ColumnRefs indexDesc) {
048        if (tableSchema == null)
049            throw new IllegalArgumentException("tableSchema cannot be null");
050
051        if (indexDesc == null)
052            throw new IllegalArgumentException("indexDesc cannot be null");
053
054        // Get the name of the table that the index will be built against.
055        // (And, make sure there is only one table name in the schema...)
056        Set<String> tableNames = tableSchema.getTableNames();
057        if (tableSchema.getTableNames().size() != 1) {
058            throw new IllegalArgumentException(
059                "Schema must have exactly one table name");
060        }
061
062        String tableName = tableNames.iterator().next();
063
064        // Add all the referenced columns from the table schema.
065        TableSchema indexSchema = new TableSchema();
066        for (int iCol : indexDesc.getCols())
067            indexSchema.addColumnInfo(tableSchema.getColumnInfo(iCol));
068
069        // Add a tuple-pointer field for the index as well.
070        ColumnInfo filePtr = new ColumnInfo("#TUPLE_PTR", tableName,
071            new ColumnType(SQLDataType.FILE_POINTER));
072        indexSchema.addColumnInfo(filePtr);
073
074        return indexSchema;
075    }

136    /**
137     * This method constructs a {@link ForeignKeyColumnRefs} object that
138     * includes the columns named in the input list, as well as the referenced
139     * table and column names.  Note that this method <u>does not</u> update
140     * the schema stored on disk, or create any other supporting files.
141     *
142     * @param columnNames a list of column names that are in the key
143     *
144     * @param refTableName the table referenced by this key
145     *
146     * @param refTableSchema the schema of the table referenced by this key
147     *
148     * @param refColumnNames the columns in the referenced table that this
149     *        table's columns reference
150     *
151     * @param onDelete the {@link ForeignKeyValueChangeOption} for ON DELETE
152     *
153     * @param onUpdate the {@link ForeignKeyValueChangeOption} for ON UPDATE
154     *
155     * @return a new <tt>ForeignKeyColumns</tt> object with the indexes of the
156     *         columns stored in the object
157     *
158     * @throws SchemaNameException if a column-name cannot be found, or if a
159     *         column-name is ambiguous (unlikely), or if a column is specified
160     *         multiple times in the input list.
161     */
162    public static ForeignKeyColumnRefs makeForeignKey(TableSchema tableSchema,
163        List<String> columnNames, String refTableName,
164        TableSchema refTableSchema, List<String> refColumnNames,
165        ForeignKeyValueChangeOption onDelete,
166        ForeignKeyValueChangeOption onUpdate) {
167
168        if (tableSchema == null)
169            throw new IllegalArgumentException("tableSchema must be specified");
170
171        if (columnNames == null)
172            throw new IllegalArgumentException("columnNames must be specified");
173
174        if (refTableName == null)
175            throw new IllegalArgumentException("refTableName must be specified");
176
177        if (refTableSchema == null)
178            throw new IllegalArgumentException("refTableSchema must be specified");
179
180        if (refColumnNames == null)
181            throw new IllegalArgumentException("refColumnNames must be specified");
182
183        if (columnNames.isEmpty()) {
184            throw new IllegalArgumentException(
185                "columnNames must specify at least one column");
186        }
187
188        if (columnNames.size() != refColumnNames.size()) {
189            throw new IllegalArgumentException("columnNames and " +
190                "refColumnNames must specify the same number of columns");
191        }
192
193        int[] colIndexes = tableSchema.getColumnIndexes(columnNames);
194        int[] refColIndexes = refTableSchema.getColumnIndexes(refColumnNames);
195
196        if (!refTableSchema.hasKeyOnColumns(new ColumnRefs(refColIndexes))) {
197            throw new SchemaNameException(String.format("Referenced columns " +
198                    "%s in table %s are not a primary or candidate key",
199                refColumnNames, refTableName));
200        }
201
202        ArrayList<ColumnInfo> myColInfos = tableSchema.getColumnInfos(colIndexes);
203        ArrayList<ColumnInfo> refColInfos = refTableSchema.getColumnInfos(refColIndexes);
204
205        // Check if the column relations are the same types
206        for (int i = 0; i < myColInfos.size(); i++) {
207            ColumnType myType = myColInfos.get(i).getType();
208            ColumnType refType = refColInfos.get(i).getType();
209            if (!myType.equals(refType)) {
210                throw new IllegalArgumentException("columns in " +
211                    "child and parent tables of the foreign key must be " +
212                    "of the same type!");
213            }
214        }
215
216        // The onDelete and onUpdate values could be null if they are
217        // unspecified in the constructor.  They are set to
218        // ForeignKeyValueChangeOption.RESTRICT as a default in this case in
219        // the constructor for ForeignKeyColumnIndexes.
220        return new ForeignKeyColumnRefs(colIndexes, refTableName,
221            refColIndexes, onDelete, onUpdate);
222    }

225    /**
226     * <p>
227     * This helper function creates a {@link TupleLiteral} that holds the
228     * key-values necessary for probing, storing or deleting a tuple in a
229     * table's index.  <b>Note that this operation only works for a tuple that
230     * has the same schema as the index's owning table; it is not a general
231     * operation for creating search-keys on a particular index.</b>
232     * </p>
233     * <p>
234     * This method can be used to find a specific tuple in a table by
235     * including a file-pointer to the tuple in the table.  (For example, this
236     * is necessary when a tuple is being deleted from a table, so that the
237     * index can be updated to reflect the removal of that specific tuple.)
238     * This functionality requires that the tuple's class implement
239     * {@link Tuple#getExternalReference}.
240     * </p>
241     * <p>
242     * If this specific feature is not required, any kind of {@link Tuple}
243     * can be passed as an argument, as long as it has the same schema as
244     * the table that owns the index.
245     * </p>
246     *
247     * @param columnRefs the columns that the index is built on
248     *
249     * @param tuple the tuple from the original table, that the key will be
250     *        created from.
251     *
252     * @param findExactTuple if {@code true}, this method will include the
253     *        {@code tuple}'s file-pointer so that the exact tuple can be
254     *        found in the index.
255     *
256     * @return a tuple-literal that can be used for storing, looking up, or
257     *         deleting the specific tuple {@code ptup}.
258     */
259    public static TupleLiteral makeTableSearchKey(ColumnRefs columnRefs,
260        Tuple tuple, boolean findExactTuple) {
261
262        // Build up a new tuple-literal containing the search key.
263        TupleLiteral searchKeyVal = new TupleLiteral();
264        for (int i = 0; i < columnRefs.size(); i++)
265            searchKeyVal.addValue(tuple.getColumnValue(columnRefs.getCol(i)));
266
267        if (findExactTuple) {
268            // Include the file-pointer as the last value in the tuple, so
269            // that all key-values are unique in the index.
270            searchKeyVal.addValue(tuple.getExternalReference());
271        }
272
273        return searchKeyVal;
274    }

277    /**
278     * Given an index tuple-file and a search key, this method attempts to
279     * find the first tuple in the index that matches the search key.
280     *
281     * @param key the search-key value to probe the index with
282     *
283     * @param idxTupleFile the index to probe with the search-key
284     *
285     * @return the first matching tuple in the index, or {@code null} if
286     *         no matching tuple could be found
287     *
288     * @throws IOException if an IO error occurs during the operation
289     */
290    public static PageTuple findTupleInIndex(Tuple key, TupleFile idxTupleFile)
291        throws IOException {
292
293        PageTuple idxPageTup;
294
295        if (idxTupleFile instanceof SequentialTupleFile) {
296            SequentialTupleFile seqTupleFile = (SequentialTupleFile) idxTupleFile;
297            idxPageTup = (PageTuple) seqTupleFile.findFirstTupleEquals(key);
298        }
299        else if (idxTupleFile instanceof HashedTupleFile) {
300            HashedTupleFile hashTupleFile = (HashedTupleFile) idxTupleFile;
301            idxPageTup = (PageTuple) hashTupleFile.findFirstTupleEquals(key);
302        }
303        else {
304            throw new IllegalStateException("Index files must " +
305                "be sequential or hashing tuple files.");
306        }
307
308        return idxPageTup;
309    }

323    /**
324     * This method performs a basic but important verification, that every
325     * tuple in a table is referenced once by the table's index, and no index
326     * entry has a tuple-reference to a non-existent tuple.  This
327     * functionality is of course greatly supplemented by tuple-file formats
328     * that implement the {@link edu.caltech.nanodb.storage.TupleFile#verify}
329     * method, which does complete verification of the internal structure of
330     * a particular kind of tuple file.
331     *
332     * @param tableTupleFile the tuple file holding the table data
333     * @param indexTupleFile the tuple file holding the index data
334     *
335     * @return A list of string error messages identified during the
336     *         verification scan.  This list will be empty if there are no
337     *         errors.
338     *
339     * @throws IOException
340     */
341    public static List<String> verifyIndex(TupleFile tableTupleFile,
342        TupleFile indexTupleFile) throws IOException {
343
344        ArrayList<String> errors = new ArrayList<>();
345        HashSet<FilePointer> tableTuples = new HashSet<>();
346        HashSet<FilePointer> indexTuples = new HashSet<>();
347        HashSet<FilePointer> referencedTableTuples = new HashSet<>();
348        Tuple tup;
349
350        // Scan through all tuples in the table file, and record the file
351        // pointer to each one.
352        tup = tableTupleFile.getFirstTuple();
353        while (tup != null) {
354            if (!tableTuples.add(tup.getExternalReference())) {
355                // This should never happen.
356                throw new IllegalStateException("The impossible has " +
357                    "happened:  two tuples had the same external reference!");
358            }
359            tup = tableTupleFile.getNextTuple(tup);
360        }
361
362        // Scan through all entries in the index, and record the file pointer
363        // stored in each index record.
364        Schema indexSchema = indexTupleFile.getSchema();
365        int iCol = indexSchema.getColumnIndex("#TUPLE_PTR");
366
367        tup = indexTupleFile.getFirstTuple();
368        while (tup != null) {
369            FilePointer fptr = (FilePointer) tup.getColumnValue(iCol);
370
371            if (indexTuples.contains(fptr)) {
372                errors.add("Tuple at location " + fptr +
373                    " appears multiple times in the index.");
374            }
375
376            if (!tableTuples.contains(fptr)) {
377                errors.add(
378                    "Index references a nonexistent tuple at location " +
379                    fptr + ".");
380            }
381
382            indexTuples.add(fptr);
383            referencedTableTuples.add(fptr);
384
385            tup = indexTupleFile.getNextTuple(tup);
386        }
387
388        HashSet<FilePointer> diff = new HashSet<>(tableTuples);
389        diff.removeAll(referencedTableTuples);
390        if (!diff.isEmpty()) {
391            for (FilePointer fptr : diff) {
392                errors.add("Tuple at location " + fptr +
393                           " wasn't referenced by the index.");
394            }
395        }
396
397        return errors;
398    }
