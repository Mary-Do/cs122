083    @Override
084    public IndexInfo addIndexToTable(TableInfo tableInfo,
085        ColumnRefs indexColRefs) throws IOException {
086
087        if (tableInfo == null)
088            throw new IllegalArgumentException("tableInfo cannot be null");
089
090        if (indexColRefs == null)
091            throw new IllegalArgumentException("indexColRefs cannot be null");
092
093        // Figure out the schema and other essential details of the index.
094
095        String tableName = tableInfo.getTableName();
096        TableSchema tableSchema = tableInfo.getSchema();
097
098        // TODO:  Check if current columns already are an index for this table.
099        //        (Not essential, but you don't want to be redundant.)
100        //        (NOTE:  Also need to make sure the index types are the same.)
101        /*
102        getIndexesOnColumns(columnNames, / * exactMatch * / true);
103
104        for (Map.Entry<String, ColumnIndexes> entry :
105            tableSchema.getIndexes().entrySet()) {
106            if (index.equalsColumns(entry.getValue())) {
107                throw new ExecutionException(String.format(
108                    "Could not create new index on table \"%s\". The table " +
109                        "already has an index \"%s\" with the same columns " +
110                        "in the same order.", tableName, entry.getKey()));
111            }
112        }
113        */
114
115        String indexName = indexColRefs.getIndexName();
116        logger.debug(String.format("Creating an IndexInfo object " +
117            "describing the new index %s on table %s.",
118            indexName != null ? indexName : "[unnamed]", tableName));
119
120        IndexInfo indexInfo = new IndexInfo(tableInfo, indexColRefs);
121        if (indexName == null) {
122            // This is an unnamed index.
123            logger.debug("Creating the new unnamed index on disk.");
124            createUnnamedIndex(indexInfo);
125        }
126        else {
127            // This is a named index.
128            logger.debug("Creating the new index " + indexName + " on disk.");
129            createIndex(indexInfo, indexName);
130        }
131
132        logger.debug("Index created.  Index name is " +
133            indexInfo.getIndexName() + ", and filename is " +
134            indexInfo.getTupleFile().getDBFile());
135
136        TableConstraintType constraintType = indexColRefs.getConstraintType();
137        if (constraintType != null && constraintType.isUnique())
138            tableSchema.addCandidateKey((KeyColumnRefs) indexColRefs);
139        else
140            tableSchema.addIndex(indexColRefs);
141
142        // Write schema with new index to file
143        storageManager.getTableManager().saveTableInfo(tableInfo);
144
145        // Populate the new index from the table's tuples.
146        populateIndex(tableInfo, indexInfo);
147
148        return indexInfo;
149    }

152    private void populateIndex(TableInfo srcTableInfo, IndexInfo newIndexInfo)
153        throws IOException {
154
155        String tableName = srcTableInfo.getTableName();
156        String indexName = newIndexInfo.getIndexName();
157
158        ColumnRefs columnRefs = newIndexInfo.getTableColumnRefs();
159        TableConstraintType constraintType = columnRefs.getConstraintType();
160        boolean unique = (constraintType != null && constraintType.isUnique());
161
162        logger.debug(String.format("Populating new index %s with existing " +
163            "tuples in table %s.", indexName, tableName));
164
165        TupleFile tableTupleFile = srcTableInfo.getTupleFile();
166        TupleFile indexTupleFile = newIndexInfo.getTupleFile();
167
168        // Traverse the tuples in the table, so we can populate the index.
169        PageTuple curTuple = (PageTuple) tableTupleFile.getFirstTuple();
170        while (curTuple != null) {
171            TupleLiteral idxTup;
172            if (unique) {
173                // Check if the index already has a tuple with this value.
174                // The tuple we generate for this shouldn't include a tuple-
175                // pointer since we just want to see if the value is repeated.
176                idxTup = IndexUtils.makeTableSearchKey(columnRefs, curTuple,
177                    /* findExactTuple */ false);
178
179                if (IndexUtils.findTupleInIndex(idxTup, indexTupleFile) != null) {
180                    // Adding this row would violate the unique index.
181                    throw new IllegalStateException("Unique index " +
182                        "already contains a tuple with this value.");
183                }
184            }
185
186            // Generate the "index-schema" version of the tuple from the
187            // "table-schema" version of the tuple.
188            idxTup = IndexUtils.makeTableSearchKey(columnRefs, curTuple,
189                /* findExactTuple */ true);
190            indexTupleFile.addTuple(idxTup);
191
192            // Move on to the next tuple in the table file.
193            curTuple = (PageTuple) tableTupleFile.getNextTuple(curTuple);
194        }
195    }

198    /**
199     * Creates a new index file with the index name, table name, and column list
200     * specified in the passed-in <tt>IndexInfo</tt> object.  Additional
201     * details such as the data file and the index manager are stored into the
202     * passed-in <tt>IndexInfo</tt> object upon successful creation of the
203     * new index.
204     *
205     * @param indexInfo This object is an in/out parameter.  It is used to
206     *        specify the name and details of the new index being created.  When
207     *        the index is successfully created, the object is updated with the
208     *        actual file that the index is stored in.
209     * @param indexName The name of the index created by the CREATE INDEX cmd.
210     *        If there is no idxName, use the createUnnamedIndex method
211     *        instead.
212     *
213     * @throws IOException if the file cannot be created, or if an error occurs
214     *         while storing the initial index data.
215     */
216    @Override
217    public void createIndex(IndexInfo indexInfo, String indexName)
218        throws IOException {
219
220        String idxFileName = getIndexFileName(indexInfo.getTableName(), indexName);
221
222        // TODO:  the file type and page size should be specified in the
223        //        IndexInfo object
224        int pageSize = StorageManager.getCurrentPageSize();
225        DBFileType type = DBFileType.BTREE_TUPLE_FILE;
226        TupleFileManager tupleFileManager = storageManager.getTupleFileManager(type);
227
228        // First, create a new DBFile that the tuple file will go into.
229        FileManager fileManager = storageManager.getFileManager();
230        DBFile dbFile = fileManager.createDBFile(idxFileName, type, pageSize);
231        logger.debug("Created new DBFile for index " + indexName +
232                     " at path " + dbFile.getDataFile());
233
234        // Generate a schema based on the index information.
235        TableSchema indexSchema = IndexUtils.makeIndexSchema(
236            indexInfo.getTableInfo().getSchema(),
237            indexInfo.getTableColumnRefs());
238
239        // Now, initialize it to be a tuple file with the specified type and
240        // schema.
241        TupleFile tupleFile = tupleFileManager.createTupleFile(dbFile, indexSchema);
242        indexInfo.setTupleFile(tupleFile);
243
244        // Cache this index since it's now considered "open".
245        openIndexes.put(indexInfo.getTableName(), indexInfo);
246    }

249    @Override
250    public void createUnnamedIndex(IndexInfo indexInfo) throws IOException {
251
252        File baseDir = storageManager.getBaseDir();
253        String tableName = indexInfo.getTableName();
254        TableConstraintType constraintType =
255            indexInfo.getTableColumnRefs().getConstraintType();
256
257        // Figure out an index name and filename for the unnamed index.
258        // Primary keys are handled separately, since each table only has
259        // one primary key.  All other indexes are named by concatenating a
260        // prefix with a numeric value that makes the index's name unique.
261        File f;
262        String indexName, indexFilename;
263        String prefix = getUnnamedIndexPrefix(indexInfo);
264        if (constraintType == TableConstraintType.PRIMARY_KEY) {
265            indexName = prefix;
266            indexFilename = getIndexFileName(tableName, indexName);
267            f = new File(baseDir, indexFilename);
268            if (!f.createNewFile()) {
269                throw new IOException("Couldn't create file " + f +
270                                      " for primary-key index " + indexName);
271            }
272        }
273        else {
274            String pattern = prefix + "_%03d";
275            int i = 0;
276            do {
277                i++;
278                indexName = String.format(pattern, i);
279                indexFilename = getIndexFileName(tableName, indexName);
280                f = new File(baseDir, indexFilename);
281            }
282            while (!f.createNewFile());
283        }
284
285        indexInfo.setIndexName(indexName);
286
287        // Delete the file so we can create it in the createIndex() call.
288        // TODO:  This is REALLY gross.
289        f.delete();
290        createIndex(indexInfo, indexName);
291    }

294    /**
295     * This method opens the data file corresponding to the specified index
296     * name and reads in the index's details.  If the index is already open
297     * then the cached data is simply returned.
298     *
299     * @param tableInfo the table that the index is defined on
300     *
301     * @param indexName the name of the index to open.  Indexes are not
302     *        referenced directly except by CREATE/ALTER/DROP INDEX statements,
303     *        so these index names are stored in the table schema files, and are
304     *        generally opened when the optimizer needs to know what indexes are
305     *        available.
306     *
307     * @return an object representing the details of the open index
308     *
309     * @throws java.io.FileNotFoundException if no index-file exists for the
310     *         index; in other words, it doesn't yet exist.
311     *
312     * @throws IOException if an IO error occurs when attempting to open the
313     *         index.
314     */
315    @Override
316    public IndexInfo openIndex(TableInfo tableInfo, String indexName)
317        throws IOException {
318
319        IndexInfo indexInfo;
320
321        // If the index is already open, just return the cached information.
322        String indexKey = tableInfo.getTableName() + "." + indexName;
323        indexInfo = openIndexes.get(indexKey);
324        if (indexInfo != null)
325            return indexInfo;
326
327        // Open the data file for the index; read out its type and page-size.
328
329        String idxFileName = getIndexFileName(tableInfo.getTableName(), indexName);
330        TupleFile tupleFile = storageManager.openTupleFile(idxFileName);
331
332        TableSchema tableSchema = tableInfo.getSchema();
333        ColumnRefs columnRefs = tableSchema.getIndex(indexName);
334
335        indexInfo = new IndexInfo(tableInfo, columnRefs, tupleFile);
336
337        // Cache this index since it's now considered "open".
338        openIndexes.put(indexKey, indexInfo);
339
340        // Defer to the appropriate index-manager to read in the remainder of
341        // the details.
342        loadIndexInfo(indexInfo);
343
344        return indexInfo;
345    }

348    /**
349     * This function generates the prefix of a name for an index with no
350     * actual name specified.  Since indexes and other constraints don't
351     * necessarily require names to be specified, we need some way to
352     * generate these names.
353     *
354     * @param indexInfo the information describing the index to be named
355     *
356     * @return a string containing a prefix to use for naming the index.
357     */
358    public String getUnnamedIndexPrefix(IndexInfo indexInfo) {
359        // Generate a prefix based on the contents of the IndexInfo object.
360        ColumnRefs colRefs = indexInfo.getTableColumnRefs();
361        TableConstraintType constraintType = colRefs.getConstraintType();
362
363        if (constraintType == null)
364            return "IDX_" + indexInfo.getTableName();
365
366        switch (constraintType) {
367        case PRIMARY_KEY:
368            return "PK_" + indexInfo.getTableName();
369
370        case UNIQUE:
371            return "CK_" + indexInfo.getTableName();
372
373        case FOREIGN_KEY:
374            return "FK_" + indexInfo.getTableName();
375
376        default:
377            throw new IllegalArgumentException("Unrecognized constraint type " +
378                constraintType);
379        }
380    }
