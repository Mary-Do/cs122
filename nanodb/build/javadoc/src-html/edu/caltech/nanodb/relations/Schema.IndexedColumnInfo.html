020/**
021 * <p>
022 * A schema is an ordered collection of column names and associated types.
023 * </p>
024 * <p>
025 * Many different entities in the database code can have schema associated
026 * with them.  Both tables and tuples have schemas, for obvious reasons.
027 * <tt>SELECT</tt> and <tt>FROM</tt> clauses also have schemas, used by the
028 * database engine to verify the semantics of database queries.  Finally,
029 * relational algebra plan nodes also have schemas, which specify the kinds of
030 * tuples that they generate.
031 * </p>
032 */
033public class Schema implements Serializable, Iterable<ColumnInfo> {
034
035    /**
036     * This helper class is used for the internal hashed column structure, so
037     * that we can do fast lookups based on table names and column names.
038     */
039    private static class IndexedColumnInfo implements Serializable {
040        /** The index in the schema that the column appears at. */
041        public int colIndex;
042
043        /** The details of the column at the stored index. */
044        public ColumnInfo colInfo;
045
046        /** Stores the specified index and column-info value. */
047        public IndexedColumnInfo(int colIndex, ColumnInfo colInfo) {
048            if (colInfo == null)
049                throw new NullPointerException("colInfo cannot be null");
050
051            if (colIndex < 0) {
052                throw new IllegalArgumentException("colIndex must be >= 0; got " +
053                    colIndex);
054            }
055
056            this.colIndex = colIndex;
057            this.colInfo = colInfo;
058        }
059    }
060
061
062    /**
063     * The collection of the column-info objects describing the columns in the
064     * schema.
065     */
066    private ArrayList<ColumnInfo> columnInfos;
067
068
069    /**
070     * A mapping that provides fast lookups for columns based on table name and
071     * column name.  The outer hash-map has table names for keys; "no table" is
072     * indicated with a <code>null</code> key, which {@link java.util.HashMap}
073     * supports.  The inner hash-map has column names for keys, and maps to
074     * column information objects.
075     */
076    private HashMap<String, HashMap<String, IndexedColumnInfo>> colsHashedByTable;
077
078
079    /**
080     * A mapping that provides fast lookups for columns based only on column
081     * name.  Because multiple columns could have the same column name (but
082     * different table names) in a single schema, the values in the mapping are
083     * lists.
084     */
085    private HashMap<String, ArrayList<IndexedColumnInfo>> colsHashedByColumn;
086

089    public Schema() {
090        columnInfos = new ArrayList<ColumnInfo>();
091        colsHashedByTable =
092            new HashMap<String, HashMap<String, IndexedColumnInfo> >();
093        colsHashedByColumn = new HashMap<String, ArrayList<IndexedColumnInfo>>();
094    }

143    public ArrayList<ColumnInfo> getColumnInfos(int[] colIndexes) {
144        ArrayList<ColumnInfo> result = new ArrayList<ColumnInfo>(colIndexes.length);
145
146        for (int i = 0; i < colIndexes.length; i++)
147            result.add(getColumnInfo(colIndexes[i]));
148
149        return result;
150    }

158    public int addColumnInfo(ColumnInfo colInfo) {
159        if (colInfo == null)
160            throw new NullPointerException("colInfo cannot be null");
161
162        String colName = colInfo.getName();
163        String tblName = colInfo.getTableName();
164
165        // Check the hashed-columns structure to see if this column already
166        // appears in the schema.
167
168        HashMap<String, IndexedColumnInfo> colMap = colsHashedByTable.get(tblName);
169        if (colMap != null && colMap.containsKey(colName)) {
170            throw new SchemaNameException("Specified column " + colInfo +
171            " is a duplicate of an existing column.");
172        }
173
174        int colIndex = columnInfos.size();
175        columnInfos.add(colInfo);
176
177        IndexedColumnInfo indexedColInfo = new IndexedColumnInfo(colIndex, colInfo);
178
179        // Update the hashed-columns structures for fast/easy lookup.
180
181        if (colMap == null) {
182            colMap = new HashMap<String, IndexedColumnInfo>();
183            colsHashedByTable.put(tblName, colMap);
184        }
185        colMap.put(colName, indexedColInfo);
186
187        ArrayList<IndexedColumnInfo> colList = colsHashedByColumn.get(colName);
188        if (colList == null) {
189            colList = new ArrayList<IndexedColumnInfo>();
190            colsHashedByColumn.put(colName, colList);
191        }
192        colList.add(indexedColInfo);
193
194        // Finally, return the index.
195
196        return colIndex;
197    }

200    /**
201     * Append another schema to this schema.
202     *
203     * @throws SchemaNameException if any of the input column-info objects
204     *         overlap the names of columns already in the schema.
205     */
206    public void append(Schema s) throws SchemaNameException {
207        for (ColumnInfo colInfo : s)
208             addColumnInfo(colInfo);
209    }

217     *         already in the schema.

220
221        for (ColumnInfo colInfo : colInfos)

238     * this schema and the specified schema.  Note that not all columns of a

240     * there just has to be at least one column from the table in both schemas

282     * Returns the number of columns that have the specified column name.  Note
283     * that multiple columns can have the same column name but different table
284     * names.
285     *
286     * @param colName the column name to return the count for
287     * @return the number of columns with the specified column name

289    public int numColumnsWithName(String colName) {
290        ArrayList<IndexedColumnInfo> list = colsHashedByColumn.get(colName);
291        if (list != null)
292            return list.size();
293
294        return 0;

298    /**
299     * This helper method returns true if this schema contains any columns with
300     * the same column name but different table names.  If so, the schema is not
301     * valid for use on one side of a <tt>NATURAL</tt> join.
302     *
303     * @return true if the schema has multiple columns with the same column name
304     *         but different table names, or false otherwise.
305     */
306    public boolean hasMultipleColumnsWithSameName() {
307        for (String cName : colsHashedByColumn.keySet()) {
308            if (colsHashedByColumn.get(cName).size() > 1)
309                return true;
310        }
311        return false;
312    }

315    /**
316     * Presuming that exactly one column has the specified name, this method
317     * returns the column information for that column name.
318     *
319     * @param colName the name of the column to retrieve the information for
320     *
321     * @return the column information for the specified column
322     *
323     * @throws SchemaNameException if the specified column name doesn't appear
324     *         in this schema, or if it appears multiple times
325     */
326    public ColumnInfo getColumnInfo(String colName) {
327        ArrayList<IndexedColumnInfo> list = colsHashedByColumn.get(colName);
328        if (list == null || list.size() == 0)
329            throw new SchemaNameException("No columns with name " + colName);
330
331        if (list.size() > 1)
332            throw new SchemaNameException("Multiple columns with name " + colName);
333
334        return list.get(0).colInfo;
335    }

338    /**
339     * This method iterates through all columns in this schema and sets them all
340     * to be on the specified table.  This method will throw an exception if the
341     * result would be an invalid schema with duplicate column names.
342     *
343     * @throws SchemaNameException if the schema contains columns with the same
344     *         column name but different table names.  In this case, resetting the
345     *         table name will produce an invalid schema with ambiguous column
346     *         names.
347     *
348     * @design (donnie) At present, this method does this by replacing each
349     *         {@link edu.caltech.nanodb.relations.ColumnInfo} object with a new
350     *         object with updated information.  This is because
351     *         <code>ColumnInfo</code> is currently immutable.
352     */
353    public void setTableName(String tableName) throws SchemaNameException {
354        // First, verify that overriding the table names will not produce multiple
355        // ambiguous column names.
356        ArrayList<String> duplicateNames = null;
357
358        for (Map.Entry<String, ArrayList<IndexedColumnInfo> > entry :
359             colsHashedByColumn.entrySet()) {
360
361            if (entry.getValue().size() > 1) {
362                if (duplicateNames == null)
363                    duplicateNames = new ArrayList<String>();
364
365                duplicateNames.add(entry.getKey());
366            }
367        }
368
369        if (duplicateNames != null) {
370            throw new SchemaNameException("Overriding table-name to \"" +
371                tableName + "\" would produce ambiguous columns:  " +
372                duplicateNames);
373        }
374
375        // If we get here, we know that we can safely override the table name for
376        // all columns.
377
378        ArrayList<ColumnInfo> oldColInfos = columnInfos;
379
380        columnInfos = new ArrayList<ColumnInfo>();
381        colsHashedByTable =
382            new HashMap<String, HashMap<String, IndexedColumnInfo>>();
383        colsHashedByColumn = new HashMap<String, ArrayList<IndexedColumnInfo>>();
384
385        // Iterate over the columns in the same order as they were in originally.
386        // For each one, override the table name, then use addColumnInfo() to
387        // properly update the internal hash structure.
388
389        for (ColumnInfo colInfo : oldColInfos) {
390            ColumnInfo newColInfo =
391                new ColumnInfo(colInfo.getName(), tableName, colInfo.getType());
392
393            addColumnInfo(newColInfo);
394        }
395    }

398    public int getColumnIndex(ColumnName colName) {
399        if (colName.isColumnWildcard())
400            throw new IllegalArgumentException("colName cannot be a wildcard");
401
402        return getColumnIndex(colName.getTableName(), colName.getColumnName());
403    }

412        return getColumnIndex(null, colName);
413    }

415
416    public int getColumnIndex(String tblName, String colName) {
417        ArrayList<IndexedColumnInfo> colList = colsHashedByColumn.get(colName);
418
419        if (colList == null)
420            return -1;
421
422        if (tblName == null) {
423            if (colList.size() > 1) {
424                throw new SchemaNameException("Column name \"" + colName +
425                    "\" is ambiguous in this schema.");
426            }
427
428            return colList.get(0).colIndex;
429        }
430        else {
431            // Table-name is specified.
432
433            for (IndexedColumnInfo c : colList) {
434                if (tblName.equals(c.colInfo.getTableName()))
435                    return c.colIndex;
436            }
437        }
438
439        return -1;
440    }

443    /**
444     * Given a list of column names, this method returns an array containing
445     * the indexes of the specified columns.
446     *
447     * @param columnNames a list of column names in the schema
448     *
449     * @return an array containing the indexes of the columns specified in the
450     *         input
451     *
452     * @throws SchemaNameException if a column name is specified multiple
453     *         times in the input list, or if a column name doesn't appear in
454     *         the schema
455     */
456    public int[] getColumnIndexes(List<String> columnNames) {
457        int[] result = new int[columnNames.size()];
458        HashSet<String> s = new HashSet<String>();
459
460        int i = 0;
461        for (String colName : columnNames) {
462            if (!s.add(colName)) {
463                throw new SchemaNameException(String.format(
464                    "Column %s was specified multiple times", colName));
465            }
466
467            result[i] = getColumnIndex(colName);
468            if (result[i] == -1) {
469                throw new SchemaNameException(String.format(
470                    "Schema doesn't contain a column named %s", colName));
471            }
472
473            i++;
474        }
475
476        return result;
477    }

480    /**
481     * Given a (possibly wildcard) column-name, this method returns the collection
482     * of all columns that match the specified column name.  The collection is a
483     * mapping from integer indexes (the keys) to <code>ColumnInfo</code> objects
484     * from the schema.
485     * <p>
486     * Any valid column-name object will work, so all of these options are
487     * available:
488     * <ul>
489     *   <li><b>No table, only a column name</b> - to resolve an unqualified
490     *     column name, e.g. in an expression or predicate</li>
491     *   <li><b>A table and column name</b> - to check whether the schema contains
492     *     such a column</li>
493     *   <li><b>A wildcard without a table name</b> - to retrieve all columns in
494     *     the schema</li>
495     *   <li><b>A wildcard with a table name</b> - to retrieve all columns
496     *     associated with a particular table name</li>
497     * </ul>
498     */
499    public SortedMap<Integer, ColumnInfo> findColumns(ColumnName colName) {
500
501        TreeMap<Integer, ColumnInfo> found = new TreeMap<Integer, ColumnInfo>();
502
503        if (colName.isColumnWildcard()) {
504            // Some kind of wildcard column-name object.
505
506            if (!colName.isTableSpecified()) {
507                // Wildcard with no table name:  *
508                // Add all columns in the schema to the result.
509
510                for (int i = 0; i < columnInfos.size(); i++)
511                    found.put(i, columnInfos.get(i));
512            }
513            else {
514                // Wildcard with a table name:  tbl.*
515                // Find the table info and add its columns to the result.
516
517                HashMap<String, IndexedColumnInfo> tableCols =
518                    colsHashedByTable.get(colName.getTableName());
519
520                if (tableCols != null) {
521                    for (IndexedColumnInfo indexedColInfo: tableCols.values())
522                        found.put(indexedColInfo.colIndex, indexedColInfo.colInfo);
523                }
524            }
525        }
526        else {
527            // A non-wildcard column-name object.
528
529            if (!colName.isTableSpecified()) {
530                // Column name with no table name:  col
531                // Look up the list of column-info objects grouped by column name.
532
533                ArrayList<IndexedColumnInfo> colList =
534                    colsHashedByColumn.get(colName.getColumnName());
535
536                if (colList != null) {
537                    for (IndexedColumnInfo indexedColInfo : colList)
538                        found.put(indexedColInfo.colIndex, indexedColInfo.colInfo);
539                }
540            }
541            else {
542                // Column name with a table name:  tbl.col
543                // Find the table info and see if it has the specified column.
544
545                HashMap<String, IndexedColumnInfo> tableCols =
546                    colsHashedByTable.get(colName.getTableName());
547
548                if (tableCols != null) {
549                    IndexedColumnInfo indexedColInfo =
550                        tableCols.get(colName.getColumnName());
551                    if (indexedColInfo != null)
552                        found.put(indexedColInfo.colIndex, indexedColInfo.colInfo);
553                }
554            }
555        }
556
557        return found;
558    }
