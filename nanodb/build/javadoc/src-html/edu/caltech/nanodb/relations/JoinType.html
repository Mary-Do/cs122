001package edu.caltech.nanodb.relations;
002
003
004/** An enumeration specifying the different types of join operation. */
005public enum JoinType {
006
007    /** Inner joins, where only matching rows are included in the result. */
008    INNER,
009
010
011    /**
012     * Left outer joins, where non-matching rows from the left table are
013     * included in the results.
014     */
015    LEFT_OUTER,
016
017
018    /**
019     * Right outer joins, where non-matching rows from the right table are
020     * included in the results.
021     */
022    RIGHT_OUTER,
023
024
025    /**
026     * Full outer joins, where non-matching rows from either the left or right
027     * table are included in the results.
028     */
029    FULL_OUTER,
030
031
032    /** Cross joins, which are simply a Cartesian product. */
033    CROSS,
034
035
036    /**
037     * Semijoin, where the left table's rows are included when they match
038     * one or more rows from the right table.
039     */
040    SEMIJOIN,
041
042
043    /**
044     * Antijoin (aka anti-semijoin), where the left table's rows are included
045     * when they match none of the rows from the right table.
046     */
047    ANTIJOIN
048}
