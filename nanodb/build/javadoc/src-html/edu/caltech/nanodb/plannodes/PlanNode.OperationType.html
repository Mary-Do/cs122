022/**
023 * Represents a query plan node in its most abstract form.  To create actual
024 * plan nodes, use the subclasses of this node.
025 */
026public abstract class PlanNode implements Cloneable {
027
028    /** Node type enumeration. */
029    public enum OperationType {
030        /** Relational algebra Select operator. */
031        SELECT,
032
033        /** Relational algebra Project operator. */
034        PROJECT,
035
036        /** Relational algebra Rename operator. */
037        RENAME,
038
039        /** Relational algebra Theta Join operator. **/
040        THETA_JOIN,
041
042        /** Relational algebra Group By / Aggregate operator. */
043        GROUP_AGGREGATE,
044
045        /** Sorting operator. */
046        SORT,
047
048        /** A materialize plan-node. */
049        MATERIALIZE
050    }
051
052
053    /** The type of this plan node. */
054    protected OperationType nodeType;
055
056
057    /**
058     * The left child of this plan node.  If the plan node only has one child,
059     * use this field.
060     */
061    protected PlanNode leftChild;
062
063
064    /**
065     * If the plan node has two children, this field is set to the right child
066     * node.  If the plan has only one child, this field will be <tt>null</tt>.
067     */
068    protected PlanNode rightChild;
069
070
071    /**
072     * The schema of the results produced by this plan-node.  The schema is
073     * initialized by the {@link #prepare} method; before that method has been
074     * called, this will be <tt>null</tt>.
075     */
076    protected Schema schema;
077
078
079    /**
080     * The estimated cost of executing this plan and its subplans.  This cost is
081     * computed by the {@link #prepare} method; before that method has been
082     * called, this will be <tt>null</tt>.
083     */
084    protected PlanCost cost;
085
086
087    /**
088     * Statistics (possibly estimated) describing the results produced by this
089     * plan node.  These statistics are produced when the {@link #prepare}
090     * method is called; before that method has been called, this will be
091     * <tt>null</tt>.
092     * <p>
093     * This collection corresponds to the {@link #schema} object; it will
094     * contain the same number of columns, and the individual columns will
095     * correspond with the specific columns in the schema object.
096     */
097    protected ArrayList<ColumnStats> stats;
098
099
100    /**
101     * The environment used to evaluate expressions against tuples being
102     * processed.
103     */
104    protected Environment environment = new Environment();
105
106
107    /**
108     * Constructs a PlanNode with a given operation type.  This method will be
109     * called by subclass constructors.
110     *
111     * @param op the operation type of the node.
112     *
113     * @throws IllegalArgumentException if <tt>op</tt> is <tt>null</tt>
114     */
115    protected PlanNode(OperationType op) {
116        if (op == null)
117            throw new IllegalArgumentException("op cannot be null");
118
119        nodeType = op;
120    }
121
122
123    /**
124     * Constructs a PlanNode with a given operation type, and the specified left
125     * child-plan.  This method will be called by subclass constructors.
126     *
127     * @param op the operation type of the node.
128     * @param leftChild the left subplan of this node.
129     *
130     * @throws IllegalArgumentException if <tt>op</tt> or <tt>leftChild</tt> is
131     *         <tt>null</tt>
132     */
133    protected PlanNode(OperationType op, PlanNode leftChild) {
134        this(op);
135
136        if (leftChild == null)
137            throw new IllegalArgumentException("leftChild cannot be null");
138
139        this.leftChild = leftChild;
140    }
141
142
143    /**
144     * Constructs a PlanNode with a given operation type, and the specified left
145     * child-plan.  This method will be called by subclass constructors.
146     *
147     * @param op the operation type of the node.
148     * @param leftChild the left subplan of this node.
149     * @param rightChild the right subplan of this node.
150     *
151     * @throws IllegalArgumentException if <tt>op</tt>, <tt>leftChild</tt>, or
152     *         <tt>rightChild</tt> is <tt>null</tt>
153     */
154    protected PlanNode(OperationType op, PlanNode leftChild, PlanNode rightChild) {
155        this(op, leftChild);
156
157        if (rightChild == null)
158            throw new IllegalArgumentException("rightChild cannot be null");
159
160        this.rightChild = rightChild;
161    }
162
163
164    /**
165     * If the results are ordered in some way, this method returns a collection
166     * of expressions specifying what columns or expressions the results are
167     * ordered by.  If the results are not ordered then this method may return
168     * either an empty list or a <tt>null</tt> value.
169     * <p>
170     * When this method returns a list of ordering expressions, the order of the
171     * expressions themselves also matters.  The entire set of results will be
172     * ordered by the first expression; rows with the same value for the first
173     * expression will be ordered by the second expression; etc.
174     *
175     * @return If the plan node produces ordered results, this will be a list
176     *         of objects specifying the ordering.  If the node doesn't produce
177     *         ordered results then the return-value will either be an empty
178     *         list or it will be <tt>null</tt>.
179     */
180    public abstract List<OrderByExpression> resultsOrderedBy();
181
182
183    /**
184     * This method reports whether this plan node supports marking a certain
185     * point in the tuple-stream so that processing can return to that point
186     * as needed.
187     *
188     * @return true if the node supports position marking, false otherwise.
189     */
190    public abstract boolean supportsMarking();
191
192
193    /**
194     * This method reports whether this plan node requires the left child to
195     * support marking for proper evaluation.
196     *
197     * @return true if the node requires that its left child supports marking,
198     *         false otherwise.
199     */
200    public abstract boolean requiresLeftMarking();
201
202
203    /**
204     * This method reports whether this plan node requires the right child to
205     * support marking for proper evaluation.
206     *
207     * @return true if the node requires that its right child supports marking,
208     *         false otherwise.
209     */
210    public abstract boolean requiresRightMarking();
211
212
213    /**
214     * This method is responsible for computing critical details about the plan
215     * node, such as the schema of the results that are produced, the estimated
216     * cost of evaluating the plan node (and its children), and statistics
217     * describing the results produced by the plan node.
218     */
219    public abstract void prepare();
220
221
222    /**
223     * Returns the schema of the results that this node produces.  Some nodes
224     * such as Select will not change the input schema but others, such as
225     * Project, Rename, and ThetaJoin, must change it.
226     * <p>
227     * The schema is not computed until the {@link #prepare} method is called;
228     * until that point, this method will return <tt>null</tt>.
229     *
230     * @return the schema produced by this plan-node.
231     */
232    public final Schema getSchema() {
233        return schema;
234    }
235
236
237    /**
238     * Returns the estimated cost of this plan node's operation.  The estimate
239     * depends on which algorithm the node uses and the data it is working with.
240     * <p>
241     * The cost is not computed until the {@link #prepare} method is called;
242     * until that point, this method will return <tt>null</tt>.
243     *
244     * @return an object containing various cost measures such as the worst-case
245     *         number of disk accesses, the number of tuples produced, etc.
246     */
247    public final PlanCost getCost() {
248        return cost;
249    }
250
251
252    /**
253     * Returns statistics (possibly estimated) describing the results that this
254     * plan node will produce.  Estimating statistics for output results is a
255     * very imprecise task, to say the least.
256     * <p>
257     * These statistics are not computed until the {@link #prepare} method is
258     * called; until that point, this method will return <tt>null</tt>.
259     *
260     * @return statistics describing the results that will be produced by this
261     *         plan-node.
262     */
263    public final ArrayList<ColumnStats> getStats() {
264        return stats;
265    }

267
268    /**
269     * Returns the environment that this plan node uses during query
270     * evaluation.
271     *
272     * @return the environment this plan node uses during evaluation.
273     */
274    public Environment getEnvironment() {
275        return environment;
276    }

278
279
280    /**
281     * Sets the plan node's environment to the specified environment object.
282     * This can be used to connect subqueries to their parent environment,
283     * among other things.
284     *
285     * @param env the environment for this plan node to use during evaluation.
286     */
287    public void setEnvironment(Environment env) {
288        if (env == null)
289            throw new IllegalArgumentException("env cannot be null");
290
291        environment = env;
292    }
293
294
295    /**
296     * This method adds a parent environment to the entire plan tree rooted at
297     * this plan node.  This allows an execution plan to be used as a subquery
298     * that references an outer plan's tuples during evaluation.
299     *
300     * @param parentEnv the parent environment to add to this plan tree.
301     */
302    public void addParentEnvironmentToPlanTree(Environment parentEnv) {
303        if (parentEnv == null)
304            throw new IllegalArgumentException("parentEnv cannot be null");
305
306        environment.addParentEnvironment(parentEnv);
307
308        if (leftChild != null)
309            leftChild.addParentEnvironmentToPlanTree(parentEnv);
310
311        if (rightChild != null)
312            rightChild.addParentEnvironmentToPlanTree(parentEnv);
313    }
314
315
316    /**
317     * Does any initialization the node might need, including setting up the
318     * node's execution environment.  Subclasses of {@code PlanNode} can
319     * extend this method to perform other kinds of initialization, but they
320     * must be sure to call their parent class' implementation as well.
321     */
322    public void initialize() {
323        // Do nothing.
324    }
325
326
327    /**
328     * Gets the next tuple that fulfills the conditions for this plan node.
329     * If the node has a child, it should call getNextTuple() on the child.
330     * If the node is a leaf, the tuple comes from some external source such
331     * as a table file, the network, etc.
332     *
333     * @return the next tuple to be generated by this plan, or <tt>null</tt>
334     *         if the plan has finished generating plan nodes.
335     *
336     * @throws IOException if table data cannot be read from the filesystem
337     * @throws IllegalStateException if a plan node is not properly initialized
338     */
339    public abstract Tuple getNextTuple()
340        throws IllegalStateException, IOException;
341
342
343    /**
344     * Marks the current tuple in the tuple-stream produced by this node.  The
345     * {@link #resetToLastMark} method can be used to return to this tuple.
346     * Note that only one marker can be set in the tuple-stream at a time.
347     *
348     * @throws UnsupportedOperationException if the node does not support marking.
349     *
350     * @throws IllegalStateException if there is no "current tuple" to mark.
351     *         This will occur if {@link #getNextTuple} hasn't yet been called
352     *         (i.e. we are before the first tuple in the tuple-stream), or if
353     *         we have already reached the end of the tuple-stream (i.e. we are
354     *         after the last tuple in the stream).
355     */
356    public abstract void markCurrentPosition();
357
358
359    /**
360     * Resets the node's tuple-stream to the most recently marked position.
361     * Note that only one marker can be set in the tuple-stream at a time.
362     *
363     * @throws UnsupportedOperationException if the node does not support marking.
364     *
365     * @throws IllegalStateException if {@link #markCurrentPosition} hasn't yet
366     *         been called on this plan-node
367     */
368    public abstract void resetToLastMark();
369
370
371    /**
372     * Perform any necessary clean up tasks. This should probably be called
373     * when we are done with this plan node.
374     */
375    public abstract void cleanUp();

378    /**
379     * Reports this node and its vital parameters as a string.
380     *
381     * @return the node in string format.
382     *
383     * @design We re-declare this here to force its implementation in subclasses.
384     */
385    @Override
386    public abstract String toString();

389    /**
390     * Prints the entire plan subtree starting at this node.
391     *
392     * @param out PrintStream to be used for output.
393     * @param includeCosts a flag controlling whether the cost estimates will be
394     *        included in the plan output
395     * @param indent the indentation level.
396     */
397    public void printNodeTree(PrintStream out, boolean includeCosts, String indent) {
398        StringBuilder buf = new StringBuilder();
399        buf.append(indent);
400        buf.append(toString());
401        if (includeCosts) {
402            if (cost != null)
403                buf.append(" cost=").append(cost);
404            else
405                buf.append(" cost is unknown");
406        }
407
408        out.println(buf.toString());
409
410        if (leftChild != null)
411            leftChild.printNodeTree(out, includeCosts, indent + "    ");
412
413        if (rightChild != null)
414            rightChild.printNodeTree(out, includeCosts, indent + "    ");
415    }

417
418    /**
419     * Prints the entire node tree with indentation to the specified output
420     * stream starting from indentation level 0.
421     *
422     * @param out the output stream.
423     * @param includeCosts a flag controlling whether costs are included in the
424     *        output
425     */
426    public void printNodeTree(PrintStream out, boolean includeCosts) {
427        printNodeTree(out, includeCosts, "");
428    }

431    /**
432     * Prints the entire node tree with indentation to the specified output
433     * stream starting from indentation level 0.  Costs are not included in the
434     * output.
435     *
436     * @param out the output stream.
437     */
438    public void printNodeTree(PrintStream out) {
439        printNodeTree(out, false);
440    }

443    /**
444     * Generates the same result as {@link #printNodeTree(PrintStream)}, but
445     * into a string instead of to an output stream.
446     *
447     * @param plan The plan tree to print to a string.
448     * @param includeCosts a flag controlling whether costs are included in the
449     *        output
450     *
451     * @return A string containing the indented printout of the plan.
452     */
453    public static String printNodeTreeToString(PlanNode plan,
454                                               boolean includeCosts) {
455
456        // The character-encoding "US-ASCII" is defined to be supported
457        // on all Java implementations, so this should never throw.
458        try {
459            ByteArrayOutputStream baos = new ByteArrayOutputStream();
460            PrintStream ps = new PrintStream(baos, false, "US-ASCII");
461            plan.printNodeTree(ps, includeCosts);
462            ps.flush();
463
464            return baos.toString("US-ASCII");
465        }
466        catch (UnsupportedEncodingException e) {
467            throw new RuntimeException(e);
468        }
469    }

473     * Generates the same result as {@link #printNodeTree(PrintStream)}, but
474     * into a string instead of to an output stream.
475     *
476     * @param plan The plan tree to print to a string.
477     *
478     * @return A string containing the indented printout of the plan.
479     */
480    public static String printNodeTreeToString(PlanNode plan) {
481        return printNodeTreeToString(plan, false);

498     * Computes the hash-code of a plan-node, including any sub-plans of this

508    /**
509     * Creates a copy of this plan node and its subtree.  Note that this method
510     * is only used internally, because plan nodes also reference their parent
511     * node, and it is not possible for this method to set that parent-reference
512     * properly.  To create a deep copy of an entire plan tree, the
513     * {@link #duplicate} method should be used.
514     */
515    @Override
516    protected PlanNode clone() throws CloneNotSupportedException {
517        PlanNode node = (PlanNode) super.clone();
518
519        // NodeType is immutable.
520        node.nodeType = this.nodeType;
521
522        // Environment is used for execution.
523        node.environment = new Environment();
524
525        // Copy the children if applicable.
526        if (this.leftChild != null)
527            node.leftChild = this.leftChild.clone();
528        else
529            node.leftChild = null;
530
531        if (this.rightChild != null)
532            node.rightChild = this.rightChild.clone();
533        else
534            node.rightChild = null;
535
536        // PARENT CANNOT BE COPIED, SET TO NULL AND NOTE THAT WE NEED TO
537        // RUN THROUGH THE TREE AT THE END TO SET THE PARENT!
538
539        return node;
540    }

543    /**
544     * Iterates through the tree and sets node parents to correct references.
545     * This method should be called after cloning an entire tree because cloning
546     * cannot correctly clone parent references.
547     **/
548    private void setNodeParents() {
549        if (leftChild != null) {
550            leftChild.setNodeParents();
551        }
552        if (rightChild != null) {
553            rightChild.setNodeParents();
554        }
555    }
556
557
558    /**
559     * Returns a deep copy of this plan tree.
560     *
561     * @return a root to the duplicated plan tree.
562     */
563    public PlanNode duplicate() {
564        PlanNode dup = null;
565        try {
566            dup = clone();
567
568            // We have successfully cloned the node tree.
569            // Set the top node's parent to null and set all the other parents.
570            dup.setNodeParents();
571        }
572        catch (CloneNotSupportedException e) {
573            e.printStackTrace();
574        }
575
576        return dup;
577    }
