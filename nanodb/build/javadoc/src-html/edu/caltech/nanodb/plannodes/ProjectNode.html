018/**
019 * PlanNode representing the <tt>SELECT</tt> clause in a <tt>SELECT</tt>
020 * operation.  This is the relational algebra Project operator.
021 */
022public class ProjectNode extends PlanNode {

024    /** A logging object for reporting anything interesting that happens. **/
025    private static Logger logger = Logger.getLogger(ProjectNode.class);

028    /**
029     * This is our guess of how many unique values will be produced by a
030     * project expression, when we cannot determine it from the child plan's
031     * statistics.
032     */
033    private static final int GUESS_NUM_UNIQUE_VALUES = 100;

063    /**
064     * Constructs a ProjectNode that pulls tuples from a child node.
065     *
066     * @param leftChild the child to pull tuples from
067     * @param projectionSpec the set of expressions specifying how to project
068     *                       input tuples.
069     */
070    public ProjectNode(PlanNode leftChild, List<SelectValue> projectionSpec) {
071        // This node is a Project node.
072        super(OperationType.PROJECT);
073
074        // This node has a child, load child information.
075        this.leftChild = leftChild;
076
077        this.projectionSpec = projectionSpec;
078    }

081    /**
082     * Constructs a "leaf ProjectNode"; i.e. one that doesn't pull tuples
083     * from anywhere, but rather generates one tuple from its projection
084     * specification and then stops.
085     *
086     * @param projectionSpec the set of expressions specifying how to project
087     *                       input tuples.

089    public ProjectNode(List<SelectValue> projectionSpec) {
090        this(null, projectionSpec);
091    }

094    public void prepare() {
095        if (leftChild != null) {
096            // Need to prepare the left child-node before we can do our own
097            // work.
098            leftChild.prepare();
099
100            // Use the helper function to prepare the schema of this
101            // project-node, since it is a complicated operation.
102            prepareSchemaStats(leftChild.getSchema(), leftChild.getStats());
103
104            // Come up with a cost estimate now.  Projection does require some
105            // computation, so increase the CPU cost based on the number of
106            // tuples expected to come into this plan-node.
107
108            inputCost = leftChild.getCost();
109            if (inputCost != null) {
110                cost = new PlanCost(inputCost);
111                cost.cpuCost += inputCost.numTuples;
112            }
113            else {
114                logger.debug(
115                    "Child's cost not available; not computing this node's cost.");
116            }
117        }
118        else {
119            // The project operator is a leaf in the plan, so it must generate
120            // only one tuple.
121
122            // Prepare the schema and statistics using empty schema and stats
123            // values.
124            prepareSchemaStats(new Schema(), new ArrayList<ColumnStats>());
125
126            // Estimated number of tuples:  1
127            // Tuple size:  TODO
128            // Estimated CPU cost:  1
129            // Number of block IOs:  0
130            inputCost = new PlanCost(1, /* TODO:  tupleSize */ 100, 1, 0);
131            cost = inputCost;
132        }
133
134        // TODO:  Estimate the final tuple-size.  It isn't hard, just tedious.
135    }

138    /**
139     * This helper function computes the schema of the project plan-node, based
140     * on the schema of its child-plan, and also the expressions specified in
141     * the project operation.
142     */
143    protected void prepareSchemaStats(Schema inputSchema,
144                                      ArrayList<ColumnStats> inputStats) {
145        this.inputSchema = inputSchema;
146
147        schema = new Schema();
148        nonWildcardColumnInfos = new ArrayList<ColumnInfo>();
149
150        stats = new ArrayList<ColumnStats>();
151
152        for (SelectValue selVal : projectionSpec) {
153            if (selVal.isWildcard()) {
154                ColumnName wildcard = selVal.getWildcard();
155                if (wildcard.isTableSpecified()) {
156                    // Need to find all columns that are associated with the
157                    // specified table.
158                    SortedMap<Integer, ColumnInfo> found =
159                        inputSchema.findColumns(wildcard);
160
161                    // Add each column that was found, as well as its stats.
162                    schema.append(found.values());
163                    for (Integer idx : found.keySet())
164                        stats.add(inputStats.get(idx));
165                }
166                else {
167                    // No table is specified, so this is all columns in the
168                    // child schema.
169                    schema.append(inputSchema);
170                    stats.addAll(inputStats);
171                }
172            }
173            else if (selVal.isExpression()) {
174                // Determining the schema is relatively straightforward.  The
175                // statistics, unfortunately, are a different matter:  if the
176                // expression is a simple column-reference then we can look up
177                // the stats from the subplan, but if the expression is an
178                // arithmetic operation, we need to guess...
179
180                Expression expr = selVal.getExpression();
181                ColumnInfo colInfo;
182
183                if (expr instanceof ColumnValue) {
184                    // This is a simple column-reference.  Pull out the schema
185                    // and the statistics from the input.
186                    ColumnValue colValue = (ColumnValue) expr;
187                    int colIndex = inputSchema.getColumnIndex(colValue.getColumnName());
188                    colInfo = inputSchema.getColumnInfo(colIndex);
189                    stats.add(inputStats.get(colIndex));
190                }
191                else {
192                    // This is a more complicated expression.  Guess the schema,
193                    // and assume that every row will have a distinct value.
194
195                    colInfo = expr.getColumnInfo(inputSchema);
196
197                    // TODO:  We could be more sophisticated about this...
198                    ColumnStats colStat = new ColumnStats();
199
200                    if (inputCost != null) {
201                        // Adding 0.5 and casting to int is equivalent to
202                        // rounding, as long as the input >= 0.
203                        colStat.setNumUniqueValues(
204                            (int) (inputCost.numTuples + 0.5f));
205                    }
206                    else {
207                        colStat.setNumUniqueValues(GUESS_NUM_UNIQUE_VALUES);
208                    }
209
210                    stats.add(colStat);
211                }
212
213                // Apply any aliases here...
214                String alias = selVal.getAlias();
215                if (alias != null)
216                    colInfo = new ColumnInfo(alias, colInfo.getType());
217
218                schema.addColumnInfo(colInfo);
219                nonWildcardColumnInfos.add(colInfo);
220            }
221            else if (selVal.isScalarSubquery()) {
222                throw new UnsupportedOperationException(
223                    "Scalar subquery support is currently incomplete.");
224            }
225        }
226    }

255    /**
256     * Gets the next tuple and projects it.
257     *
258     * @return the tuple to be passed up to the next node.
259     *
260     * @throws java.io.IOException if a db file failed to open at some point
261     */
262    public Tuple getNextTuple() throws IOException {
263
264        // If this node is finished finding tuples, return null until it is
265        // re-initialized.
266        if (done)
267            return null;
268
269        if (leftChild != null) {
270            // Advance the left child to the next tuple.  If there are no more
271            // tuples, the projection process is over - set the done flag.
272            advanceCurrentTuple();
273            if (currentTuple == null)
274                done = true;
275        }
276        else {
277            // Since we are here, we know that done == false and we haven't
278            // generated our tuple yet.
279            currentTuple = new TupleLiteral();
280            done = true;
281        }
282
283        if (currentTuple != null)
284            return projectTuple(currentTuple);
285
286        done = true;
287        return null;
288    }

291    /** Helper function that advances the current tuple reference in the node.
292     *
293     * @throws java.lang.IllegalStateException if this is a node no child.
294     * @throws java.io.IOException if a db file failed to open at some point
295     */
296    private void advanceCurrentTuple() throws IOException {
297        currentTuple = leftChild.getNextTuple();
298    }

301    /**
302     * This helper method takes an input tuple and projects it to a result tuple
303     * based on the project
304     *
305     * @param tuple the tuple to project
306     *
307     * @return the projected version of the tuple
308     */
309    private Tuple projectTuple(Tuple tuple) {
310
311        // If the projection-spec is simply a single wildcard value, e.g.
312        // "SELECT * FROM foo", then just return the tuple unmodified.
313        if (isTrivial())
314            return tuple;
315
316        // The projection is *not* trivial, so we need to do some evaluatin'.
317
318        environment.clear();
319        environment.addTuple(inputSchema, tuple);
320
321        // Create an empty tuple to add values to.
322        TupleLiteral newTuple = new TupleLiteral();
323
324        // For each select value, evaluate it and add it to the tuple.
325
326        Iterator<ColumnInfo> iterNonWildcardCols =
327            nonWildcardColumnInfos.iterator();
328
329        for (SelectValue selVal : projectionSpec) {
330            if (selVal.isWildcard()) {
331                // This value is a wildcard.  Find the columns that match the
332                // wildcard, then add their values one by one.
333
334                // Wildcard expressions cannot rename their results.
335
336                ColumnName wildcard = selVal.getWildcard();
337                if (wildcard.isTableSpecified()) {
338                    // Need to find all columns that are associated with the
339                    // specified table.
340
341                    SortedMap<Integer, ColumnInfo> matchCols =
342                        inputSchema.findColumns(wildcard);
343
344                    for (int iCol : matchCols.keySet())
345                        newTuple.addValue(tuple.getColumnValue(iCol));
346                }
347                else {
348                    // No table is specified, so this is all columns in the
349                    // child schema.
350                    newTuple.appendTuple(tuple);
351                }
352            }
353            else if (selVal.isExpression()) {
354                // This value is a simple expression.
355                Expression expr = selVal.getExpression();
356                String alias = selVal.getAlias();
357
358                // Get the result of the projection for this value.
359
360                Object result = expr.evaluate(environment);
361                ColumnInfo colInfo = iterNonWildcardCols.next();
362
363                logger.debug(String.format(
364                    "Expression:  %s \tColInfo:  %s\tAlias:  %s",
365                    expr, colInfo, alias));
366
367                // Add the result to the tuple.
368
369                /*
370                if (alias != null)
371                    colInfo = new ColumnInfo(alias, colInfo.getType());
372
373                logger.debug(String.format(
374                    "Result:  %s \tColInfo:  %s\n", result, colInfo));
375                */
376
377                newTuple.addValue(result);
378            }
379            else if (selVal.isScalarSubquery()) {
380                throw new UnsupportedOperationException(
381                    "Scalar subquery support is currently incomplete");
382            }
383            else {
384                throw new IllegalStateException(
385                    "Select-value doesn't specify a value");
386            }
387        }
388
389        return newTuple;
390    }

393    /** Do initialization for the select operation.  Resets state variables. */
394    public void initialize() {
395        super.initialize();
396
397        done = false;
398        currentTuple = null;
399
400        if (leftChild != null)
401            leftChild.initialize();
402    }

427    /**
428     * Checks if the argument is a plan node tree with the same structure,
429     * but not necesarily the same references.
430     *
431     * @param obj the object to which we are comparing
432     */
433    @Override
434    public boolean equals(Object obj) {
435
436        if (obj instanceof ProjectNode) {
437            ProjectNode other = (ProjectNode) obj;
438
439            return projectionSpec.equals(other.projectionSpec) &&
440                (leftChild == null && other.leftChild == null ||
441                 leftChild != null && leftChild.equals(other.leftChild));
442        }
443
444        return false;
445    }

448    /** Computes and returns the hash-code of a project node. */
449    @Override
450    public int hashCode() {
451        int hash = 7;
452        hash = 31 * hash + projectionSpec.hashCode();
453        hash = 31 * hash + (leftChild != null ? leftChild.hashCode() : 0);
454        return hash;
455    }

469    /**
470     * Creates a copy of this project node and its subtree.  This method is used
471     * by {@link PlanNode#duplicate} to copy a plan tree.
472     */
473    @Override
474    protected PlanNode clone() throws CloneNotSupportedException {
475        ProjectNode node = (ProjectNode) super.clone();
476
477        ArrayList<SelectValue> newList = new ArrayList<SelectValue>();
478        for (SelectValue sel : this.projectionSpec) {
479            SelectValue newSel = (SelectValue) sel.clone();
480            newList.add(newSel);
481        }
482        node.projectionSpec = newList;
483
484        return node;
485    }

488    /**
489     * Returns true if the project node is a trivial <tt>*</tt> projection
490     * with no table references.  If the projection is trivial then it could
491     * even be removed.
492     *
493     * @return true if the select value is a full wildcard value, not even
494     *         specifying a table name
495     */
496    public boolean isTrivial() {
497        if (projectionSpec.size() == 1) {
498            SelectValue sel = projectionSpec.get(0);
499            if (sel.isWildcard()) {
500                ColumnName wildcard = sel.getWildcard();
501                if (wildcard.getTableName() == null) {
502                    return true;
503                }
504            }
505        }
506
507        return false;
508    }
