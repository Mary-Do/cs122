008/**
009 * This Command class represents the <tt>EXIT</tt> or <tt>QUIT</tt> SQL
010 * commands.  These commands aren't standard SQL of course, but are the
011 * way that we tell the database to stop.
012 */
013public class ExitCommand extends Command {
014
015    /** Construct an exit command. */
016    public ExitCommand() {
017        super(Command.Type.UTILITY);
018    }
019
020    /**
021     * This method really doesn't do anything, and it isn't intended to be
022     * called.  The server looks for this command when commands are being
023     * executed, and handles it separately.
024     *
025     * @review (Donnie) We could actually have this command operate on the
026     *         server now that we pass it in...  Need to think about this.
027     */
028    @Override
029    public void execute(NanoDBServer server) throws ExecutionException {
030        // Do nothing.
031    }
032}
