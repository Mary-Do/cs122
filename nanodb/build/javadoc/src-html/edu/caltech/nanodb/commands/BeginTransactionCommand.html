001package edu.caltech.nanodb.commands;
002
003
004import edu.caltech.nanodb.server.NanoDBServer;
005import edu.caltech.nanodb.storage.StorageManager;
006import edu.caltech.nanodb.transactions.TransactionException;
007
008
009/**
010 * This class represents a command that starts a transaction, such as
011 * <tt>BEGIN</tt>, <tt>BEGIN WORK</tt>, or <tt>START TRANSACTION</tt>.
012 */
013public class BeginTransactionCommand extends Command {
014    public BeginTransactionCommand() {
015        super(Type.UTILITY);
016    }
017
018
019    @Override
020    public void execute(NanoDBServer server) throws ExecutionException {
021        // Begin a transaction.
022        try {
023            // Pass true for the "user-started transaction" flag, since the
024            // user issued the command to do it!
025            StorageManager storageManager = server.getStorageManager();
026            storageManager.getTransactionManager().startTransaction(true);
027        }
028        catch (TransactionException e) {
029            throw new ExecutionException(e);
030        }
031    }
032}
