194    public boolean isTrivialProject() {
195        if (selectValues.size() == 1) {
196            SelectValue selVal = selectValues.get(0);
197            if (selVal.isWildcard() && !selVal.getWildcard().isTableSpecified())
198                return true;
199        }
200        return false;
201    }

341    /**
342     * This method computes the resulting schema from this query, and in the
343     * process it performs various semantic checks as well.
344     *
345     * @param tableManager the table manager to use for retrieving schema info
346     *
347     * @param parentSelect the enclosing SELECT query if this is a nested
348     *        subquery, or {@code null} if this is a top-level query
349     *
350     * @return the schema of this select clause's result
351     *
352     * @throws IOException if the database cannot read schema from the disk
353     *         along the way
354     *
355     * @throws SchemaNameException if the select clause contains some kind of
356     *         semantic error involving schemas that are referenced
357     */
358    public Schema computeSchema(TableManager tableManager,
359        SelectClause parentSelect) throws IOException, SchemaNameException {
360
361        this.parentSelect = parentSelect;
362
363        // Compute the schema of the FROM clause first.  We assume that no
364        // subqueries in the FROM clause are correlated with this or enclosing
365        // queries.
366        if (fromClause != null) {
367            fromSchema = fromClause.computeSchema(tableManager);
368        }
369        else {
370            // No FROM clause - no FROM schema...
371            fromSchema = new Schema();
372        }
373
374        // This helper-object is used for handling subqueries in the SELECT,
375        // WHERE and HAVING clauses.  These nested queries need to have their
376        // schemas computed, and possible correlated evaluation identified.
377        SubquerySchemaComputer subquerySchemaComputer =
378            new SubquerySchemaComputer(this, tableManager);
379
380        // Make sure that all expressions in this SELECT clause reference
381        // known and non-ambiguous names from the FROM clause.
382
383        // SELECT values:  SELECT a, b + c, tbl.* ...
384        resultSchema = new Schema();
385        Set<String> fromTables = fromSchema.getTableNames();
386        for (SelectValue selVal : selectValues) {
387            if (selVal.isWildcard()) {
388                // Make sure that if a table name is specified, that the table
389                // name is in the query's FROM-clause schema.
390                ColumnName colName = selVal.getWildcard();
391                if (colName.isTableSpecified()) {
392                    if (!fromTables.contains(colName.getTableName())) {
393                        throw new SchemaNameException("SELECT-value " + colName +
394                            " specifies an unrecognized table name.");
395                    }
396                }
397            }
398            else {
399                // An expression that is not a wildcard.  It could contain
400                // column references that need to be resolved.  Also, it could
401                // contain a scalar subquery, particularly one that requires
402                // correlated evaluation.
403                Expression expr = selVal.getExpression();
404
405                // Get the list of column-values, and resolve each one.
406                // Note that this will not descend into subqueries.
407                resolveExpressionRefs("SELECT-value", expr, fromSchema,
408                    /* checkParentQueries */ true);
409
410                // If the expression contains any subqueries, resolve any
411                // column-references that reference this or enclosing queries.
412                expr.traverse(subquerySchemaComputer);
413            }
414
415            // Update the result-schema with this select-value's column-info(s).
416            resultSchema.append(selVal.getColumnInfos(fromSchema, resultSchema));
417        }
418
419        logger.debug("Query schema:  " + resultSchema);
420        logger.debug("FROM-clause schema:  " + fromSchema);
421
422        // WHERE clause:
423        if (whereExpr != null) {
424            resolveExpressionRefs("WHERE clause", whereExpr, fromSchema,
425                /* checkParentQueries */ true);
426
427            whereExpr.traverse(subquerySchemaComputer);
428        }
429
430        // GROUP BY clauses:
431        for (Expression expr : groupByExprs) {
432            // GROUP BY expressions aren't allowed to have subqueries.
433            resolveExpressionRefs("GROUP BY clause", expr, fromSchema,
434                /* checkParentQueries */ false);
435        }
436
437        // HAVING clause:
438        if (havingExpr != null) {
439            // The HAVING clause can reference values produced by grouping and
440            // aggregation.  However, according to the SQL standard, it won't
441            // see any values except from the FROM-clause's schema.
442            resolveExpressionRefs("HAVING clause", havingExpr, fromSchema,
443                /* checkParentQueries */ true);
444
445            havingExpr.traverse(subquerySchemaComputer);
446        }
447
448        // ORDER BY clauses:
449        for (OrderByExpression expr : orderByExprs) {
450            // ORDER BY expressions aren't allowed to have subqueries.
451            resolveExpressionRefs("ORDER BY clause", expr.getExpression(),
452                fromSchema, /* checkParentQueries */ false);
453        }
454
455        // All done!  Return the computed schema.
456        return resultSchema;
457    }

471    /**
472     * This helper function goes through the expression and verifies that
473     * every symbol-reference corresponds to an actual value produced by the
474     * <tt>FROM</tt>-clause of the <tt>SELECT</tt> query.  Any column-names
475     * that don't include a table-name are also updated to include the proper
476     * table-name.
477     * <p>
478     * It's possible to have symbols that don't reference columns in the
479     * <tt>FROM</tt> clause, if a query is correlated with an enclosing query.
480     * In these cases, the unresolvable attributes are collected so they can
481     * be resolved at the end of the process.
482     *
483     * @param desc A short string describing the context of the expression,
484     *        since expressions can appear in the <tt>SELECT</tt> clause, the
485     *        <tt>WHERE</tt> clause, the <tt>GROUP BY</tt> clause, etc.
486     *
487     * @param expr The expression that will be evaluated.
488     *
489     * @param s The schema against which the expression will be evaluated.
490     *
491     * @param checkParentQueries if this is true and the column-name can't be
492     *        resolved against the current query, any parent queries will also
493     *        be checked for the column-name.  This allows correlated queries
494     *        to be resolved properly.
495     *
496     * @throws SchemaNameException if an expression-reference cannot be resolved
497     *         against the specified schema, either because the named column
498     *         or table doesn't appear in the schema, or if a column name is
499     *         ambiguous.
500     */
501    private void resolveExpressionRefs(String desc, Expression expr, Schema s,
502        boolean checkParentQueries) throws SchemaNameException {
503
504        // Get the list of column-values in the expression, and resolve each one.
505        // (This won't include subquery expressions, since they will reference
506        // the subquery's schema.)
507        ArrayList<ColumnName> exprColumns = new ArrayList<>();
508        expr.getAllSymbols(exprColumns);
509
510        for (ColumnName colName : exprColumns) {
511            assert !colName.isColumnWildcard();
512            resolveColumnRef(desc, expr, colName, s, checkParentQueries);
513        }
514    }

517    /**
518     * This helper function attempts to resolve a specific column-name against
519     * a query's schema.
520
521     * @param desc A short string describing the context of the expression,
522     *        since expressions can appear in the <tt>SELECT</tt> clause, the
523     *        <tt>WHERE</tt> clause, the <tt>GROUP BY</tt> clause, etc.
524     *
525     * @param expr The expression that will be evaluated.
526     *
527     * @param colName The column-name to resolve.
528     *
529     * @param s The schema against which the expression will be evaluated.
530     *
531     * @param checkParentQueries if this is true and the column-name can't be
532     *        resolved against the current query, any parent queries will also
533     *        be checked for the column-name.  This allows correlated queries
534     *        to be resolved properly.
535     *
536     * @throws SchemaNameException if an expression-reference cannot be resolved
537     *         against the specified schema, either because the named column
538     *         or table doesn't appear in the schema, or if a column name is
539     *         ambiguous.
540     */
541    private void resolveColumnRef(String desc, Expression expr,
542        ColumnName colName, Schema s, boolean checkParentQueries)
543        throws SchemaNameException {
544
545        // This flag indicates whether we are referencing a column in a parent
546        // query.
547        boolean parentRef = false;
548
549        // This is the query that we are examining.
550        SelectClause clause = this;
551        while (true) {
552            // Attempt to resolve the column-name against the current schema's
553            // columns.
554            SortedMap<Integer, ColumnInfo> found = s.findColumns(colName);
555
556            if (found.size() == 1) {
557                // Found exactly one column that matches!
558                if (!colName.isTableSpecified()) {
559                    // Update the column-reference with the table name.
560                    ColumnInfo colInfo = found.get(found.firstKey());
561                    colName.setTableName(colInfo.getTableName());
562                }
563
564                if (parentRef) {
565                    // This column-name references a column produced by a
566                    // parent query, not this query.
567                    assert clause != this;
568
569                    // Sanity-check:  have we already seen this column name?
570                    // If so, make sure that we find the same clause each time
571                    if (correlatedWith.containsKey(colName) &&
572                        correlatedWith.get(colName) != clause) {
573                        throw new IllegalStateException(String.format(
574                            "Column name %s is associated with two " +
575                            "different queries", colName));
576                    }
577                    correlatedWith.put(colName, clause);
578                }
579
580                return;
581            }
582
583            if (found.size() > 1) {
584                // In this case the column name is ambiguous - there may be
585                // multiple columns in the schema with the same column name
586                // but different table names.
587                throw new SchemaNameException(String.format(
588                    "%s %s contains an ambiguous column %s; found %s.",
589                    desc, expr, colName, found.values()));
590            }
591
592            // We should only get to this point in the loop if we couldn't find
593            // a column with the specified name.
594            assert found.size() == 0;
595
596            if (checkParentQueries && clause.parentSelect != null) {
597                clause = clause.parentSelect;
598                s = clause.getFromSchema();
599                parentRef = true;
600            }
601            else {
602                throw new SchemaNameException(String.format(
603                    "%s %s references an unknown column %s.",
604                    desc, expr, colName));
605            }
606        }
607    }

610    @Override
611    public String toString() {
612        StringBuilder buf = new StringBuilder();
613
614        buf.append("SelectClause[id=").append(System.identityHashCode(this)).append('\n');
615
616        if (selectValues.size() > 0)
617            buf.append("\tvalues=").append(selectValues).append('\n');
618
619        if (fromClause != null)
620            buf.append("\tfrom=").append(fromClause).append('\n');
621
622        if (whereExpr != null)
623            buf.append("\twhere=").append(whereExpr).append('\n');
624
625        if (groupByExprs.size() > 0)
626            buf.append("\tgroup_by=").append(groupByExprs).append('\n');
627
628        if (havingExpr != null)
629            buf.append("\thaving=").append(havingExpr).append('\n');
630
631        if (orderByExprs.size() > 0)
632            buf.append("\torder_by=").append(orderByExprs).append('\n');
633
634        if (correlatedWith.size() > 0) {
635            buf.append("\tcorrelatedWith=[");
636
637            boolean first = true;
638            for (Map.Entry e: correlatedWith.entrySet()) {
639                if (first)
640                    first = false;
641                else
642                    buf.append(',');
643
644                buf.append(e.getKey()).append(':');
645                buf.append(System.identityHashCode(e.getValue()));
646            }
647            buf.append("]\n");
648        }
649
650        buf.append(']');
651
652        return buf.toString();
653    }
