023 * This utility class is used to estimate the selectivity of predicates that
024 * appear on Select and Theta-Join plan-nodes.

032    /**
033     * This collection specifies the data-types that support comparison
034     * selectivity estimates (not including equals or not-equals).  It must be
035     * possible to use the {@link #computeRatio} on the data-type so that an
036     * estimate can be made about where a value fits within the minimum and
037     * maximum values for the column.
038     * <p>
039     * Note that we can compute selectivity for equals and not-equals simply
040     * from the number of distinct values in the column.
041     */
042    private static HashSet<SQLDataType> SUPPORTED_TYPES_COMPARE_ESTIMATES;
043
044
045    static {
046        // Initialize the set of types that support comparison selectivity
047        // estimates.  In time, types like dates, times, NUMERIC, etc. could be
048        // added as well.
049
050        SUPPORTED_TYPES_COMPARE_ESTIMATES = new HashSet<SQLDataType>();
051
052        SUPPORTED_TYPES_COMPARE_ESTIMATES.add(SQLDataType.INTEGER);
053        SUPPORTED_TYPES_COMPARE_ESTIMATES.add(SQLDataType.BIGINT);
054        SUPPORTED_TYPES_COMPARE_ESTIMATES.add(SQLDataType.SMALLINT);
055        SUPPORTED_TYPES_COMPARE_ESTIMATES.add(SQLDataType.TINYINT);
056        SUPPORTED_TYPES_COMPARE_ESTIMATES.add(SQLDataType.FLOAT);
057        SUPPORTED_TYPES_COMPARE_ESTIMATES.add(SQLDataType.DOUBLE);
058    }

061    /**
062     * This constant specifies the default selectivity assumed when a select
063     * predicate is too complicated to compute more accurate estimates.  We are
064     * assuming that generally people are going to do things that limit the
065     * results produced.
066     */
067    public static final float DEFAULT_SELECTIVITY = 0.25f;

076    /**
077     * Returns true if the database supports selectivity estimates for
078     * comparisons (other than equals and not-equals) on the specified SQL data
079     * type.  SQL types that support these selectivity estimates will include
080     * minimum and maximum values in their column-statistics.
081     *
082     * @param type the SQL data type being considered
083     *
084     * @return true if the database supports selectivity estimates for the type
085     */
086    public static boolean typeSupportsCompareEstimates(SQLDataType type) {
087        return SUPPORTED_TYPES_COMPARE_ESTIMATES.contains(type);
088    }

091    /**
092     * This function computes the selectivity of a selection predicate, using
093     * table statistics and other estimates to make an educated guess.  The
094     * result is between 0.0 and 1.0, with 1.0 meaning that all rows will be
095     * selected by the predicate.
096     *
097     * @param expr the expression whose selectivity we are estimating
098     *
099     * @param exprSchema a schema describing the environment that the expression
100     *        will be evaluated within
101     *
102     * @param stats statistics that may be helpful in estimating the selectivity
103     *
104     * @return the estimated selectivity as a float
105     */
106    public static float estimateSelectivity(Expression expr, Schema exprSchema,
107                                            ArrayList<ColumnStats> stats) {
108        float selectivity = DEFAULT_SELECTIVITY;
109
110        if (expr instanceof BooleanOperator) {
111            // A Boolean AND, OR, or NOT operation.
112            BooleanOperator bool = (BooleanOperator) expr;
113            selectivity = estimateBoolOperSelectivity(bool, exprSchema, stats);
114        }
115        else if (expr instanceof CompareOperator) {
116            // This is a simple comparison between expressions.
117            CompareOperator comp = (CompareOperator) expr;
118            selectivity = estimateCompareSelectivity(comp, exprSchema, stats);
119        }
120
121        return selectivity;
122    }

175    /**
176     * This function computes a selectivity estimate for a general comparison
177     * operation.  The method examines the types of the arguments in the
178     * comparison and determines if it will be possible to make a reasonable
179     * guess as to the comparison's selectivity; if not then a default
180     * selectivity estimate is used.
181     *
182     * @param comp the comparison expression
183     *
184     * @param exprSchema a schema specifying the environment that the expression
185     *        will be evaluated within
186     *
187     * @param stats a collection of column-statistics to use in making
188     *        selectivity estimates
189     *
190     * @return a selectivity estimate in the range [0, 1].
191     */
192    public static float estimateCompareSelectivity(CompareOperator comp,
193        Schema exprSchema, ArrayList<ColumnStats> stats) {
194
195        float selectivity = DEFAULT_SELECTIVITY;
196
197        // Move the comparison into a normalized order so that it's easier to
198        // write the logic for analysis.  Specifically, this will ensure that
199        // if we are comparing a column and a value, the column will always be
200        // on the left and the value will always be on the right.
201        comp.normalize();
202
203        Expression left = comp.getLeftExpression();
204        Expression right = comp.getRightExpression();
205
206        // If the comparison is simple enough then compute its selectivity.
207        // Otherwise, just use the default selectivity.
208        if (left instanceof ColumnValue && right instanceof LiteralValue) {
209            // Comparison:  column op value
210            selectivity = estimateCompareColumnValue(comp.getType(),
211                (ColumnValue) left, (LiteralValue) right, exprSchema, stats);
212
213            logger.debug("Estimated selectivity of cmp-col-val operator \"" +
214                comp + "\" as " + selectivity);
215        }
216        else if (left instanceof ColumnValue && right instanceof ColumnValue) {
217            // Comparison:  column op column
218            selectivity = estimateCompareColumnColumn(comp.getType(),
219                (ColumnValue) left, (ColumnValue) right, exprSchema, stats);
220
221            logger.debug("Estimated selectivity of cmp-col-col operator \"" +
222                comp + "\" as " + selectivity);
223        }
224
225        return selectivity;
226    }

367    /**
368     * This method computes the function
369     * (<em>high</em><sub>1</sub> - <em>low</em><sub>1</sub>) /
370     * (<em>high</em><sub>2</sub> - <em>low</em><sub>2</sub>), given
371     * <tt>Object</tt>-values that can be coerced into types that can
372     * be used for arithmetic.  This operation is useful for estimating the
373     * selectivity of comparison operations, if we know the minimum and maximum
374     * values for a column.
375     * <p>
376     * The result of this operation is clamped to the range [0, 1].
377     *
378     * @param low1 the low value for the numerator
379     * @param high1 the high value for the numerator
380     * @param low2 the low value for the denominator
381     * @param high2 the high value for the denominator
382     *
383     * @return the ratio of (<em>high</em><sub>1</sub> - <em>low</em><sub>1</sub>) /
384     *         (<em>high</em><sub>2</sub> - <em>low</em><sub>2</sub>), clamped
385     *         to the range [0, 1].
386     */
387    private static float computeRatio(Object low1, Object high1,
388                                      Object low2, Object high2) {
389
390        Object diff1 = ArithmeticOperator.evalObjects(
391            ArithmeticOperator.Type.SUBTRACT, high1, low1);
392
393        Object diff2 = ArithmeticOperator.evalObjects(
394            ArithmeticOperator.Type.SUBTRACT, high2, low2);
395
396        Object ratio = ArithmeticOperator.evalObjects(
397            ArithmeticOperator.Type.DIVIDE, diff1, diff2);
398
399        float fltRatio = TypeConverter.getFloatValue(ratio);
400
401        logger.debug(String.format("Ratio:  (%s - %s) / (%s - %s) = %.2f",
402            high1, low1, high2, low2, fltRatio));
403
404        // Clamp the value to the range [0, 1].
405        if (fltRatio < 0.0f)
406            fltRatio = 0.0f;
407        else if (fltRatio > 1.0f)
408            fltRatio = 1.0f;
409
410        return fltRatio;
411    }
