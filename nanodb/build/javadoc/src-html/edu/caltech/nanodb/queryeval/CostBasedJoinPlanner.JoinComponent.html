025/**
026 * This planner implementation uses dynamic programming to devise an optimal
027 * join strategy for the query.  As always, queries are optimized in units of
028 * <tt>SELECT</tt>-<tt>FROM</tt>-<tt>WHERE</tt> subqueries; optimizations
029 * don't currently span multiple subqueries.
030 */
031public class CostBasedJoinPlanner extends AbstractPlannerImpl {

037    /**
038     * This helper class is used to keep track of one "join component" in the
039     * dynamic programming algorithm.  A join component is simply a query plan
040     * for joining one or more leaves of the query.
041     * <p>
042     * In this context, a "leaf" may either be a base table or a subquery in
043     * the <tt>FROM</tt>-clause of the query.  However, the planner will
044     * attempt to push conjuncts down the plan as far as possible, so even if
045     * a leaf is a base table, the plan may be a bit more complex than just a
046     * single file-scan.
047     */
048    private static class JoinComponent {
049        /**
050         * This is the join plan itself, that joins together all leaves
051         * specified in the {@link #leavesUsed} field.
052         */
053        public PlanNode joinPlan;
054
055        /**
056         * This field specifies the collection of leaf-plans that are joined by
057         * the plan in this join-component.
058         */
059        public HashSet<PlanNode> leavesUsed;
060
061        /**
062         * This field specifies the collection of all conjuncts use by this join
063         * plan.  It allows us to easily determine what join conjuncts still
064         * remain to be incorporated into the query.
065         */
066        public HashSet<Expression> conjunctsUsed;
067
068        /**
069         * Constructs a new instance for a <em>leaf node</em>.  It should not
070         * be used for join-plans that join together two or more leaves.  This
071         * constructor simply adds the leaf-plan into the {@link #leavesUsed}
072         * collection.
073         *
074         * @param leafPlan the query plan for this leaf of the query.
075         *
076         * @param conjunctsUsed the set of conjuncts used by the leaf plan.
077         *        This may be an empty set if no conjuncts apply solely to
078         *        this leaf, or it may be nonempty if some conjuncts apply
079         *        solely to this leaf.
080         */
081        public JoinComponent(PlanNode leafPlan, HashSet<Expression> conjunctsUsed) {
082            leavesUsed = new HashSet<>();
083            leavesUsed.add(leafPlan);
084
085            joinPlan = leafPlan;
086
087            this.conjunctsUsed = conjunctsUsed;
088        }
089
090        /**
091         * Constructs a new instance for a <em>non-leaf node</em>.  It should
092         * not be used for leaf plans!
093         *
094         * @param joinPlan the query plan that joins together all leaves
095         *        specified in the <tt>leavesUsed</tt> argument.
096         *
097         * @param leavesUsed the set of two or more leaf plans that are joined
098         *        together by the join plan.
099         *
100         * @param conjunctsUsed the set of conjuncts used by the join plan.
101         *        Obviously, it is expected that all conjuncts specified here
102         *        can actually be evaluated against the join plan.
103         */
104        public JoinComponent(PlanNode joinPlan, HashSet<PlanNode> leavesUsed,
105                             HashSet<Expression> conjunctsUsed) {
106            this.joinPlan = joinPlan;
107            this.leavesUsed = leavesUsed;
108            this.conjunctsUsed = conjunctsUsed;
109        }
110    }

153    /**
154     * Given the top-level {@code FromClause} for a SELECT-FROM-WHERE block,
155     * this helper generates an optimal join plan for the {@code FromClause}.
156     *
157     * @param fromClause the top-level {@code FromClause} of a
158     *        SELECT-FROM-WHERE block.
159     * @param extraConjuncts any extra conjuncts (e.g. from the WHERE clause,
160     *        or HAVING clause)
161     * @return a {@code JoinComponent} object that represents the optimal plan
162     *         corresponding to the FROM-clause
163     * @throws IOException if an IO error occurs during planning.
164     */
165    private JoinComponent makeJoinPlan(FromClause fromClause,
166        Collection<Expression> extraConjuncts) throws IOException {
167
168        // These variables receive the leaf-clauses and join conjuncts found
169        // from scanning the sub-clauses.  Initially, we put the extra conjuncts
170        // into the collection of conjuncts.
171        HashSet<Expression> conjuncts = new HashSet<>();
172        ArrayList<FromClause> leafFromClauses = new ArrayList<>();
173
174        collectDetails(fromClause, conjuncts, leafFromClauses);
175
176        logger.debug("Making join-plan for " + fromClause);
177        logger.debug("    Collected conjuncts:  " + conjuncts);
178        logger.debug("    Collected FROM-clauses:  " + leafFromClauses);
179        logger.debug("    Extra conjuncts:  " + extraConjuncts);
180
181        if (extraConjuncts != null)
182            conjuncts.addAll(extraConjuncts);
183
184        // Make a read-only set of the input conjuncts, to avoid bugs due to
185        // unintended side-effects.
186        Set<Expression> roConjuncts = Collections.unmodifiableSet(conjuncts);
187
188        // Create a subplan for every single leaf FROM-clause, and prepare the
189        // leaf-plan.
190
191        logger.debug("Generating plans for all leaves");
192        ArrayList<JoinComponent> leafComponents = generateLeafJoinComponents(
193            leafFromClauses, roConjuncts);
194
195        // Print out the results, for debugging purposes.
196        if (logger.isDebugEnabled()) {
197            for (JoinComponent leaf : leafComponents) {
198                logger.debug("    Leaf plan:\n" +
199                    PlanNode.printNodeTreeToString(leaf.joinPlan, true));
200            }
201        }
202
203        // Build up the full query-plan using a dynamic programming approach.
204
205        JoinComponent optimalJoin =
206            generateOptimalJoin(leafComponents, roConjuncts);
207
208        PlanNode plan = optimalJoin.joinPlan;
209        logger.info("Optimal join plan generated:\n" +
210            PlanNode.printNodeTreeToString(plan, true));
211
212        return optimalJoin;
213    }

235    /**
236     * This helper method performs the first step of the dynamic programming
237     * process to generate an optimal join plan, by generating a plan for every
238     * leaf from-clause identified from analyzing the query.  Leaf plans are
239     * usually very simple; they are built either from base-tables or
240     * <tt>SELECT</tt> subqueries.  The most complex detail is that any
241     * conjuncts in the query that can be evaluated solely against a particular
242     * leaf plan-node will be associated with the plan node.  <em>This is a
243     * heuristic</em> that usually produces good plans (and certainly will for
244     * the current state of the database), but could easily interfere with
245     * indexes or other plan optimizations.
246     *
247     * @param leafFromClauses the collection of from-clauses found in the query
248     *
249     * @param conjuncts the collection of conjuncts that can be applied at this
250     *                  level
251     *
252     * @return a collection of {@link JoinComponent} object containing the plans
253     *         and other details for each leaf from-clause
254     *
255     * @throws IOException if a particular database table couldn't be opened or
256     *         schema loaded, for some reason
257     */
258    private ArrayList<JoinComponent> generateLeafJoinComponents(
259        Collection<FromClause> leafFromClauses, Collection<Expression> conjuncts)
260        throws IOException {
261
262        // Create a subplan for every single leaf FROM-clause, and prepare the
263        // leaf-plan.
264        ArrayList<JoinComponent> leafComponents = new ArrayList<>();
265        for (FromClause leafClause : leafFromClauses) {
266            HashSet<Expression> leafConjuncts = new HashSet<>();
267
268            PlanNode leafPlan =
269                makeLeafPlan(leafClause, conjuncts, leafConjuncts);
270
271            JoinComponent leaf = new JoinComponent(leafPlan, leafConjuncts);
272            leafComponents.add(leaf);
273        }
274
275        return leafComponents;
276    }

319    /**
320     * This helper method builds up a full join-plan using a dynamic programming
321     * approach.  The implementation maintains a collection of optimal
322     * intermediate plans that join <em>n</em> of the leaf nodes, each with its
323     * own associated cost, and then uses that collection to generate a new
324     * collection of optimal intermediate plans that join <em>n+1</em> of the
325     * leaf nodes.  This process completes when all leaf plans are joined
326     * together; there will be <em>one</em> plan, and it will be the optimal
327     * join plan (as far as our limited estimates can determine, anyway).
328     *
329     * @param leafComponents the collection of leaf join-components, generated
330     *        by the {@link #generateLeafJoinComponents} method.
331     *
332     * @param conjuncts the collection of all conjuncts found in the query
333     *
334     * @return a single {@link JoinComponent} object that joins all leaf
335     *         components together in an optimal way.
336     */
337    private JoinComponent generateOptimalJoin(
338        ArrayList<JoinComponent> leafComponents, Set<Expression> conjuncts) {
339
340        // This object maps a collection of leaf-plans (represented as a
341        // hash-set) to the optimal join-plan for that collection of leaf plans.
342        //
343        // This collection starts out only containing the leaf plans themselves,
344        // and on each iteration of the loop below, join-plans are grown by one
345        // leaf.  For example:
346        //   * In the first iteration, all plans joining 2 leaves are created.
347        //   * In the second iteration, all plans joining 3 leaves are created.
348        //   * etc.
349        // At the end, the collection will contain ONE entry, which is the
350        // optimal way to join all N leaves.  Go Go Gadget Dynamic Programming!
351        HashMap<HashSet<PlanNode>, JoinComponent> joinPlans = new HashMap<>();
352
353        // Initially populate joinPlans with just the N leaf plans.
354        for (JoinComponent leaf : leafComponents)
355            joinPlans.put(leaf.leavesUsed, leaf);
356
357        while (joinPlans.size() > 1) {
358            logger.debug("Current set of join-plans has " + joinPlans.size() +
359                " plans in it.");
360
361            // This is the set of "next plans" we will generate.  Plans only
362            // get stored if they are the first plan that joins together the
363            // specified leaves, or if they are better than the current plan.
364            HashMap<HashSet<PlanNode>, JoinComponent> nextJoinPlans =
365                new HashMap<>();
366
367            // TODO:  IMPLEMENT THE CODE THAT GENERATES OPTIMAL PLANS THAT
368            //        JOIN N + 1 LEAVES
369
370            // Now that we have generated all plans joining N leaves, time to
371            // create all plans joining N + 1 leaves.
372            joinPlans = nextJoinPlans;
373        }
374
375        // At this point, the set of join plans should only contain one plan,
376        // and it should be the optimal plan.
377
378        assert joinPlans.size() == 1 : "There can be only one optimal join plan!";
379        return joinPlans.values().iterator().next();
380    }

383    /**
384     * Constructs a simple select plan that reads directly from a table, with
385     * an optional predicate for selecting rows.
386     * <p>
387     * While this method can be used for building up larger <tt>SELECT</tt>
388     * queries, the returned plan is also suitable for use in <tt>UPDATE</tt>
389     * and <tt>DELETE</tt> command evaluation.  In these cases, the plan must
390     * only generate tuples of type {@link edu.caltech.nanodb.storage.PageTuple},
391     * so that the command can modify or delete the actual tuple in the file's
392     * page data.
393     *
394     * @param tableName The name of the table that is being selected from.
395     *
396     * @param predicate An optional selection predicate, or {@code null} if
397     *        no filtering is desired.
398     *
399     * @return A new plan-node for evaluating the select operation.
400     *
401     * @throws IOException if an error occurs when loading necessary table
402     *         information.
403     */
404    public SelectNode makeSimpleSelect(String tableName, Expression predicate,
405        List<SelectClause> enclosingSelects) throws IOException {
406        if (tableName == null)
407            throw new IllegalArgumentException("tableName cannot be null");
408
409        if (enclosingSelects != null) {
410            // If there are enclosing selects, this subquery's predicate may
411            // reference an outer query's value, but we don't detect that here.
412            // Therefore we will probably fail with an unrecognized column
413            // reference.
414            logger.warn("Currently we are not clever enough to detect " +
415                "correlated subqueries, so expect things are about to break...");
416        }
417
418        // Open the table.
419        TableInfo tableInfo = storageManager.getTableManager().openTable(tableName);
420
421        // Make a SelectNode to read rows from the table, with the specified
422        // predicate.
423        SelectNode selectNode = new FileScanNode(tableInfo, predicate);
424        selectNode.prepare();
425        return selectNode;
426    }
