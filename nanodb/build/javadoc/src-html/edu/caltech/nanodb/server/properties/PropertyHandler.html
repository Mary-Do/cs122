001package edu.caltech.nanodb.server.properties;
002
003
004import edu.caltech.nanodb.expressions.TypeCastException;
005
006
007/**
008 * <p>
009 * This interface can be implemented by components of the database that want
010 * to expose properties that can be viewed or manipulated by the NanoDB client
011 * via the <tt>SHOW VARIABLES</tt> or <tt>SET <em>varname</em> = ...</tt>
012 * SQL commands.  The interface allows multiple properties to be handled by a
013 * single component by requiring the property name to be specified when
014 * getting or setting the property value.
015 * </p>
016 * <p>
017 * Components that want to expose one or more properties must implement this
018 * interface, and then register the specific properties on the
019 * {@code PropertyRegistry} by calling
020 * {@link PropertyRegistry#registerProperties} during initialization.
021 * </p>
022 */
023public interface PropertyHandler {
024
025    /**
026     * Retrieves the value of the specified property.
027     *
028     * @param propertyName the name of the property to retrieve
029     *
030     * @return the current value of the property
031     */
032    Object getPropertyValue(String propertyName)
033        throws UnrecognizedPropertyException;
034
035    /**
036     * Sets the value of the specified property.
037     *
038     * @param propertyName the name of the property to set
039     *
040     * @param value the new value to set the property to
041     *
042     * @throws ReadOnlyPropertyException if the specified property is
043     *         read-only and cannot be modified.
044     *
045     * @throws TypeCastException if the specified value is not the correct
046     *         type for the property, and cannot be coerced into the correct
047     *         type by the {@link edu.caltech.nanodb.expressions.TypeConverter}.
048     */
049    void setPropertyValue(String propertyName, Object value)
050        throws UnrecognizedPropertyException, ReadOnlyPropertyException,
051               TypeCastException;
052}
