005 * Created by donnie on 10/24/14.
006 */
007public enum CommandState {
008    /**
009     * This state value indicates that the server has received the command
010     * sent by the client, and is proceeding to execute it.
011     */
012    COMMAND_RECEIVED,
013
014    /**
015     * This state value indicates that the command will be generating a
016     * sequence of tuples as the result of a query.
017     */
018    BEGIN_TUPLES,
019
020    /**
021     * This state value indicates that the command has generated all tuples
022     * it will generate.
023     */
024    END_TUPLES,
025
026    /**
027     * This state value indicates that the command to be executed has been
028     * completed by the server.
029     */
030    COMMAND_COMPLETED
031}
