001package edu.caltech.nanodb.server;
002
003
004import edu.caltech.nanodb.commands.Command;
005
006
007/**
008 * This interface can be implemented by components that need to do processing
009 * before and/or after a command is executed.  Command-event listeners need to
010 * be registered on the {@link EventDispatcher} before they will be invoked.
011 */
012public interface CommandEventListener {
013    /**
014     * This method is called before a command is executed, to allow listeners to
015     * perform before-command processing.
016     *
017     * @param cmd the command about to be executed
018     */
019    void beforeCommandExecuted(Command cmd) throws EventDispatchException;
020
021
022    /**
023     * This method is called after a command is executed, to allow listeners to
024     * perform after-command processing.
025     *
026     * @param cmd the command that was just executed
027     */
028    void afterCommandExecuted(Command cmd) throws EventDispatchException;
029}
