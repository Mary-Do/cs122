011/**
012 * This class implements simple binary arithmetic operations.  The supported
013 * operations are:
014 * <ul>
015 *   <li>addition, <tt>+</tt></li>
016 *   <li>subtraction, <tt>-</tt></li>
017 *   <li>multiplication, <tt>*</tt></li>
018 *   <li>division, <tt>/</tt></li>
019 *   <li>remainder, <tt>%</tt></li>
020 *   <li>exponentiation, <tt>^</tt></li>
021 * </ul>
022 */
023public class ArithmeticOperator extends Expression {
024
025    /**
026     * This enum specifies the arithmetic operations that this class can provide.
027     * Each arithmetic operation also holds its string representation, which is
028     * used when converting an arithmetic expression into a string for display.
029     */
030    public enum Type {
031        ADD("+"),
032        SUBTRACT("-"),
033        MULTIPLY("*"),
034        DIVIDE("/"),
035        REMAINDER("%"),
036        POWER("^");
037
038
039        /** The string representation for each operator.  Used for printing. */
040        private final String stringRep;
041
042        /**
043         * Construct a Type enum with the specified string representation.
044         *
045         * @param rep the string representation of the arithmetic operation
046         */
047        Type(String rep) {
048            stringRep = rep;
049        }
050
051        /**
052         * Accessor for the operator type's string representation.
053         *
054         * @return the string representation of the arithmetic operation
055         */
056        public String stringRep() {
057            return stringRep;
058        }
059    }
060
061
062    /** The kind of comparison, such as "subtract" or "multiply." */
063    Type type;
064
065    /** The left expression in the comparison. */
066    Expression leftExpr;
067
068    /** The right expression in the comparison. */
069    Expression rightExpr;
070
071
072
073    public ArithmeticOperator(Type type, Expression lhs, Expression rhs) {
074        if (type == null || lhs == null || rhs == null)
075            throw new NullPointerException();
076
077        leftExpr = lhs;
078        rightExpr = rhs;
079
080        this.type = type;
081    }

084    @Override
085    public ColumnInfo getColumnInfo(Schema schema) throws SchemaNameException {
086        ColumnInfo ltColInfo = leftExpr.getColumnInfo(schema);
087        ColumnInfo rtColInfo = rightExpr.getColumnInfo(schema);
088
089        SQLDataType resultSQLType = getSQLResultType(
090            ltColInfo.getType().getBaseType(), rtColInfo.getType().getBaseType());
091
092        ColumnType colType = new ColumnType(resultSQLType);
093        return new ColumnInfo(toString(), colType);
094    }

096
097    private SQLDataType getSQLResultType(SQLDataType lType, SQLDataType rType) {
098        // In case of division, we always return a double-precision result.
099        if (type == Type.DIVIDE)
100            return SQLDataType.DOUBLE;
101
102        // This array specifies the type-conversion sequence.  If at least one of
103        // the arguments is type typeOrder[i], then both arguments are coerced to
104        // that type.  (This is not entirely accurate at the moment, but is
105        // perfectly sufficient for our needs.)
106        SQLDataType[] typeOrder = {
107            SQLDataType.NUMERIC, SQLDataType.DOUBLE, SQLDataType.FLOAT,
108            SQLDataType.BIGINT, SQLDataType.INTEGER, SQLDataType.SMALLINT,
109            SQLDataType.TINYINT
110        };
111
112        for (SQLDataType aTypeOrder : typeOrder) {
113            if (lType == aTypeOrder || rType == aTypeOrder)
114                return aTypeOrder;
115        }
116
117        // Just guess INTEGER.  Works for C...
118        return SQLDataType.INTEGER;
119    }

122    public Object evaluate(Environment env) throws ExpressionException {
123        // Evaluate the left and right subexpressions.
124        Object lhsValue = leftExpr.evaluate(env);
125        Object rhsValue = rightExpr.evaluate(env);
126
127        // If either the LHS value or RHS value is NULL (represented by Java
128        // null value) then the entire expression evaluates to NULL.
129        if (lhsValue == null || rhsValue == null)
130            return null;
131
132        return evalObjects(type, lhsValue, rhsValue);
133    }

136    /**
137     * This static helper method can be used to compute basic arithmetic
138     * operations between two arguments.  It is of course used to evaluate
139     * <tt>ArithmeticOperator</tt> objects, but it can also be used to evaluate
140     * specific arithmetic operations within other components of the database
141     * system.
142     *
143     * @param type the arithmetic operation to perform
144     * @param aObj the first operand value for the operation
145     * @param bObj the second operand value for the operation
146     *
147     * @return the result of the arithmetic operation
148     *
149     * @throws ExpressionException if the operand type is unrecognized
150     */
151    public static Object evalObjects(Type type, Object aObj, Object bObj) {
152        // Coerce the values to both have the same numeric type.
153
154        TypeConverter.Pair coerced = TypeConverter.coerceArithmetic(aObj, bObj);
155
156        Object result;
157
158        if (coerced.value1 instanceof Double) {
159            result = evalDoubles(type, (Double) coerced.value1, (Double) coerced.value2);
160        }
161        else if (coerced.value1 instanceof Float) {
162            result = evalFloats(type, (Float) coerced.value1, (Float) coerced.value2);
163        }
164        else if (coerced.value1 instanceof Long) {
165            result = evalLongs(type, (Long) coerced.value1, (Long) coerced.value2);
166        }
167        else {
168            assert coerced.value1 instanceof Integer;
169            result = evalIntegers(type, (Integer) coerced.value1, (Integer) coerced.value2);
170        }
171
172        return result;
173    }

176    /**
177     * This helper implements the arithmetic operations for <tt>Double</tt>
178     * values.  Note that division of two <tt>Double</tt>s will produce a
179     * <tt>Double</tt>.
180     *
181     * @param type the arithmetic operation to perform
182     * @param aObj the first operand value for the operation
183     * @param bObj the second operand value for the operation
184     *
185     * @return the result of the arithmetic operation
186     *
187     * @throws ExpressionException if the operand type is unrecognized
188     */
189    private static Double evalDoubles(Type type, Double aObj, Double bObj) {
190        double a = aObj.doubleValue();
191        double b = bObj.doubleValue();
192        double result;
193
194        switch (type) {
195        case ADD:
196            result = a + b;
197            break;
198
199        case SUBTRACT:
200            result = a - b;
201            break;
202
203        case MULTIPLY:
204            result = a * b;
205            break;
206
207        case DIVIDE:
208            // TODO:  How to handle divide-by-zero?
209            result = a / b;
210            break;
211
212        case REMAINDER:
213            result = a % b;
214            break;
215
216        case POWER:
217            // TODO:  How to handle 0^0?
218            result = Math.pow(a, b);
219            break;
220
221        default:
222            throw new ExpressionException("Unrecognized arithmetic type " + type);
223        }
224
225        return Double.valueOf(result);
226    }

396    @Override
397    public Expression traverse(ExpressionProcessor p) {
398        p.enter(this);
399        leftExpr = leftExpr.traverse(p);
400        rightExpr = rightExpr.traverse(p);
401        return p.leave(this);
402    }

405    /**
406     * Returns a string representation of this arithmetic expression and its
407     * subexpressions, including parentheses where necessary to specify
408     * precedence.
409     */
410    @Override
411    public String toString() {
412        // Convert all of the components into string representations.
413        String leftStr = leftExpr.toString();
414        String rightStr = rightExpr.toString();
415        String opStr = " " + type.stringRep() + " ";
416
417        // Figure out if I need parentheses around the subexpressions.
418
419        if (type == Type.MULTIPLY || type == Type.DIVIDE || type == Type.REMAINDER) {
420            if (leftExpr instanceof ArithmeticOperator) {
421                ArithmeticOperator leftOp = (ArithmeticOperator) leftExpr;
422                if (leftOp.type == Type.ADD || leftOp.type == Type.SUBTRACT)
423                    leftStr = "(" + leftStr + ")";
424            }
425
426            if (rightExpr instanceof ArithmeticOperator) {
427                ArithmeticOperator rightOp = (ArithmeticOperator) rightExpr;
428                if (rightOp.type == Type.ADD || rightOp.type == Type.SUBTRACT)
429                    rightStr = "(" + rightStr + ")";
430            }
431        }
432
433        return leftStr + opStr + rightStr;
434    }

437    /**
438     * Simplifies an arithmetic expression, computing as much of the expression
439     * as possible.
440     */
441    public Expression simplify() {
442        leftExpr = leftExpr.simplify();
443        rightExpr = rightExpr.simplify();
444
445        if (!leftExpr.hasSymbols())
446            leftExpr = new LiteralValue(leftExpr.evaluate());
447
448        if (!rightExpr.hasSymbols())
449            rightExpr = new LiteralValue(rightExpr.evaluate());
450
451        if (!hasSymbols())
452            return new LiteralValue(evaluate());
453
454        return this;
455    }

458    /**
459     * Checks if the argument is an expression with the same structure, but not
460     * necesarily the same references.
461     *
462     * @param obj the object to which we are comparing
463     */
464    @Override
465    public boolean equals(Object obj) {
466        if (obj instanceof ArithmeticOperator) {
467            ArithmeticOperator other = (ArithmeticOperator) obj;
468            return (type.equals(other.type) &&
469                    leftExpr.equals(other.leftExpr) &&
470                    rightExpr.equals(other.rightExpr));
471        }
472        return false;
473    }

476    /**
477     * Computes the hashcode of an Expression.  This method is used to see if
478     * two expressions CAN be equal.
479     */
480    @Override
481    public int hashCode() {
482        int hash = 7;
483
484        hash = 31 * hash + type.hashCode();
485
486        hash = 31 * hash + leftExpr.hashCode();
487        hash = 31 * hash + rightExpr.hashCode();
488
489        return hash;
490    }

493    /** Creates a copy of expression. */
494    @Override
495    protected Object clone() throws CloneNotSupportedException {
496        ArithmeticOperator expr = (ArithmeticOperator)super.clone();
497
498        // Type is immutable, copy it.
499        expr.type = this.type;
500
501        // Clone the subexpressions
502        expr.leftExpr = (Expression) leftExpr.clone();
503        expr.rightExpr = (Expression) rightExpr.clone();
504
505        return expr;
506    }
