010/**
011 * This class provides a whole bunch of helper functions for performing type
012 * conversions on values produced by various expressions.  All values are passed
013 * around as <code>Object</code> references, and this means they may need to be
014 * converted into specific value types at some point.  This class contains all
015 * of the conversion logic so it's in one place.
016 *
017 * @todo Update these methods to support <tt>NULL</tt> (i.e. Java <tt>null</tt>)
018 *       values.  Right now we just throw a type-cast exception.
019 */
020public class TypeConverter {
021
022    /**
023     * A mapping from the various Java types used for expression results, to their
024     * corresponding SQL data types.  The mapping is populated by a static
025     * initializer block.
026     */
027    private static HashMap<Class, SQLDataType> sqlTypeMapping;
028
029    static {
030        sqlTypeMapping = new HashMap<Class, SQLDataType>();
031
032        sqlTypeMapping.put(Null.class, SQLDataType.NULL);
033
034        sqlTypeMapping.put(Boolean.class, SQLDataType.TINYINT);
035
036        sqlTypeMapping.put(Byte.class, SQLDataType.TINYINT);
037        sqlTypeMapping.put(Short.class, SQLDataType.SMALLINT);
038        sqlTypeMapping.put(Integer.class, SQLDataType.INTEGER);
039        sqlTypeMapping.put(Long.class, SQLDataType.BIGINT);
040
041        sqlTypeMapping.put(Float.class, SQLDataType.FLOAT);
042        sqlTypeMapping.put(Double.class, SQLDataType.DOUBLE);
043
044        sqlTypeMapping.put(String.class, SQLDataType.VARCHAR);
045
046        // TODO:  others, in time...
047    }

050    /**
051     * A simple wrapper-class for holding two values.  This is used to return
052     * results from the coercion functions, since they take two inputs and must
053     * coerce them into being the same types.
054     */
055    public static class Pair {
056        /** The first value of the pair. */
057        public Object value1;
058
059        /** The second value of the pair. */
060        public Object value2;
061
062        /**
063         * Construct a new pair using the specified values.
064         *
065         * @param obj1 the first value to store
066         * @param obj2 the second value to store
067         */
068        public Pair(Object obj1, Object obj2) {
069            value1 = obj1;
070            value2 = obj2;
071        }
072    }

075    /**
076     * This method attempts to convert the input value into a
077     * {@link java.lang.Boolean} value.  If the input is a nonzero number then
078     * the result is {@link java.lang.Boolean#TRUE}; if it is a zero number then
079     * the result is {@link java.lang.Boolean#FALSE}.  Otherwise a
080     * {@link TypeCastException} is thrown.
081     *
082     * @param obj the input value to cast
083     *
084     * @return the input value cast to a <tt>Boolean</tt>
085     *
086     * @throws TypeCastException if the input value cannot be cast to
087     *         a Boolean value.
088     */
089    public static Boolean getBooleanValue(Object obj) {
090        if (obj == null)
091            return null;
092
093        Boolean result;
094
095        if (obj instanceof Boolean) {
096            result = (Boolean) obj;
097        }
098        else if (obj instanceof Number) {
099            // If it's a nonzero number, return TRUE.  Otherwise, FALSE.
100            Number num = (Number) obj;
101            result = Boolean.valueOf(num.intValue() != 0);
102        }
103        else {
104            throw new TypeCastException("Cannot convert type " +
105                obj.getClass() + " to byte.");
106        }
107
108        return result;
109    }

112    /**
113     * This method attempts to convert the input value into a
114     * {@link java.lang.Byte} value.  If the input is a number then the result
115     * is generated from the {@link java.lang.Number#byteValue} method, possibly
116     * causing truncation or overflow to occur.  If the input is a
117     * {@link java.lang.String} that can be parsed into a byte then the result
118     * is the parsed value.  If none of these cases hold then a
119     * {@link TypeCastException} is thrown.
120     *
121     * @param obj the input value to cast
122     *
123     * @return the input value cast to a <tt>Byte</tt>
124     *
125     * @throws TypeCastException if the input value cannot be cast to a byte.
126     */
127    public static Byte getByteValue(Object obj) {
128        if (obj == null)
129            return null;
130
131        Byte result;
132
133        if (obj instanceof Byte) {
134            result = (Byte) obj;
135        }
136        else if (obj instanceof Number) {
137            // TODO:  Would be nice to detect overflow or truncation issues and
138            //        log warnings about them!
139
140            Number num = (Number) obj;
141            result = Byte.valueOf(num.byteValue());
142        }
143        else if (obj instanceof String) {
144            try {
145                result = Byte.valueOf((String) obj);
146            }
147            catch (NumberFormatException nfe) {
148                throw new TypeCastException("Cannot convert string to byte.", nfe);
149            }
150        }
151        else {
152            throw new TypeCastException("Cannot convert type \"" +
153                obj.getClass() + "\" to byte.");
154        }
155
156        return result;
157    }

414    /**
415     * This function takes two arguments and coerces them to be the same numeric
416     * type, for use with arithmetic operations.
417     * <p>
418     * If either or both of the arguments are <tt>null</tt> then no coercion is
419     * performed, and the results are returned in a pair-object.  The reason for
420     * this is that the comparison will simply evaluate to <tt>UNKNOWN</tt>, so
421     * no coercion is required.
422     *
423     * @param obj1 the first input value to cast
424     * @param obj2 the second input value to cast
425     *
426     * @return an object holding the two input values, both converted to a type
427     *         suitable for arithmetic
428     *
429     * @throws TypeCastException if both inputs are non-<tt>null</tt> and they
430     *         cannot both be cast to types suitable for arithmetic.
431     */
432    public static Pair coerceArithmetic(Object obj1, Object obj2)
433        throws TypeCastException {
434
435        if (obj1 != null && obj2 != null) {
436            if (obj1 instanceof Number || obj2 instanceof Number) {
437                if (obj1 instanceof Double || obj2 instanceof Double) {
438                    // At least one is a Double, so convert both to Doubles.
439                    obj1 = getDoubleValue(obj1);
440                    obj2 = getDoubleValue(obj2);
441                }
442                else if (obj1 instanceof Float || obj2 instanceof Float) {
443                    // At least one is a Float, so convert both to Floats.
444                    obj1 = getFloatValue(obj1);
445                    obj2 = getFloatValue(obj2);
446                }
447                else if (obj1 instanceof Long || obj2 instanceof Long) {
448                    // At least one is a Long, so convert both to Longs.
449                    obj1 = getLongValue(obj1);
450                    obj2 = getLongValue(obj2);
451                }
452                else {
453                    // Any other integer-type we will just coerce into being Integers.
454                    obj1 = getIntegerValue(obj1);
455                    obj2 = getIntegerValue(obj2);
456                }
457            }
458            else {
459                throw new TypeCastException(String.format(
460                    "Cannot coerce types \"%s\" and \"%s\" for arithmetic.",
461                    obj1.getClass(), obj2.getClass()));
462            }
463
464            assert obj1.getClass().equals(obj2.getClass());
465        }
466
467        return new Pair(obj1, obj2);
468    }

471    /**
472     * This function takes two arguments and coerces them to be the same numeric
473     * type, for use with comparison operations.  It is up to the caller to
474     * ensure that the types are actually comparable (although all recognized
475     * types in this function are comparable).
476     * <p>
477     * If either or both of the arguments are <tt>null</tt> then no coercion is
478     * performed, and the results are returned in a pair-object.  The reason for
479     * this is that the comparison will simply evaluate to <tt>UNKNOWN</tt>, so
480     * no coercion is required.
481     *
482     * @design When new types are added in the future (e.g. date and/or time
483     *         values), this function will need to be updated with the new types.
484     *
485     * @param obj1 the first input value to cast
486     * @param obj2 the second input value to cast
487     *
488     * @return an object holding the two input values, both converted to a type
489     *         suitable for comparison
490     *
491     * @throws TypeCastException if both inputs are non-<tt>null</tt> and they
492     *         cannot both be cast to types suitable for comparison.
493     */
494    public static Pair coerceComparison(Object obj1, Object obj2)
495        throws TypeCastException {
496
497        if (obj1 != null && obj2 != null) {
498            if (obj1.getClass().equals(obj2.getClass())) {
499                // The two objects are already of the same type, so no need
500                // to coerce anything.
501            }
502            else if (obj1 instanceof Number || obj2 instanceof Number) {
503                // Reuse the same logic in the arithmetic coercion code.
504                return coerceArithmetic(obj1, obj2);
505            }
506            else if (obj1 instanceof Boolean || obj2 instanceof Boolean) {
507                obj1 = getBooleanValue(obj1);
508                obj2 = getBooleanValue(obj2);
509            }
510            else if (obj1 instanceof String || obj2 instanceof String) {
511                obj1 = getStringValue(obj1);
512                obj2 = getStringValue(obj2);
513            }
514            else {
515                // Inputs are different types, and we don't know how to make
516                // them the same.
517                throw new TypeCastException(String.format(
518                    "Cannot coerce types \"%s\" and \"%s\" for comparison.",
519                    obj1.getClass(), obj2.getClass()));
520            }
521
522            assert obj1.getClass().equals(obj2.getClass());
523        }
524
525        return new Pair(obj1, obj2);
526    }

529    /**
530     * This method attempts to assign a SQL data type to a value produced by the
531     * expression classes.  If a Java type is not recognized as a particular SQL
532     * data type then <tt>null</tt> is returned.
533     *
534     * @param obj The object to determine a SQL data type for.
535     *
536     * @return The corresponding SQL data type, or <tt>null</tt> if the input
537     *         value's type doesn't have an obvious corresponding SQL data type.
538     */
539    public static SQLDataType getSQLType(Object obj) {
540        if (obj == null)
541            return sqlTypeMapping.get(Null.class);
542
543        return sqlTypeMapping.get(obj.getClass());
544    }

547    /**
548     * This method attempts to create a
549     * {@link edu.caltech.nanodb.relations.ColumnType} object for a value
550     * produced by the expression classes.  If the Java type is not recognized
551     * as a particular SQL data type then <tt>null</tt> is returned.  If the
552     * corresponding SQL data type has a length or precision, this method sets
553     * that information appropriately on the <tt>ColumnType</tt> object.
554     * <p>
555     * This method uses the {@link #getSQLType} method to determine the SQL data
556     * type, so the two methods are consistent with each other.
557     *
558     * @param obj The object to create a <tt>ColumnType</tt> object for.
559     *
560     * @return The corresponding <tt>ColumnType</tt> object, or <tt>null</tt> if
561     *         the input value's type doesn't have an obvious corresponding SQL
562     *         data type.
563     */
564    public static ColumnType getColumnType(Object obj) {
565        ColumnType colType = null;
566
567        SQLDataType sqlType = getSQLType(obj);
568        if (sqlType != null) {
569            colType = new ColumnType(sqlType);
570
571            if (sqlType == SQLDataType.VARCHAR)
572                colType.setLength(obj.toString().length());
573        }
574
575        return colType;
576    }
