016/**
017 * This class provides the standard Boolean logical operators AND, OR, and NOT,
018 * for two or more terms (or exactly one term, in the case of NOT).  In
019 * addition, there are several methods provided for manipulating Boolean
020 * expressions, since many plan equivalence-rules involve manipulations of the
021 * Boolean logical expressions associated with Select and Theta-Join plan nodes.
022 */
023public class BooleanOperator extends Expression {
024
025    /**
026     * This enumeration specifies the different kinds of Boolean operators that
027     * this class can implement.
028     */
029    public enum Type {
030        AND_EXPR("AND"),
031        OR_EXPR("OR"),
032        NOT_EXPR("NOT");
033
034        /** The string representation for each operator.  Used for printing. **/
035        private final String stringRep;
036
037        /**
038         * Construct a Type enum with the specified string representation.
039         *
040         * @param rep the string representation of the type
041         */
042        Type(String rep) {
043            stringRep = rep;
044        }
045
046        /**
047         * Accessor for the operator type's string representation.
048         *
049         * @return the string representation of the operator type
050         */
051        public String stringRep() {
052            return stringRep;
053        }
054    }
055
056
057    /** Specifies the type of this Boolean operator. */
058    private Type type;
059
060
061    /**
062     * The list of one or more terms in this expression.  If the expression type
063     * is {@link BooleanOperator.Type#NOT_EXPR} then this will contain exactly
064     * one term.
065     */
066    private ArrayList<Expression> terms;
067
068
069    public BooleanOperator(Type type, Collection<Expression> terms) {
070
071        if (type == null)
072            throw new NullPointerException("type cannot be null");
073
074        this.type = type;
075        this.terms = new ArrayList<Expression>();
076
077        if (terms != null) {
078            for (Expression term : terms)
079                addTerm(term);
080        }
081    }
082
083
084    public BooleanOperator(Type type) {
085        this(type, null);
086    }

099    public void addTerm(Expression term) {
100        if (term == null)
101            throw new NullPointerException("term cannot be null");
102
103        terms.add(term);
104    }

107    public Expression getTerm(int i) {
108        return terms.get(i);

112    public ColumnInfo getColumnInfo(Schema schema) throws SchemaNameException {
113        // Boolean operators always return Boolean values, so just pass a
114        // Boolean value in to the TypeConverter to get out the corresponding
115        // SQL datatype.
116        ColumnType colType =
117            new ColumnType(TypeConverter.getSQLType(Boolean.FALSE));
118
119        return new ColumnInfo(colType);
120    }

123    public Object evaluate(Environment env) throws ExpressionException {
124
125        if (terms.size() == 0) {
126            throw new ExpressionException("Boolean " + type +
127                " expression contains no terms!");
128        }
129
130        Object objResult;
131        boolean boolResult;
132
133        if (type == Type.NOT_EXPR) {
134            if (terms.size() != 1) {
135                throw new ExpressionException(
136                    "NOT-expressions may have only one term.");
137            }
138
139            // Evaluate the term...
140            objResult = terms.get(0).evaluate(env);
141            if (objResult == null)
142                boolResult = false;  // TODO:  this is UNKNOWN, not FALSE.
143            else
144                boolResult = TypeConverter.getBooleanValue(objResult);
145
146            // ...then negate it.
147            boolResult = !boolResult;
148        }
149        else {
150            // AND/OR expression.
151
152            if (terms.size() < 1) {
153                throw new ExpressionException(type.stringRep() +
154                    "-expressions must have at least one term.");
155            }
156
157            if (type == Type.AND_EXPR) {
158                boolResult = true;
159            }
160            else {
161                assert type == Type.OR_EXPR : "Unexpected type value " + type + "!";
162                boolResult = false;
163            }
164
165            for (Expression term : terms) {
166                // Evaluate the i-th term, and combine it with the current answer.
167                objResult = term.evaluate(env);
168
169                boolean termValue;
170                if (objResult == null)
171                    termValue = false;  // TODO:  this is UNKNOWN, not FALSE.
172                else
173                    termValue = TypeConverter.getBooleanValue(objResult);
174
175                if (type == Type.AND_EXPR && !termValue) {
176                    // AND term is false, so we are done.  Answer is false.
177                    boolResult = false;
178                    break;
179                }
180                else if (type == Type.OR_EXPR && termValue) {
181                    // OR term is true, so we are done.  Answer is true.
182                    boolResult = true;
183                    break;
184                }
185            }
186        }
187
188        return Boolean.valueOf(boolResult);
189    }

192    @Override
193    public Expression traverse(ExpressionProcessor p) {
194        p.enter(this);
195
196        for (int i = 0; i < terms.size(); i++) {
197            Expression e = terms.get(i).traverse(p);
198            terms.set(i, e);
199        }
200
201        return p.leave(this);
202    }

205    /**
206     * Returns a string representation of this Boolean logical expression and
207     * its subexpressions, including parentheses where necessary to specify
208     * precedence.
209     */
210    @Override
211    public String toString() {
212        // Convert all of the components into string representations.
213
214        StringBuilder buf = new StringBuilder();
215
216        if (type == Type.NOT_EXPR) {
217            assert terms.size() == 1 : "NOT expressions must have exactly one term";
218
219            buf.append(type.stringRep());
220            buf.append(' ');
221
222            Expression term = terms.get(0);
223            String termStr = term.toString();
224            if (term instanceof BooleanOperator) {
225                BooleanOperator termOp = (BooleanOperator) term;
226                if (termOp.type != Type.NOT_EXPR)
227                    termStr = '(' + termStr + ')';
228            }
229
230            buf.append(termStr);
231        }
232        else if (type == Type.AND_EXPR) {
233            assert terms.size() >= 1 : "AND expressions must have at least one term";
234
235            boolean first = true;
236                for (Expression term : terms) {
237                if (first)
238                    first = false;
239                else
240                    buf.append(' ').append(type.stringRep()).append(' ');
241
242                String termStr = term.toString();
243
244                if (term instanceof BooleanOperator) {
245                    BooleanOperator termOp = (BooleanOperator) term;
246                    if (termOp.type == Type.OR_EXPR)
247                        termStr = '(' + termStr + ')';
248                }
249
250                buf.append(termStr);
251            }
252        }
253        else {  // OR_EXPR
254            assert type == Type.OR_EXPR;
255            assert terms.size() >= 1 : "OR expressions must have at least one term";
256
257            boolean first = true;
258            for (Expression term : terms) {
259                if (first)
260                    first = false;
261                else
262                    buf.append(' ').append(type.stringRep()).append(' ');
263
264                buf.append(term.toString());
265            }
266        }
267
268        return buf.toString();
269    }

272    /**
273     * This method returns true if this Boolean expression contains any terms
274     * that reference the exact set of tables specified in the arguments.  See
275     * the documentation for {@link #getTermsReferencingAllTables} for more
276     * details on what is considered to be a "matching" term by this method.
277     *
278     * @design This method is <em>slightly</em> more efficient than the
279     *         get/remove methods, since it stops after it has found the first
280     *         term that satisfies the specified conditions.  However, if there
281     *         will be a subsequent call to the get or remove method, it's
282     *         probably most efficient to just call it directly.
283     *
284     * @param tableNames a collection of table-names to look for in the terms.
285     *
286     * @return A list of terms that reference all of the tables specified in the
287     *         input list, and no others.
288     */
289    public boolean hasTermsReferencingAllTables(String... tableNames) {
290
291        // Put table names into a set so we can easily do membership tests.
292        HashSet<String> inputTables = new HashSet<String>();
293        for (String tableName : tableNames)
294            inputTables.add(tableName);
295
296        // Iterate through each term in this Boolean operator.  For each term,
297        // find out what symbols the term contains.  Then, see if the term only
298        // references the tables in the input set.
299        HashSet<ColumnName> symbols = new HashSet<ColumnName>();
300        HashSet<String> exprTables = new HashSet<String>();
301        for (Expression term : terms) {
302            symbols.clear();
303            exprTables.clear();
304
305            term.getAllSymbols(symbols);
306            for (ColumnName colName : symbols) {
307                if (colName.isTableSpecified())
308                    exprTables.add(colName.getTableName());
309            }
310
311            if (inputTables.equals(exprTables))
312                return true;
313        }
314
315        return false;
316    }

319    /**
320     * This method finds and returns a list of all terms in this Boolean
321     * expression that reference the exact set of tables specified in the
322     * arguments.  If a term references other tables outside of this set then it
323     * will not be returned.  If a term doesn't reference some table in this set
324     * then it will not be returned.
325     *
326     * @param tableNames a collection of table-names to look for in the terms.
327     *
328     * @return A list of terms that reference all of the tables specified in the
329     *         input list, and no others.
330     */
331    public List<Expression> getTermsReferencingAllTables(String... tableNames) {
332        // Call the helper, with "remove" flag set to false.
333        return _getTermsReferencingAllTables(tableNames, false);
334    }
335
336
337    /**
338     * This method finds, removes, and returns a list of all terms in this
339     * Boolean expression that reference the exact set of tables specified in
340     * the arguments.  If a term references other tables outside of this set
341     * then it will not be returned.  If a term doesn't reference some table in
342     * this set then it will not be returned.
343     * <p>
344     * The sole difference between this and {@link #getTermsReferencingAllTables}
345     * is that this method also removes the found terms from this Boolean
346     * expression object.
347     *
348     * @param tableNames a collection of table-names to look for in the terms.
349     *
350     * @return A list of terms that reference all of the tables specified in the
351     *         input list, and no others.  These terms are also removed from the
352     *         Boolean expression object.
353     */
354    public List<Expression> removeTermsReferencingAllTables(String... tableNames) {
355        // Call the helper, with "remove" flag set to true.
356        return _getTermsReferencingAllTables(tableNames, true);
357    }

360    /**
361     * This is a private helper function used by both the
362     * {@link #getTermsReferencingAllTables} and
363     * {@link #removeTermsReferencingAllTables} methods.
364     *
365     * @param tableNames an array of table-names to look for in the terms.
366     * @param remove if <tt>true</tt> then matching terms will be removed from
367     *        this Boolean operator's list of terms.
368     *
369     * @return A list of terms that reference all of the tables specified in the
370     *         input list, and no others.
371     */
372    private List<Expression> _getTermsReferencingAllTables(String[] tableNames,
373        boolean remove) {
374
375        ArrayList<Expression> found = new ArrayList<Expression>();
376
377        // Put table names into a set so we can easily do membership tests.
378        HashSet<String> inputTables = new HashSet<String>();
379        for (String tableName : tableNames)
380            inputTables.add(tableName);
381
382        // Iterate through each term in this Boolean operator.  For each term,
383        // find out what symbols the term contains.  Then, see if the term only
384        // references the tables in the input set.
385        HashSet<ColumnName> symbols = new HashSet<ColumnName>();
386        HashSet<String> exprTables = new HashSet<String>();
387        Iterator<Expression> termIter = terms.iterator();
388        while (termIter.hasNext()) {
389            Expression term = termIter.next();
390
391            symbols.clear();
392            exprTables.clear();
393
394            term.getAllSymbols(symbols);
395            for (ColumnName colName : symbols) {
396                if (colName.isTableSpecified())
397                    exprTables.add(colName.getTableName());
398            }
399
400            if (inputTables.equals(exprTables)) {
401                found.add(term);
402                if (remove)
403                    termIter.remove();
404            }
405        }
406
407        return found;
408    }

411    /**
412     * Returns the number of terms in the boolean expression.
413     *
414     * @return the number of terms in the boolean expression.
415     */
416    public int getNumTerms() {
417        return terms.size();
418    }
419
420
421    /**
422     * Removes the i<sup>th</sup> term, starting from 0.
423     *
424     * @param i the index of the term to remove.
425     *
426     * @throws IllegalArgumentException if the specified index is invalid.
427     */
428    public void removeTerm(int i) {
429        if (i < 0 || i >= getNumTerms()) {
430            throw new IllegalArgumentException("Term-index " + i +
431                " is out of range [0, " + getNumTerms() + ")");
432        }
433
434        terms.remove(i);
435    }

438    /**
439     * Performs a value-eqality test for whether the specified object is an
440     * expression with the same structure and contents.
441     *
442     * @param obj the object to which we are comparing
443     */
444    @Override
445    public boolean equals(Object obj) {
446        if (obj instanceof BooleanOperator) {
447            BooleanOperator other = (BooleanOperator) obj;
448    
449            return (type == other.type && terms.equals(other.terms));
450        }
451        return false;
452    }

455    /**
456     * Computes the hash-code of an Expression.
457     */
458    @Override
459    public int hashCode() {
460        int hash = 7;
461        hash = 31 * hash + type.hashCode();
462        hash = 31 * hash + terms.hashCode();
463        return hash;
464    }

468     * Creates a copy of expression.
469     */
470    @Override
471    @SuppressWarnings("unchecked")
472    protected Object clone() throws CloneNotSupportedException {
473        BooleanOperator expr = (BooleanOperator) super.clone();
474
475        // Type is immutable, copy it.
476        expr.type = this.type;
477
478        expr.terms = (ArrayList<Expression>) terms.clone();
479
