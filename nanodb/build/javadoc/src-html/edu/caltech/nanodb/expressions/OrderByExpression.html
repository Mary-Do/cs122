001package edu.caltech.nanodb.expressions;
002
003
004/**
005 * This class represents an expression that results are ordered by, as well as
006 * whether the order is ascending or descending.  Lists of this object specify
007 * ordering of results from a query, as well as the ordering of tuples in a
008 * sequential file, or attributes in an ordered index.
009 */
010public class OrderByExpression {
011
012    /** The expression that the results should be ordered by. */
013    private Expression expression;
014
015    /**
016     * A flag indicating whether the results should be in ascending or
017     * descending order.  A value of <tt>true</tt> means ascending order.
018     */
019    private boolean ascending;
020
021
022    /**
023     * Initialize a new order-by object with the specified expression.  Results
024     * will be in ascending order.
025     *
026     * @param expression The expression to order results by.
027     */
028    public OrderByExpression(Expression expression) {
029        this(expression, true);
030    }
031
032
033    /**
034     * Initialize a new order-by object with the specified expression and order.
035     *
036     * @param expression The expression to order results by.
037     *
038     * @param ascending a value of <tt>true</tt> will cause the results to be
039     *        in ascending order; a value of <tt>false</tt> will cause them to
040     *        be in descending order
041     */
042    public OrderByExpression(Expression expression, boolean ascending) {
043        this.expression = expression;
044        this.ascending = ascending;
045    }
046
047
048    /**
049     * Returns the expression that results will be sorted on
050     *
051     * @return the expression that results will be sorted on
052     */
053    public Expression getExpression() {
054        return expression;
055    }
056
057
058    /**
059     * Returns the flag indicating whether the results should be in ascending or
060     * descending order.
061     *
062     * @return <tt>true</tt> if the results should be in ascending order,
063     *         <tt>false</tt> otherwise.
064     */
065    public boolean isAscending() {
066        return ascending;
067    }
068
069
070    @Override
071    public String toString() {
072        return expression.toString() + ' ' + (ascending ? "ASC" : "DESC");
073    }
074}
